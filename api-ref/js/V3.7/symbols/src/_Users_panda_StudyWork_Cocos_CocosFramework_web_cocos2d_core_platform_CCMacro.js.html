 71 /**
 72  * <p>
 73  * simple macro that swaps 2 variables<br/>
 74  *  modified from c++ macro, you need to pass in the x and y variables names in string, <br/>
 75  *  and then a reference to the whole object as third variable
 76  * </p>
 77  * @param {String} x
 78  * @param {String} y
 79  * @param {Object} ref
 80  * @function
 81  * @deprecated since v3.0
 82  */
 83 cc.swap = function (x, y, ref) {
 84     if (cc.isObject(ref) && !cc.isUndefined(ref.x) && !cc.isUndefined(ref.y)) {
 85         var tmp = ref[x];
 86         ref[x] = ref[y];
 87         ref[y] = tmp;
 88     } else
 89         cc.log(cc._LogInfos.swap);
 90 };

 92 /**
 93  * <p>
 94  *     Linear interpolation between 2 numbers, the ratio sets how much it is biased to each end
 95  * </p>
 96  * @param {Number} a number A
 97  * @param {Number} b number B
 98  * @param {Number} r ratio between 0 and 1
 99  * @function
100  * @example
101  * cc.lerp(2,10,0.5)//returns 6<br/>
102  * cc.lerp(2,10,0.2)//returns 3.6
103  */
104 cc.lerp = function (a, b, r) {
105     return a + (b - a) * r;
106 };

761 /**
762  * Verify Array's Type
763  * @param {Array} arr
764  * @param {function} type
765  * @return {Boolean}
766  * @function
767  */
768 cc.arrayVerifyType = function (arr, type) {
769     if (arr && arr.length > 0) {
770         for (var i = 0; i < arr.length; i++) {
771             if (!(arr[i] instanceof  type)) {
772                 cc.log("element type is wrong!");
773                 return false;
774             }
775         }
776     }
777     return true;
778 };

780 /**
781  * Searches for the first occurance of object and removes it. If object is not found the function has no effect.
782  * @function
783  * @param {Array} arr Source Array
784  * @param {*} delObj  remove object
785  */
786 cc.arrayRemoveObject = function (arr, delObj) {
787     for (var i = 0, l = arr.length; i < l; i++) {
788         if (arr[i] === delObj) {
789             arr.splice(i, 1);
790             break;
791         }
792     }
793 };

795 /**
796  * Removes from arr all values in minusArr. For each Value in minusArr, the first matching instance in arr will be removed.
797  * @function
798  * @param {Array} arr Source Array
799  * @param {Array} minusArr minus Array
800  */
801 cc.arrayRemoveArray = function (arr, minusArr) {
802     for (var i = 0, l = minusArr.length; i < l; i++) {
803         cc.arrayRemoveObject(arr, minusArr[i]);
804     }
805 };

821  * Copy an array's item to a new array (its performance is better than Array.slice)

823  * @return {Array}

825 cc.copyArray = function(arr){

828         arr_clone[i] = arr[i];
