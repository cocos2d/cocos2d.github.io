 27 /** cc.Layer is a subclass of cc.Node that implements the TouchEventsDelegate protocol.<br/>
 28  * All features from cc.Node are valid, plus the bake feature: Baked layer can cache a static layer to improve performance
 29  * @class
 30  * @extends cc.Node
 31  */
 32 cc.Layer = cc.Node.extend(/** @lends cc.Layer# */{
 33     _className: "Layer",
 34 
 35     /**
 36      * <p>Constructor of cc.Layer, override it to extend the construction behavior, remember to call "this._super()" in the extended "ctor" function.</p>
 37      */
 38     ctor: function () {
 39         cc.Node.prototype.ctor.call(this);
 40         this._ignoreAnchorPointForPosition = true;
 41         this.setAnchorPoint(0.5, 0.5);
 42         this.setContentSize(cc.winSize);
 43         this._cascadeColorEnabled = false;
 44         this._cascadeOpacityEnabled = false;
 45     },
 46 
 47     /**
 48      * Sets the layer to cache all of children to a bake sprite, and draw itself by bake sprite. recommend using it in UI.<br/>
 49      * This is useful only in html5 engine
 50      * @function
 51      * @see cc.Layer#unbake
 52      */
 53     bake: function () {
 54         this._renderCmd.bake();
 55     },
 56 
 57     /**
 58      * Cancel the layer to cache all of children to a bake sprite.<br/>
 59      * This is useful only in html5 engine
 60      * @function
 61      * @see cc.Layer#bake
 62      */
 63     unbake: function () {
 64         this._renderCmd.unbake();
 65     },
 66 
 67     /**
 68      * Determines if the layer is baked.
 69      * @function
 70      * @returns {boolean}
 71      * @see cc.Layer#bake and cc.Layer#unbake
 72      */
 73     isBaked: function () {
 74         return this._renderCmd._isBaked;
 75     },
 76 
 77     visit: function (parent) {
 78         // quick return if not visible
 79         if (!this._visible)
 80             return;
 81 
 82         var renderer = cc.renderer, cmd = this._renderCmd;
 83         cmd.visit(parent && parent._renderCmd);
 84 
 85         if (cmd._isBaked) {
 86             renderer.pushRenderCommand(cmd);
 87             cmd._bakeSprite.visit(this);
 88         }
 89         else {
 90             var i, children = this._children, len = children.length, child;
 91             if (len > 0) {
 92                 if (this._reorderChildDirty) {
 93                     this.sortAllChildren();
 94                 }
 95                 // draw children zOrder < 0
 96                 for (i = 0; i < len; i++) {
 97                     child = children[i];
 98                     if (child._localZOrder < 0) {
 99                         child.visit(this);
100                     }
101                     else {
102                         break;
103                     }
104                 }
105 
106                 renderer.pushRenderCommand(cmd);
107                 for (; i < len; i++) {
108                     children[i].visit(this);
109                 }
110             } else {
111                 renderer.pushRenderCommand(cmd);
112             }
113         }
114         cmd._dirtyFlag = 0;
115     },
116 
117     addChild: function (child, localZOrder, tag) {
118         cc.Node.prototype.addChild.call(this, child, localZOrder, tag);
119         this._renderCmd._bakeForAddChild(child);
120     },
121 
122     _createRenderCmd: function () {
123         if (cc._renderType === cc.game.RENDER_TYPE_CANVAS)
124             return new cc.Layer.CanvasRenderCmd(this);
125         else
126             return new cc.Layer.WebGLRenderCmd(this);
127     }
128 });

303         } else {

305             locBlendFunc.dst = dst;
306         }
307         this._renderCmd.updateBlendFunc(locBlendFunc);
308     },

338 /**
339  * <p>
340  * CCLayerGradient is a subclass of cc.LayerColor that draws gradients across the background.<br/>
341  *<br/>
342  * All features from cc.LayerColor are valid, plus the following new features:<br/>
343  * <ul><li>direction</li>
344  * <li>final color</li>
345  * <li>interpolation mode</li></ul>
346  * <br/>
347  * Color is interpolated between the startColor and endColor along the given<br/>
348  * vector (starting at the origin, ending at the terminus).  If no vector is<br/>
349  * supplied, it defaults to (0, -1) -- a fade from top to bottom.<br/>
350  * <br/>
351  * If 'compressedInterpolation' is disabled, you will not see either the start or end color for<br/>
352  * non-cardinal vectors; a smooth gradient implying both end points will be still<br/>
353  * be drawn, however.<br/>
354  *<br/>
355  * If 'compressedInterpolation' is enabled (default mode) you will see both the start and end colors of the gradient.
356  * </p>
357  * @class
358  * @extends cc.LayerColor
359  *
360  * @param {cc.Color} start Starting color
361  * @param {cc.Color} end Ending color
362  * @param {cc.Point} [v=cc.p(0, -1)] A vector defines the gradient direction, default direction is from top to bottom
363  *
364  * @property {cc.Color} startColor              - Start color of the color gradient
365  * @property {cc.Color} endColor                - End color of the color gradient
366  * @property {Number}   startOpacity            - Start opacity of the color gradient
367  * @property {Number}   endOpacity              - End opacity of the color gradient
368  * @property {Number}   vector                  - Direction vector of the color gradient
369  * @property {Number}   compressedInterpolation  - Indicate whether or not the interpolation will be compressed
370  */
371 cc.LayerGradient = cc.LayerColor.extend(/** @lends cc.LayerGradient# */{
372     _endColor: null,
373     _startOpacity: 255,
374     _endOpacity: 255,
375     _alongVector: null,
376     _compressedInterpolation: false,
377     _className: "LayerGradient",
378     _colorStops: [],
379 
380     /**
381      * Constructor of cc.LayerGradient
382      * @param {cc.Color} start
383      * @param {cc.Color} end
384      * @param {cc.Point} [v=cc.p(0, -1)]
385      * @param {Array|Null} stops
386      *
387      * @example Using ColorStops argument:
388      * //startColor & endColor are for default and backward compatibility
389      * var layerGradient = new cc.LayerGradient(cc.color.RED, new cc.Color(255,0,0,0), cc.p(0, -1),
390      *                                          [{p:0, color: cc.color.RED},
391      *                                           {p:.5, color: new cc.Color(0,0,0,0)},
392      *                                           {p:1, color: cc.color.RED}]);
393      * //where p = A value between 0.0 and 1.0 that represents the position between start and end in a gradient
394      *
395      */
396     ctor: function (start, end, v, stops) {
397         cc.LayerColor.prototype.ctor.call(this);
398         this._endColor = cc.color(0, 0, 0, 255);
399         this._alongVector = cc.p(0, -1);
400         this._startOpacity = 255;
401         this._endOpacity = 255;
402 
403         if (stops && stops instanceof Array) {
404             this._colorStops = stops;
405             stops.splice(0, 0, {p: 0, color: start || cc.color.BLACK});
406             stops.push({p: 1, color: end || cc.color.BLACK});
407         } else
408             this._colorStops = [{p: 0, color: start || cc.color.BLACK}, {p: 1, color: end || cc.color.BLACK}];
409 
410         cc.LayerGradient.prototype.init.call(this, start, end, v, stops);
411     },
412 
413     /**
414      * Initialization of the layer, please do not call this function by yourself, you should pass the parameters to constructor to initialize a layer
415      * @param {cc.Color} start starting color
416      * @param {cc.Color} end
417      * @param {cc.Point|Null} v
418      * @param {Array|Null} stops
419      * @return {Boolean}
420      */
421     init: function (start, end, v, stops) {
422         start = start || cc.color(0, 0, 0, 255);
423         end = end || cc.color(0, 0, 0, 255);
424         v = v || cc.p(0, -1);
425         var _t = this;
426 
427         // Initializes the CCLayer with a gradient between start and end in the direction of v.
428         var locEndColor = _t._endColor;
429         _t._startOpacity = start.a;
430 
431         locEndColor.r = end.r;
432         locEndColor.g = end.g;
433         locEndColor.b = end.b;
434         _t._endOpacity = end.a;
435 
436         _t._alongVector = v;
437         _t._compressedInterpolation = true;
438 
439         cc.LayerColor.prototype.init.call(_t, cc.color(start.r, start.g, start.b, 255));
440         this._renderCmd.setDirtyFlag(cc.Node._dirtyFlags.colorDirty | cc.Node._dirtyFlags.opacityDirty | cc.Node._dirtyFlags.gradientDirty);
441         return true;
442     },
443

479     setStartColor: function (color) {
480         this.color = color;
481         //update the color stops
482         var stops = this._colorStops;
483         if (stops && stops.length > 0) {
484             var selColor = stops[0].color;
485             selColor.r = color.r;
486             selColor.g = color.g;
487             selColor.b = color.b;
488         }
489     },
490 
491     /**
492      * Sets the end gradient color
493      * @param {cc.Color} color
494      * @example
495      * // Example
496      * myGradientLayer.setEndColor(cc.color(255,0,0));
497      * //set the ending gradient to red
498      */
499     setEndColor: function (color) {
500         var locColor = this._endColor;
501         locColor.r = color.r;

507             var selColor = stops[stops.length - 1].color;
508             selColor.r = color.r;
509             selColor.g = color.g;
510             selColor.b = color.b;
511         }
512         this._renderCmd.setDirtyFlag(cc.Node._dirtyFlags.colorDirty);
513     },

675  * CCMultipleLayer is a CCLayer with the ability to multiplex it's children.<br/>
676  * Features:<br/>
677  *  <ul><li>- It supports one or more children</li>
678  *  <li>- Only one children will be active a time</li></ul>
679  * @class
680  * @extends cc.Layer
681  * @param {Array} layers an array of cc.Layer
682  * @example
683  * // Example
684  * var multiLayer = new cc.LayerMultiple(layer1, layer2, layer3);//any number of layers
685  */
686 cc.LayerMultiplex = cc.Layer.extend(/** @lends cc.LayerMultiplex# */{
687     _enabledLayer: 0,
688     _layers: null,
689     _className: "LayerMultiplex",
690 
691     /**
692      * Constructor of cc.LayerMultiplex
693      * @param {Array} layers an array of cc.Layer
694      */
695     ctor: function (layers) {
696         cc.Layer.prototype.ctor.call(this);
697         if (layers instanceof Array)
698             cc.LayerMultiplex.prototype.initWithLayers.call(this, layers);
699         else
700             cc.LayerMultiplex.prototype.initWithLayers.call(this, Array.prototype.slice.call(arguments));
701     },
702 
703     /**
704      * Initialization of the layer multiplex, please do not call this function by yourself, you should pass the parameters to constructor to initialize a layer multiplex
705      * @param {Array} layers an array of cc.Layer
706      * @return {Boolean}
707      */
708     initWithLayers: function (layers) {
709         if ((layers.length > 0) && (layers[layers.length - 1] == null))
710             cc.log(cc._LogInfos.LayerMultiplex_initWithLayers);
711 
712         this._layers = layers;
713         this._enabledLayer = 0;
714         this.addChild(this._layers[this._enabledLayer]);
715         return true;
716     },
717 
718     /**
719      * Switches to a certain layer indexed by n.<br/>
720      * The current (old) layer will be removed from it's parent with 'cleanup:YES'.
721      * @param {Number} n the layer index to switch to
722      */
723     switchTo: function (n) {
724         if (n >= this._layers.length) {
725             cc.log(cc._LogInfos.LayerMultiplex_switchTo);
726             return;
727         }
728 
729         this.removeChild(this._layers[this._enabledLayer], true);
730         this._enabledLayer = n;
731         this.addChild(this._layers[n]);
732     },
733 
734     /**
735      * Release the current layer and switches to another layer indexed by n.<br/>
736      * The current (old) layer will be removed from it's parent with 'cleanup:YES'.
737      * @param {Number} n the layer index to switch to
738      */
739     switchToAndReleaseMe: function (n) {
740         if (n >= this._layers.length) {
741             cc.log(cc._LogInfos.LayerMultiplex_switchToAndReleaseMe);
742             return;
743         }
744 
745         this.removeChild(this._layers[this._enabledLayer], true);
746 
747         //[layers replaceObjectAtIndex:_enabledLayer withObject:[NSNull null]];
748         this._layers[this._enabledLayer] = null;
749         this._enabledLayer = n;
750         this.addChild(this._layers[n]);
751     },
752 
753     /**
754      * Add a layer to the multiplex layers list
755      * @param {cc.Layer} layer
756      */
757     addLayer: function (layer) {
758         if (!layer) {
759             cc.log(cc._LogInfos.LayerMultiplex_addLayer);
760             return;
761         }
762         this._layers.push(layer);
763     }
764 });
765

767  * Creates a cc.LayerMultiplex with one or more layers using a variable argument list.
768  * @deprecated since v3.0, please use new construction instead
769  * @see cc.LayerMultiplex
770  * @return {cc.LayerMultiplex|Null}
771  */
772 cc.LayerMultiplex.create = function (/*Multiple Arguments*/) {
773     return new cc.LayerMultiplex(Array.prototype.slice.call(arguments));
774 };
