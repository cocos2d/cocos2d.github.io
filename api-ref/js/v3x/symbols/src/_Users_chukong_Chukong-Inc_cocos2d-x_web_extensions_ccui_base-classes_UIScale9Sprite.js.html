 32 var dataPool = {
 33     _pool: {},
 34     _lengths: [],
 35     put: function (data) {
 36         var length = data.length;
 37         if (!this._pool[length]) {
 38             this._pool[length] = [data];
 39             this._lengths.push(length);
 40             this._lengths.sort();
 41         }
 42         else {
 43             this._pool[length].push(data);
 44         }
 45     },
 46     get: function (length) {
 47         var id;
 48         for (var i = 0; i < this._lengths.length; i++) {
 49             if (this._lengths[i] >= length) {
 50                 id = this._lengths[i];
 51                 break;
 52             }
 53         }
 54         if (id) {
 55             return this._pool[id].pop();
 56         }
 57         else {
 58             return undefined;
 59         }
 60     }

 66 var simpleQuadGenerator = {
 67     _rebuildQuads_base: function (sprite, spriteFrame, contentSize, isTrimmedContentSize) {
 68         //build vertices
 69         var vertices = sprite._vertices,
 70             wt = sprite._renderCmd._worldTransform,
 71             l, b, r, t;
 72         if (isTrimmedContentSize) {
 73             l = 0;
 74             b = 0;
 75             r = contentSize.width;
 76             t = contentSize.height;
 77         } else {
 78             var originalSize = spriteFrame._originalSize;
 79             var rect = spriteFrame._rect;
 80             var offset = spriteFrame._offset;
 81             var scaleX = contentSize.width / originalSize.width;
 82             var scaleY = contentSize.height / originalSize.height;
 83             var trimmLeft = offset.x + (originalSize.width - rect.width) / 2;
 84             var trimmRight = offset.x - (originalSize.width - rect.width) / 2;
 85             var trimmedBottom = offset.y + (originalSize.height - rect.height) / 2;
 86             var trimmedTop = offset.y - (originalSize.height - rect.height) / 2;
 87 
 88             l = trimmLeft * scaleX;
 89             b = trimmedBottom * scaleY;
 90             r = contentSize.width + trimmRight * scaleX;
 91             t = contentSize.height + trimmedTop * scaleY;
 92         }
 93 
 94         if (vertices.length < 8) {
 95             dataPool.put(vertices);
 96             vertices = dataPool.get(8) || new Float32Array(8);
 97             sprite._vertices = vertices;
 98         }
 99         // bl, br, tl, tr
100         if (webgl) {
101             vertices[0] = l * wt.a + b * wt.c + wt.tx;
102             vertices[1] = l * wt.b + b * wt.d + wt.ty;
103             vertices[2] = r * wt.a + b * wt.c + wt.tx;
104             vertices[3] = r * wt.b + b * wt.d + wt.ty;
105             vertices[4] = l * wt.a + t * wt.c + wt.tx;
106             vertices[5] = l * wt.b + t * wt.d + wt.ty;
107             vertices[6] = r * wt.a + t * wt.c + wt.tx;
108             vertices[7] = r * wt.b + t * wt.d + wt.ty;
109         }
110         else {
111             vertices[0] = l;
112             vertices[1] = b;
113             vertices[2] = r;
114             vertices[3] = b;
115             vertices[4] = l;
116             vertices[5] = t;
117             vertices[6] = r;
118             vertices[7] = t;
119         }
120 
121         cornerId[0] = 0;
122         cornerId[1] = 2;
123         cornerId[2] = 4;
124         cornerId[3] = 6;
125 
126         //build uvs
127         if (sprite._uvsDirty) {
128             this._calculateUVs(sprite, spriteFrame);
129         }
130 
131         sprite._vertCount = 4;
132     },
133 
134     _calculateUVs: function (sprite, spriteFrame) {
135         var uvs = sprite._uvs;
136         var atlasWidth = spriteFrame._texture._pixelsWide;
137         var atlasHeight = spriteFrame._texture._pixelsHigh;
138         var textureRect = spriteFrame._rect;
139 
140         if (uvs.length < 8) {
141             dataPool.put(uvs);
142             uvs = dataPool.get(8) || new Float32Array(8);
143             sprite._uvs = uvs;
144         }
145 
146         //uv computation should take spritesheet into account.
147         var l, b, r, t;
148         var texelCorrect = FIX_ARTIFACTS_BY_STRECHING_TEXEL ? 0.5 : 0;
149 
150         if (spriteFrame._rotated) {
151             l = (textureRect.x + texelCorrect) / atlasWidth;
152             b = (textureRect.y + textureRect.width - texelCorrect) / atlasHeight;
153             r = (textureRect.x + textureRect.height - texelCorrect) / atlasWidth;
154             t = (textureRect.y + texelCorrect) / atlasHeight;
155             uvs[0] = l; uvs[1] = t;
156             uvs[2] = l; uvs[3] = b;
157             uvs[4] = r; uvs[5] = t;
158             uvs[6] = r; uvs[7] = b;
159         }
160         else {
161             l = (textureRect.x + texelCorrect) / atlasWidth;
162             b = (textureRect.y + textureRect.height - texelCorrect) / atlasHeight;
163             r = (textureRect.x + textureRect.width - texelCorrect) / atlasWidth;
164             t = (textureRect.y + texelCorrect) / atlasHeight;
165             uvs[0] = l; uvs[1] = b;
166             uvs[2] = r; uvs[3] = b;
167             uvs[4] = l; uvs[5] = t;
168             uvs[6] = r; uvs[7] = t;
169         }
170     }

173 var scale9QuadGenerator = {
174     x: new Array(4),
175     y: new Array(4),
176     _rebuildQuads_base: function (sprite, spriteFrame, contentSize, insetLeft, insetRight, insetTop, insetBottom) {
177         //build vertices
178         var vertices = sprite._vertices;
179         var wt = sprite._renderCmd._worldTransform;
180         var leftWidth, centerWidth, rightWidth;
181         var topHeight, centerHeight, bottomHeight;
182         var rect = spriteFrame._rect;
183 
184         leftWidth = insetLeft;
185         rightWidth = insetRight;
186         centerWidth = rect.width - leftWidth - rightWidth;
187         topHeight = insetTop;
188         bottomHeight = insetBottom;
189         centerHeight = rect.height - topHeight - bottomHeight;
190 
191         var preferSize = contentSize;
192         var sizableWidth = preferSize.width - leftWidth - rightWidth;
193         var sizableHeight = preferSize.height - topHeight - bottomHeight;
194         var xScale = preferSize.width / (leftWidth + rightWidth);
195         var yScale = preferSize.height / (topHeight + bottomHeight);
196         xScale = xScale > 1 ? 1 : xScale;
197         yScale = yScale > 1 ? 1 : yScale;
198         sizableWidth = sizableWidth < 0 ? 0 : sizableWidth;
199         sizableHeight = sizableHeight < 0 ? 0 : sizableHeight;
200         var x = this.x;
201         var y = this.y;
202         x[0] = 0;
203         x[1] = leftWidth * xScale;
204         x[2] = x[1] + sizableWidth;
205         x[3] = preferSize.width;
206         y[0] = 0;
207         y[1] = bottomHeight * yScale;
208         y[2] = y[1] + sizableHeight;
209         y[3] = preferSize.height;
210 
211         if (vertices.length < 32) {
212             dataPool.put(vertices);
213             vertices = dataPool.get(32) || new Float32Array(32);
214             sprite._vertices = vertices;
215         }
216         var offset = 0, row, col;
217         if (webgl) {
218             for (row = 0; row < 4; row++) {
219                 for (col = 0; col < 4; col++) {
220                     vertices[offset] = x[col] * wt.a + y[row] * wt.c + wt.tx;
221                     vertices[offset+1] = x[col] * wt.b + y[row] * wt.d + wt.ty;
222                     offset += 2;
223                 }
224             }
225         }
226         else {
227             for (row = 0; row < 4; row++) {
228                 for (col = 0; col < 4; col++) {
229                     vertices[offset] = x[col];
230                     vertices[offset+1] = y[row];
231                     offset += 2;
232                 }
233             }
234         }
235 
236         cornerId[0] = 0;
237         cornerId[1] = 6;
238         cornerId[2] = 24;
239         cornerId[3] = 30;
240 
241         //build uvs
242         if (sprite._uvsDirty) {
243             this._calculateUVs(sprite, spriteFrame, insetLeft, insetRight, insetTop, insetBottom);
244         }
245     },

247     _calculateUVs: function (sprite, spriteFrame, insetLeft, insetRight, insetTop, insetBottom) {
248         var uvs = sprite._uvs;
249         var rect = spriteFrame._rect;
250         var atlasWidth = spriteFrame._texture._pixelsWide;
251         var atlasHeight = spriteFrame._texture._pixelsHigh;
252 
253         //caculate texture coordinate
254         var leftWidth, centerWidth, rightWidth;
255         var topHeight, centerHeight, bottomHeight;
256         var textureRect = spriteFrame._rect;
257 
258         leftWidth = insetLeft;
259         rightWidth = insetRight;
260         centerWidth = rect.width - leftWidth - rightWidth;
261         topHeight = insetTop;
262         bottomHeight = insetBottom;
263         centerHeight = rect.height - topHeight - bottomHeight;
264 
265         if (uvs.length < 32) {
266             dataPool.put(uvs);
267             uvs = dataPool.get(32) || new Float32Array(32);
268             sprite._uvs = uvs;
269         }
270 
271         //uv computation should take spritesheet into account.
272         var u = this.x;
273         var v = this.y;
274         var texelCorrect = FIX_ARTIFACTS_BY_STRECHING_TEXEL ? 0.5 : 0;
275         var offset = 0, row, col;
276 
277         if (spriteFrame._rotated) {
278             u[0] = (textureRect.x + texelCorrect) / atlasWidth;
279             u[1] = (bottomHeight + textureRect.x) / atlasWidth;
280             u[2] = (bottomHeight + centerHeight + textureRect.x) / atlasWidth;
281             u[3] = (textureRect.x + textureRect.height - texelCorrect) / atlasWidth;
282 
283             v[3] = (textureRect.y + texelCorrect) / atlasHeight;
284             v[2] = (leftWidth + textureRect.y) / atlasHeight;
285             v[1] = (leftWidth + centerWidth + textureRect.y) / atlasHeight;
286             v[0] = (textureRect.y + textureRect.width - texelCorrect) / atlasHeight;
287 
288             for (row = 0; row < 4; row++) {
289                 for (col = 0; col < 4; col++) {
290                     uvs[offset] = u[row];
291                     uvs[offset+1] = v[3-col];
292                     offset += 2;
293                 }
294             }
295         }
296         else {
297             u[0] = (textureRect.x + texelCorrect) / atlasWidth;
298             u[1] = (leftWidth + textureRect.x) / atlasWidth;
299             u[2] = (leftWidth + centerWidth + textureRect.x) / atlasWidth;
300             u[3] = (textureRect.x + textureRect.width - texelCorrect) / atlasWidth;
301 
302             v[3] = (textureRect.y + texelCorrect) / atlasHeight;
303             v[2] = (topHeight + textureRect.y) / atlasHeight;
304             v[1] = (topHeight + centerHeight + textureRect.y) / atlasHeight;
305             v[0] = (textureRect.y + textureRect.height - texelCorrect) / atlasHeight;
306 
307             for (row = 0; row < 4; row++) {
308                 for (col = 0; col < 4; col++) {
309                     uvs[offset] = u[col];
310                     uvs[offset+1] = v[row];
311                     offset += 2;
312                 }
313             }
314         }
315     }

319  * <p>
320  * A 9-slice sprite for cocos2d UI.                                                                    <br/>
321  *                                                                                                     <br/>
322  * 9-slice scaling allows you to specify how scaling is applied                                        <br/>
323  * to specific areas of a sprite. With 9-slice scaling (3x3 grid),                                     <br/>
324  * you can ensure that the sprite does not become distorted when                                       <br/>
325  * scaled.                                                                                             <br/>
326  * @see http://yannickloriot.com/library/ios/cccontrolextension/Classes/CCScale9Sprite.html            <br/>
327  * </p>

329  * @extends cc.Node
330  *
331  * @property {cc.Size}  preferredSize   - The preferred size of the 9-slice sprite

333  * @property {Number}   insetLeft       - The left inset of the 9-slice sprite
334  * @property {Number}   insetTop        - The top inset of the 9-slice sprite
335  * @property {Number}   insetRight      - The right inset of the 9-slice sprite
336  * @property {Number}   insetBottom     - The bottom inset of the 9-slice sprite
337  */

381         cc.Node.prototype.ctor.call(this);

437     _updateCapInsets: function (rect, capInsets) {
438         if(!capInsets || !rect || cc._rectEqualToZero(capInsets)) {
439             rect = rect || {x:0, y:0, width: this._contentSize.width, height: this._contentSize.height};
440             this._capInsetsInternal = cc.rect(rect.width /3,
441                                               rect.height /3,
442                                               rect.width /3,
443                                               rect.height /3);
444         } else {
445             this._capInsetsInternal = capInsets;
446         }
447 
448         if(!cc._rectEqualToZero(rect)) {
449             this._insetLeft = this._capInsetsInternal.x;
450             this._insetTop = this._capInsetsInternal.y;
451             this._insetRight = rect.width - this._insetLeft - this._capInsetsInternal.width;
452             this._insetBottom = rect.height - this._insetTop - this._capInsetsInternal.height;
453         }

457     initWithFile: function (file, rect, capInsets) {
458         if (file instanceof cc.Rect) {
459             file = arguments[1];
460             capInsets = arguments[0];
461             rect = cc.rect(0, 0, 0, 0);
462         } else {
463             rect = rect || cc.rect(0, 0, 0, 0);
464             capInsets = capInsets || cc.rect(0, 0, 0, 0);
465         }
466 
467         if(!file)
468             throw new Error("ccui.Scale9Sprite.initWithFile(): file should be non-null");
469 
470         var texture = cc.textureCache.getTextureForKey(file);
471         if (!texture) {
472             texture = cc.textureCache.addImage(file);
473         }
474 
475         var locLoaded = texture.isLoaded();
476         this._loader.clear();
477         if (!locLoaded) {
478             this._loader.once(texture, function () {
479                 this.initWithFile(file, rect, capInsets);
480                 this.dispatchEvent("load");
481             }, this);
482             return false;
483         }
484 
485         //in this function, the texture already make sure is loaded.
486         if( cc._rectEqualToZero(rect)) {
487             var textureSize = texture.getContentSize();
488             rect = cc.rect(0, 0, textureSize.width, textureSize.height);
489         }
490         this.setTexture(texture, rect);
491         this._updateCapInsets(rect, capInsets);
492 
493         return true;
494     },

507             }, this);
508             return false;

512         this._updateCapInsets(originalRect, capInsets);
513 
514         return true;

565     _updateBlendFunc: function () {
566         // it's possible to have an untextured sprite
567         var blendFunc = this._blendFunc;
568         if (!this._spriteFrame || !this._spriteFrame._texture.hasPremultipliedAlpha()) {

575                 blendFunc.src = cc.ONE;
576             }
577             this._opacityModifyRGB = true;
578         }

622     /**

805             return;
806         }
807 
808         this._updateBlendFunc();
809 
810         this._isTriangle = false;
811         switch (this._renderingType) {
812           case RenderingType.SIMPLE:
813               simpleQuadGenerator._rebuildQuads_base(this, this._spriteFrame, this._contentSize, this._isTrimmedContentSize);
814               break;
815           case RenderingType.SLICED:
816               scale9QuadGenerator._rebuildQuads_base(this, this._spriteFrame, this._contentSize, this._insetLeft, this._insetRight, this._insetTop, this._insetBottom);
817               break;
818           default:
819               this._quadsDirty = false;
820               this._uvsDirty = false;
821               cc.error('Can not generate quad');
822               return;
823         }
824 
825 
826         this._quadsDirty = false;
827         this._uvsDirty = false;
828     },
