 27  * The main class of Armature, it plays armature animation, manages and updates bones' state.

 55     /**
 56      * Create a armature node.
 57      * Constructor of ccs.Armature
 58      * @param {String} name
 59      * @param {ccs.Bone} parentBone
 60      * @example
 61      * var armature = new ccs.Armature();
 62      */
 63     ctor: function (name, parentBone) {
 64         cc.Node.prototype.ctor.call(this);
 65         this._name = "";
 66         this._topBoneList = [];
 67         this._armatureIndexDic = {};
 68         this._offsetPoint = cc.p(0, 0);
 69         this._armatureTransformDirty = true;
 70         this._blendFunc = {src: cc.BLEND_SRC, dst: cc.BLEND_DST};
 71         name && ccs.Armature.prototype.init.call(this, name, parentBone);
 72         // Hack way to avoid RendererWebGL from skipping Armature
 73         this._texture = {};
 74     },

 76     /**
 77      * Initializes a CCArmature with the specified name and CCBone
 78      * @param {String} [name]
 79      * @param {ccs.Bone} [parentBone]
 80      * @return {Boolean}
 81      */
 82     init: function (name, parentBone) {
 83         if (parentBone)
 84             this._parentBone = parentBone;
 85         this.removeAllChildren();
 86         this.animation = new ccs.ArmatureAnimation();
 87         this.animation.init(this);
 88 
 89         this._boneDic = {};
 90         this._topBoneList.length = 0;
 91 
 92         //this._name = name || "";
 93         var armatureDataManager = ccs.armatureDataManager;
 94 
 95         var animationData;
 96         if (name !== "") {
 97             //animationData
 98             animationData = armatureDataManager.getAnimationData(name);
 99             cc.assert(animationData, "AnimationData not exist!");
100 
101             this.animation.setAnimationData(animationData);
102 
103             //armatureData
104             var armatureData = armatureDataManager.getArmatureData(name);
105             cc.assert(armatureData, "ArmatureData not exist!");
106 
107             this.armatureData = armatureData;
108 
109             //boneDataDic
110             var boneDataDic = armatureData.getBoneDataDic();
111             for (var key in boneDataDic) {
112                 var bone = this.createBone(String(key));
113 
114                 //! init bone's  Tween to 1st movement's 1st frame
115                 do {
116                     var movData = animationData.getMovement(animationData.movementNames[0]);
117                     if (!movData) break;
118 
119                     var _movBoneData = movData.getMovementBoneData(bone.getName());
120                     if (!_movBoneData || _movBoneData.frameList.length <= 0) break;
121 
122                     var frameData = _movBoneData.getFrameData(0);
123                     if (!frameData) break;
124 
125                     bone.getTweenData().copy(frameData);
126                     bone.changeDisplayWithIndex(frameData.displayIndex, false);
127                 } while (0);
128             }
129 
130             this.update(0);
131             this.updateOffsetPoint();
132         } else {
133             name = "new_armature";
134             this.armatureData = new ccs.ArmatureData();
135             this.armatureData.name = name;
136 
137             animationData = new ccs.AnimationData();
138             animationData.name = name;
139 
140             armatureDataManager.addArmatureData(name, this.armatureData);
141             armatureDataManager.addAnimationData(name, animationData);
142 
143             this.animation.setAnimationData(animationData);
144         }
145 
146         this._renderCmd.initShaderCache();
147 
148         this.setCascadeOpacityEnabled(true);
149         this.setCascadeColorEnabled(true);
150         return true;
151     },
152 
153     visit: function (parent) {
154         // quick return if not visible
155         if (!this._visible)
156             return;
157 
158         this._renderCmd.visit(parent && parent._renderCmd);
159         this._renderCmd._dirtyFlag = 0;
160     },

162     addChild: function (child, localZOrder, tag) {
163         if (child instanceof ccui.Widget) {
164             cc.log("Armature doesn't support to add Widget as its child, it will be fix soon.");
165             return;
166         }
167         cc.Node.prototype.addChild.call(this, child, localZOrder, tag);

170     /**
171      * create a bone with name
172      * @param {String} boneName
173      * @return {ccs.Bone}
174      */
175     createBone: function (boneName) {
176         var existedBone = this.getBone(boneName);
177         if (existedBone)
178             return existedBone;
179 
180         var boneData = this.armatureData.getBoneData(boneName);
181         var parentName = boneData.parentName;
182 
183         var bone = null;
184         if (parentName) {
185             this.createBone(parentName);
186             bone = new ccs.Bone(boneName);
187             this.addBone(bone, parentName);
188         } else {
189             bone = new ccs.Bone(boneName);
190             this.addBone(bone, "");
191         }
192 
193         bone.setBoneData(boneData);
194         bone.getDisplayManager().changeDisplayWithIndex(-1, false);
195         return bone;
196     },

198     /**
199      * Add a Bone to this Armature
200      * @param {ccs.Bone} bone  The Bone you want to add to Armature
201      * @param {String} parentName The parent Bone's name you want to add to. If it's  null, then set Armature to its parent
202      */
203     addBone: function (bone, parentName) {
204         cc.assert(bone, "Argument must be non-nil");
205         var locBoneDic = this._boneDic;
206         if (bone.getName())
207             cc.assert(!locBoneDic[bone.getName()], "bone already added. It can't be added again");
208 
209         if (parentName) {
210             var boneParent = locBoneDic[parentName];
211             if (boneParent)
212                 boneParent.addChildBone(bone);
213             else
214                 this._topBoneList.push(bone);
215         } else
216             this._topBoneList.push(bone);
217         bone.setArmature(this);
218 
219         locBoneDic[bone.getName()] = bone;
220         this.addChild(bone);
221     },
222 
223     /**
224      * Remove a bone with the specified name. If recursion it will also remove child Bone recursively.
225      * @param {ccs.Bone} bone The bone you want to remove
226      * @param {Boolean} recursion Determine whether remove the bone's child  recursion.
227      */
228     removeBone: function (bone, recursion) {
229         cc.assert(bone, "bone must be added to the bone dictionary!");
230 
231         bone.setArmature(null);
232         bone.removeFromParent(recursion);
233         cc.arrayRemoveObject(this._topBoneList, bone);
234 
235         delete  this._boneDic[bone.getName()];
236         this.removeChild(bone, true);
237     },
238

240      * Gets a bone with the specified name
241      * @param {String} name The bone's name you want to get
242      * @return {ccs.Bone}
243      */

248     /**
249      * Change a bone's parent with the specified parent name.
250      * @param {ccs.Bone} bone The bone you want to change parent
251      * @param {String} parentName The new parent's name
252      */
253     changeBoneParent: function (bone, parentName) {
254         cc.assert(bone, "bone must be added to the bone dictionary!");
255 
256         var parentBone = bone.getParentBone();
257         if (parentBone) {
258             cc.arrayRemoveObject(parentBone.getChildren(), bone);
259             bone.setParentBone(null);
260         }
261 
262         if (parentName) {
263             var boneParent = this._boneDic[parentName];
264             if (boneParent) {
265                 boneParent.addChildBone(bone);
266                 cc.arrayRemoveObject(this._topBoneList, bone);
267             } else
268                 this._topBoneList.push(bone);
269         }
270     },

280     /**
281      * Set contentSize and Calculate anchor point.
282      */
283     updateOffsetPoint: function () {
284         // Set contentsize and Calculate anchor point.
285         var rect = this.getBoundingBox();
286         this.setContentSize(rect);
287         var locOffsetPoint = this._offsetPoint;
288         locOffsetPoint.x = -rect.x;
289         locOffsetPoint.y = -rect.y;
290         if (rect.width !== 0 && rect.height !== 0)
291             this.setAnchorPoint(locOffsetPoint.x / rect.width, locOffsetPoint.y / rect.height);
292     },
293

353     /**
354      * This boundingBox will calculate all bones' boundingBox every time
355      * @returns {cc.Rect}
356      */
357     getBoundingBox: function () {
358         var minX, minY, maxX, maxY = 0;
359         var first = true;
360 
361         var boundingBox = cc.rect(0, 0, 0, 0), locChildren = this._children;
362 
363         var len = locChildren.length;
364         for (var i = 0; i < len; i++) {
365             var bone = locChildren[i];
366             if (bone) {
367                 var r = bone.getDisplayManager().getBoundingBox();
368                 if (r.x === 0 && r.y === 0 && r.width === 0 && r.height === 0)
369                     continue;
370 
371                 if (first) {
372                     minX = r.x;
373                     minY = r.y;
374                     maxX = r.x + r.width;
375                     maxY = r.y + r.height;
376                     first = false;
377                 } else {
378                     minX = r.x < boundingBox.x ? r.x : boundingBox.x;
379                     minY = r.y < boundingBox.y ? r.y : boundingBox.y;
380                     maxX = r.x + r.width > boundingBox.x + boundingBox.width ?
381                         r.x + r.width : boundingBox.x + boundingBox.width;
382                     maxY = r.y + r.height > boundingBox.y + boundingBox.height ?
383                         r.y + r.height : boundingBox.y + boundingBox.height;
384                 }
385 
386                 boundingBox.x = minX;
387                 boundingBox.y = minY;
388                 boundingBox.width = maxX - minX;
389                 boundingBox.height = maxY - minY;
390             }
391         }
392         return cc.rectApplyAffineTransform(boundingBox, this.getNodeToParentTransform());
393     },

395     /**
396      * when bone  contain the point ,then return it.
397      * @param {Number} x
398      * @param {Number} y
399      * @returns {ccs.Bone}
400      */
401     getBoneAtPoint: function (x, y) {
402         var locChildren = this._children;
403         for (var i = locChildren.length - 1; i >= 0; i--) {
404             var child = locChildren[i];
405             if (child instanceof ccs.Bone && child.getDisplayManager().containPoint(x, y))
406                 return child;
407         }
408         return null;
409     },

431     /**
432      * draw contour
433      */
434     drawContour: function () {
435         cc._drawingUtil.setDrawColor(255, 255, 255, 255);
436         cc._drawingUtil.setLineWidth(1);
437         var locBoneDic = this._boneDic;
438         for (var key in locBoneDic) {
439             var bone = locBoneDic[key];
440             var detector = bone.getColliderDetector();
441             if (!detector)
442                 continue;
443             var bodyList = detector.getColliderBodyList();
444             for (var i = 0; i < bodyList.length; i++) {
445                 var body = bodyList[i];
446                 var vertexList = body.getCalculatedVertexList();
447                 cc._drawingUtil.drawPoly(vertexList, vertexList.length, true);
448             }
449         }
450     },
451 
452     setBody: function (body) {
453         if (this._body === body)
454             return;
455 
456         this._body = body;
457         this._body.data = this;
458         var child, displayObject, locChildren = this._children;
459         for (var i = 0; i < locChildren.length; i++) {
460             child = locChildren[i];
461             if (child instanceof ccs.Bone) {
462                 var displayList = child.getDisplayManager().getDecorativeDisplayList();
463                 for (var j = 0; j < displayList.length; j++) {
464                     displayObject = displayList[j];
465                     var detector = displayObject.getColliderDetector();
466                     if (detector)
467                         detector.setBody(this._body);
468                 }
469             }
470         }
471     },
