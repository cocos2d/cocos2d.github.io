 97     _updateInnerContainerSize: function () {
 98         var i, locItems = this._items, length;
 99         switch (this.direction) {
100             case ccui.ScrollView.DIR_VERTICAL:
101                 length = locItems.length;
102                 var totalHeight = (length - 1) * this._itemsMargin;
103                 for (i = 0; i < length; i++) {
104                     totalHeight += locItems[i].getContentSize().height;
105                 }
106                 this.setInnerContainerSize(cc.size(this._contentSize.width, totalHeight));
107                 break;
108             case ccui.ScrollView.DIR_HORIZONTAL:
109                 length = locItems.length;
110                 var totalWidth = (length - 1) * this._itemsMargin;
111                 for (i = 0; i < length; i++) {
112                     totalWidth += locItems[i].getContentSize().width;
113                 }
114                 this.setInnerContainerSize(cc.size(totalWidth, this._contentSize.height));
115                 break;
116             default:
117                 break;
118         }

121     _remedyLayoutParameter: function (item) {
122         cc.assert(null != item, "ListView Item can't be nil!");
123 
124         var linearLayoutParameter = item.getLayoutParameter(ccui.LayoutParameter.LINEAR);
125         var isLayoutParameterExists = true;
126         if (!linearLayoutParameter) {
127             linearLayoutParameter = new ccui.LinearLayoutParameter();
128             isLayoutParameterExists = false;
129         }
130         var itemIndex = this.getIndex(item);
131         switch (this.direction) {
132             case ccui.ScrollView.DIR_VERTICAL:
133                 this._remedyVerticalLayoutParameter(linearLayoutParameter, itemIndex);
134                 break;
135             case ccui.ScrollView.DIR_HORIZONTAL:
136                 this._remedyHorizontalLayoutParameter(linearLayoutParameter, itemIndex);
137                 break;
138             default:
139                 break;
140         }
141         if (!isLayoutParameterExists)
142             item.setLayoutParameter(linearLayoutParameter);
143     },

145     //@since v3.3
146     _remedyVerticalLayoutParameter: function (layoutParameter, itemIndex) {
147         cc.assert(null != layoutParameter, "Layout parameter can't be nil!");
148 
149         switch (this._gravity) {
150             case ccui.ListView.GRAVITY_LEFT:
151                 layoutParameter.setGravity(ccui.LinearLayoutParameter.LEFT);
152                 break;
153             case ccui.ListView.GRAVITY_RIGHT:
154                 layoutParameter.setGravity(ccui.LinearLayoutParameter.RIGHT);
155                 break;
156             case ccui.ListView.GRAVITY_CENTER_HORIZONTAL:
157                 layoutParameter.setGravity(ccui.LinearLayoutParameter.CENTER_HORIZONTAL);
158                 break;
159             default:
160                 break;
161         }
162         if (0 === itemIndex)
163             layoutParameter.setMargin(ccui.MarginZero());
164         else
165             layoutParameter.setMargin(new ccui.Margin(0.0, this._itemsMargin, 0.0, 0.0));
166     },

168     //@since v3.3
169     _remedyHorizontalLayoutParameter: function (layoutParameter, itemIndex) {
170         cc.assert(null != layoutParameter, "Layout parameter can't be nil!");
171 
172         switch (this._gravity) {
173             case ccui.ListView.GRAVITY_TOP:
174                 layoutParameter.setGravity(ccui.LinearLayoutParameter.TOP);
175                 break;
176             case ccui.ListView.GRAVITY_BOTTOM:
177                 layoutParameter.setGravity(ccui.LinearLayoutParameter.BOTTOM);
178                 break;
179             case ccui.ListView.GRAVITY_CENTER_VERTICAL:
180                 layoutParameter.setGravity(ccui.LinearLayoutParameter.CENTER_VERTICAL);
181                 break;
182             default:
183                 break;
184         }
185         if (0 === itemIndex)
186             layoutParameter.setMargin(ccui.MarginZero());
187         else
188             layoutParameter.setMargin(new ccui.Margin(this._itemsMargin, 0.0, 0.0, 0.0));

438     _getHowMuchOutOfBoundary: function (addition) {
439         if (addition === undefined)
440             addition = cc.p(0, 0);
441 
442         if (!this._magneticAllowedOutOfBoundary || this._items.length === 0) {
443             return ccui.ScrollView.prototype._getHowMuchOutOfBoundary.call(this, addition);
444         }
445         else if (this._magneticType === ccui.ListView.MAGNETIC_NONE || this._magneticType === ccui.ListView.MAGNETIC_BOTH_END) {
446             return ccui.ScrollView.prototype._getHowMuchOutOfBoundary.call(this, addition);
447         }
448         else if (addition.x === 0 && addition.y === 0 && !this._outOfBoundaryAmountDirty) {
449             return this._outOfBoundaryAmount;
450         }
451 
452         // If it is allowed to be out of boundary by magnetic, adjust the boundaries according to the magnetic type.
453         var leftBoundary = this._leftBoundary;
454         var rightBoundary = this._rightBoundary;
455         var topBoundary = this._topBoundary;
456         var bottomBoundary = this._bottomBoundary;
457 
458         var lastItemIndex = this._items.length - 1;
459         var contentSize = this.getContentSize();
460         var firstItemAdjustment = cc.p(0, 0);
461         var lastItemAdjustment = cc.p(0, 0);
462 
463         switch (this._magneticType) {
464             case  ccui.ListView.MAGNETIC_CENTER:
465                 firstItemAdjustment.x = (contentSize.width - this._items[0].width) / 2;
466                 firstItemAdjustment.y = (contentSize.height - this._items[0].height) / 2;
467 
468                 lastItemAdjustment.x = (contentSize.width - this._items[lastItemIndex].width) / 2;
469                 lastItemAdjustment.y = (contentSize.height - this._items[lastItemIndex].height) / 2;
470 
471                 break;
472             case ccui.ListView.MAGNETIC_LEFT:
473             case ccui.ListView.MAGNETIC_TOP:
474                 lastItemAdjustment.x = contentSize.width - this._items[lastItemIndex].width;
475                 lastItemAdjustment.y = contentSize.height - this._items[lastItemIndex].height;
476                 break;
477             case ccui.ListView.MAGNETIC_RIGHT:
478             case ccui.ListView.MAGNETIC_BOTTOM:
479                 firstItemAdjustment.x = contentSize.width - this._items[0].width;
480                 firstItemAdjustment.y = contentSize.height - this._items[0].height;
481                 break;
482         }
483 
484         leftBoundary += firstItemAdjustment.x;
485         rightBoundary -= lastItemAdjustment.x;
486         topBoundary -= firstItemAdjustment.y;
487         bottomBoundary += lastItemAdjustment.y;
488 
489 
490         // Calculate the actual amount
491         var outOfBoundaryAmount = cc.p(0, 0);
492 
493         if (this._innerContainer.getLeftBoundary() + addition.x > leftBoundary) {
494             outOfBoundaryAmount.x = leftBoundary - (this._innerContainer.getLeftBoundary() + addition.x);
495         }
496         else if (this._innerContainer.getRightBoundary() + addition.x < rightBoundary) {
497             outOfBoundaryAmount.x = rightBoundary - (this._innerContainer.getRightBoundary() + addition.x);
498         }
499 
500         if (this._innerContainer.getTopBoundary() + addition.y < topBoundary) {
501             outOfBoundaryAmount.y = topBoundary - (this._innerContainer.getTopBoundary() + addition.y);
502         }
503         else if (this._innerContainer.getBottomBoundary() + addition.y > bottomBoundary) {
504             outOfBoundaryAmount.y = bottomBoundary - (this._innerContainer.getBottomBoundary() + addition.y);
505         }
506 
507         if (addition.x === 0 && addition.y === 0) {
508             this._outOfBoundaryAmount = outOfBoundaryAmount;
509             this._outOfBoundaryAmountDirty = false;
510         }
511         return outOfBoundaryAmount;
512     },

514     _calculateItemPositionWithAnchor: function (item, itemAnchorPoint) {
515         var origin = cc.p(item.getLeftBoundary(), item.getBottomBoundary());
516         var size = item.getContentSize();
517 
518         return cc.p(origin.x + size.width * itemAnchorPoint.x, origin.y + size.height * itemAnchorPoint.y);
519     },

521     _findClosestItem: function (targetPosition, items, itemAnchorPoint, firstIndex, distanceFromFirst, lastIndex, distanceFromLast) {
522         cc.assert(firstIndex >= 0 && lastIndex < items.length && firstIndex <= lastIndex, "");
523         if (firstIndex === lastIndex) {
524             return items[firstIndex];
525         }
526         if (lastIndex - firstIndex === 1) {
527             if (distanceFromFirst <= distanceFromLast) {
528                 return items[firstIndex];
529             }
530             else {
531                 return items[lastIndex];
532             }
533         }
534 
535         // Binary search
536         var midIndex = Math.floor((firstIndex + lastIndex) / 2);
537         var itemPosition = this._calculateItemPositionWithAnchor(items[midIndex], itemAnchorPoint);
538         var distanceFromMid = cc.pLength(cc.pSub(targetPosition, itemPosition));
539 
540         if (distanceFromFirst <= distanceFromLast) {
541             // Left half
542             return this._findClosestItem(targetPosition, items, itemAnchorPoint, firstIndex, distanceFromFirst, midIndex, distanceFromMid);
543         }
544         else {
545             // Right half
546             return this._findClosestItem(targetPosition, items, itemAnchorPoint, midIndex, distanceFromMid, lastIndex, distanceFromLast);
547         }
548     },

550     /**
551      * Query the closest item to a specific position in inner container.
552      *
553      * @param {cc.Point} targetPosition Specifies the target position in inner container's coordinates.
554      * @param {cc.Point} itemAnchorPoint Specifies an anchor point of each item for position to calculate distance.
555      * @returns {?ccui.Widget} A item instance if list view is not empty. Otherwise, returns null.
556      */
557     getClosestItemToPosition: function (targetPosition, itemAnchorPoint) {
558         if (this._items.length === 0) {
559             return null;
560         }
561 
562         // Find the closest item through binary search
563         var firstIndex = 0;
564         var firstPosition = this._calculateItemPositionWithAnchor(this._items[firstIndex], itemAnchorPoint);
565         var distanceFromFirst = cc.pLength(cc.pSub(targetPosition, firstPosition));
566 
567         var lastIndex = this._items.length - 1;
568         var lastPosition = this._calculateItemPositionWithAnchor(this._items[lastIndex], itemAnchorPoint);
569         var distanceFromLast = cc.pLength(cc.pSub(targetPosition, lastPosition));
570 
571         return this._findClosestItem(targetPosition, this._items, itemAnchorPoint, firstIndex, distanceFromFirst, lastIndex, distanceFromLast);
572     },

574     /**
575      * Query the closest item to a specific position in current view.<br/>
576      * For instance, to find the item in the center of view, call 'getClosestItemToPositionInCurrentView(cc.p(0.5, 0.5), cc.p(0.5, 0.5))'.
577      *
578      * @param {cc.Point} positionRatioInView Specifies the target position with ratio in list view's content size.
579      * @param {cc.Point} itemAnchorPoint Specifies an anchor point of each item for position to calculate distance.
580      * @returns {?ccui.Widget} A item instance if list view is not empty. Otherwise, returns null.
581      */
582 
583     getClosestItemToPositionInCurrentView: function (positionRatioInView, itemAnchorPoint) {
584         // Calculate the target position
585         var contentSize = this.getContentSize();
586         var targetPosition = cc.pMult(this._innerContainer.getPosition(), -1);
587         targetPosition.x += contentSize.width * positionRatioInView.x;
588         targetPosition.y += contentSize.height * positionRatioInView.y;
589 
590         return this.getClosestItemToPosition(targetPosition, itemAnchorPoint);
591     },

649     _calculateItemDestination: function (positionRatioInView, item, itemAnchorPoint) {

714     /**
715      * Jump to specific item
716      * @param {number} itemIndex Specifies the item's index
717      * @param {cc.Point} positionRatioInView Specifies the position with ratio in list view's content size.
718      * @param {cc.Point} itemAnchorPoint Specifies an anchor point of each item for position to calculate distance.
719      */
720     jumpToItem: function (itemIndex, positionRatioInView, itemAnchorPoint) {
721         var item = this.getItem(itemIndex);
722 
723         if (!item)
724             return;
725 
726         this.doLayout();
727 
728         var destination = this._calculateItemDestination(positionRatioInView, item, itemAnchorPoint);
729 
730         if (!this.bounceEnabled) {
731             var delta = cc.pSub(destination, this._innerContainer.getPosition());
732             var outOfBoundary = this._getHowMuchOutOfBoundary(delta);
733             destination.x += outOfBoundary.x;
734             destination.y += outOfBoundary.y;
735         }
736 
737         this._jumpToDestination(destination);
738     },

741      * Scroll to specific item

743      * @param {cc.Point} positionRatioInView Specifies the position with ratio in list view's content size.
744      * @param {cc.Point} itemAnchorPoint Specifies an anchor point of each item for position to calculate distance.
745      * @param {number} [timeInSec = 1.0] Scroll time
746      */
747     scrollToItem: function (itemIndex, positionRatioInView, itemAnchorPoint, timeInSec) {
748         if (timeInSec === undefined)
749             timeInSec = 1;
750 
751         var item = this.getItem(itemIndex);
752 
753         if (!item)
754             return;
755 
756         var destination = this._calculateItemDestination(positionRatioInView, item, itemAnchorPoint);
757         this._startAutoScrollToDestination(destination, timeInSec, true);
758     },

834      * Intercept touch event, handle its child's touch event.
835      * @param {Number} eventType
836      * @param {ccui.Widget} sender
837      * @param {cc.Touch} touch
838      */
839     interceptTouchEvent: function (eventType, sender, touch) {
840         ccui.ScrollView.prototype.interceptTouchEvent.call(this, eventType, sender, touch);
841         if (!this._touchEnabled) {
842             return;
843         }
844         if (eventType !== ccui.Widget.TOUCH_MOVED) {
845             var parent = sender;
846             while (parent) {
847                 if (parent && parent.getParent() === this._innerContainer) {
848                     this._curSelectedIndex = this.getIndex(parent);
849                     break;
850                 }
851                 parent = parent.getParent();
852             }
853             if (sender.isHighlighted())
854                 this._selectedItemEvent(eventType);
855         }

903     _startAttenuatingAutoScroll: function (deltaMove, initialVelocity) {
904         var adjustedDeltaMove = deltaMove;
905 
906         if (this._items.length !== 0 && this._magneticType !== ccui.ListView.MAGNETIC_NONE) {
907             adjustedDeltaMove = this._flattenVectorByDirection(adjustedDeltaMove);
908 
909             var howMuchOutOfBoundary = this._getHowMuchOutOfBoundary(adjustedDeltaMove);
910             // If the destination is out of boundary, do nothing here. Because it will be handled by bouncing back.
911             if (howMuchOutOfBoundary.x === 0 && howMuchOutOfBoundary.y === 0) {
912                 var magType = this._magneticType;
913                 if (magType === ccui.ListView.MAGNETIC_BOTH_END) {
914                     if (this._direction === ccui.ScrollView.DIR_HORIZONTAL) {
915                         magType = (adjustedDeltaMove.x > 0 ? ccui.ListView.MAGNETIC_LEFT : ccui.ListView.MAGNETIC_RIGHT);
916                     }
917                     else if (this._direction === ccui.ScrollView.DIR_VERTICAL) {
918                         magType = (adjustedDeltaMove.y > 0 ? ccui.ListView.MAGNETIC_BOTTOM : ccui.ListView.MAGNETIC_TOP);
919                     }
920                 }
921 
922                 // Adjust the delta move amount according to the magnetic type
923                 var magneticAnchorPoint = this._getAnchorPointByMagneticType(magType);
924                 var magneticPosition = cc.pMult(this._innerContainer.getPosition(), -1);
925                 magneticPosition.x += this.width * magneticAnchorPoint.x;
926                 magneticPosition.y += this.height * magneticAnchorPoint.y;
927 
928                 var pTargetItem = this.getClosestItemToPosition(cc.pSub(magneticPosition, adjustedDeltaMove), magneticAnchorPoint);
929                 var itemPosition = this._calculateItemPositionWithAnchor(pTargetItem, magneticAnchorPoint);
930                 adjustedDeltaMove = cc.pSub(magneticPosition, itemPosition);
931             }
932         }
933         ccui.ScrollView.prototype._startAttenuatingAutoScroll.call(this, adjustedDeltaMove, initialVelocity);
934     },
935 
936     _getAnchorPointByMagneticType: function (magneticType) {
937         switch (magneticType) {
938             case ccui.ListView.MAGNETIC_NONE:
939                 return cc.p(0, 0);
940             case ccui.ListView.MAGNETIC_BOTH_END:
941                 return cc.p(0, 1);
942             case ccui.ListView.MAGNETIC_CENTER:
943                 return cc.p(0.5, 0.5);
944             case ccui.ListView.MAGNETIC_LEFT:
945                 return cc.p(0, 0.5);
946             case ccui.ListView.MAGNETIC_RIGHT:
947                 return cc.p(1, 0.5);
948             case ccui.ListView.MAGNETIC_TOP:
949                 return cc.p(0.5, 1);
950             case ccui.ListView.MAGNETIC_BOTTOM:
951                 return cc.p(0.5, 0);
952         }
953 
954         return cc.p(0, 0);
955     },
956 
957     _startMagneticScroll: function () {
958         if (this._items.length === 0 || this._magneticType === ccui.ListView.MAGNETIC_NONE) {
959             return;
960         }
961 
962         // Find the closest item
963         var magneticAnchorPoint = this._getAnchorPointByMagneticType(this._magneticType);
964         var magneticPosition = cc.pMult(this._innerContainer.getPosition(), -1);
965         magneticPosition.x += this.width * magneticAnchorPoint.x;
966         magneticPosition.y += this.height * magneticAnchorPoint.y;
967 
968         var pTargetItem = this.getClosestItemToPosition(magneticPosition, magneticAnchorPoint);
969         this.scrollToItem(this.getIndex(pTargetItem), magneticAnchorPoint, magneticAnchorPoint);
970     }
