 27 /**
 28  * using image file to print text label on the screen, might be a bit slower than cc.Label, similar to cc.LabelBMFont
 29  * @class
 30  * @extends cc.LabelBMFont
 31  *
 32  * @property {String}   string  - Content string of label
 33  *
 34  * @param {String} strText
 35  * @param {String} charMapFile  charMapFile or fntFile
 36  * @param {Number} [itemWidth=0]
 37  * @param {Number} [itemHeight=0]
 38  * @param {Number} [startCharMap=""]
 39  * @example
 40  * //creates the cc.LabelAtlas with a string, a char map file(the atlas), the width and height of each element and the starting char of the atlas
 41  * var myLabel = new cc.LabelAtlas('Text to display', 'CharMapfile.png', 12, 20, ' ')
 42  *
 43  * //creates the cc.LabelAtlas with a string, a fnt file
 44  * var myLabel = new cc.LabelAtlas('Text to display', 'CharMapFile.plistâ€˜);
 45  */
 46 cc.LabelAtlas = cc.LabelBMFont.extend(/** @lends cc.LabelBMFont# */{
 47     _className: "LabelAtlas",
 48 
 49     /**
 50      * <p>
 51      *  Constructor function, override it to extend the construction behavior, remember to call "this._super()" in the extended "ctor" function. <br />
 52      *  Create a label atlas. <br />
 53      *  It accepts two groups of parameters: <br/>
 54      * a) string, fntFile <br/>
 55      * b) label, textureFilename, width, height, startChar <br/>
 56      * </p>
 57      * @param {String} strText
 58      * @param {String} charMapFile  charMapFile or fntFile
 59      * @param {Number} [itemWidth=0]
 60      * @param {Number} [itemHeight=0]
 61      * @param {Number} [startCharMap=""]
 62      */
 63     ctor: function (strText, charMapFile, itemWidth, itemHeight, startCharMap) {
 64         cc.SpriteBatchNode.prototype.ctor.call(this);
 65         this._imageOffset = cc.p(0, 0);
 66         this._cascadeColorEnabled = true;
 67         this._cascadeOpacityEnabled = true;
 68 
 69         charMapFile && cc.LabelAtlas.prototype.initWithString.call(this, strText, charMapFile, itemWidth, itemHeight, startCharMap);
 70     },
 71 
 72     _createRenderCmd: function () {
 73         if (cc._renderType === cc.game.RENDER_TYPE_WEBGL)
 74             return new cc.LabelBMFont.WebGLRenderCmd(this);
 75         else
 76             return new cc.LabelBMFont.CanvasRenderCmd(this);
 77     },
 78 
 79     _createFntConfig: function (texture, itemWidth, itemHeight, startCharMap) {
 80         var fnt = {};
 81         fnt.commonHeight = itemHeight;
 82 
 83         var fontDefDictionary = fnt.fontDefDictionary = {};
 84 
 85         var textureWidth = texture.pixelsWidth;
 86         var textureHeight = texture.pixelsHeight;
 87 
 88         var startCharCode = startCharMap.charCodeAt(0);
 89         var i = 0;
 90         for (var col = itemHeight; col <= textureHeight; col += itemHeight) {
 91             for (var row = 0; row < textureWidth; row += itemWidth) {
 92                 fontDefDictionary[startCharCode+i] = {
 93                     rect: {x: row, y: col - itemHeight, width:itemWidth, height: itemHeight },
 94                     xOffset: 0,
 95                     yOffset: 0,
 96                     xAdvance: itemWidth
 97                 };
 98                 ++i;
 99             }
100         }
101 
102         fnt.kerningDict = {};
103 
104         return fnt;
105     },
106 
107     /**
108      * <p>
109      *  initializes the cc.LabelAtlas with a string, a char map file(the atlas), <br/>
110      *  the width and height of each element and the starting char of the atlas <br/>
111      *  It accepts two groups of parameters: <br/>
112      * a) string, fntFile <br/>
113      * b) label, textureFilename, width, height, startChar <br/>
114      * </p>
115      * @param {String} strText
116      * @param {String|cc.Texture2D} charMapFile  charMapFile or fntFile or texture file
117      * @param {Number} [itemWidth=0]
118      * @param {Number} [itemHeight=0]
119      * @param {Number} [startCharMap=""]
120      * @return {Boolean} returns true on success
121      */
122     initWithString: function (strText, charMapFile, itemWidth, itemHeight, startCharMap) {
123         var label = strText + "", textureFilename, width, height, startChar;
124         var self = this, theString = label || "";
125         this._initialString = theString;
126         self._string = theString;
127 
128         if (itemWidth === undefined) {
129             var dict = cc.loader.getRes(charMapFile);
130             if (parseInt(dict["version"], 10) !== 1) {
131                 cc.log("cc.LabelAtlas.initWithString(): Unsupported version. Upgrade cocos2d version");
132                 return false;
133             }
134 
135             textureFilename = cc.path.changeBasename(charMapFile, dict["textureFilename"]);
136             var locScaleFactor = cc.contentScaleFactor();
137             width = parseInt(dict["itemWidth"], 10) / locScaleFactor;
138             height = parseInt(dict["itemHeight"], 10) / locScaleFactor;
139             startChar = String.fromCharCode(parseInt(dict["firstChar"], 10));
140         } else {
141             textureFilename = charMapFile;
142             width = itemWidth || 0;
143             height = itemHeight || 0;
144             startChar = startCharMap || " ";
145         }
146 
147         var texture;
148         if (charMapFile) {
149             self._fntFile = "dummy_fnt_file:" + textureFilename;
150             var spriteFrameBaseName = textureFilename;
151             var spriteFrame = cc.spriteFrameCache.getSpriteFrame(spriteFrameBaseName) || cc.spriteFrameCache.getSpriteFrame(cc.path.basename(spriteFrameBaseName));
152             if(spriteFrame) {
153                 texture = spriteFrame.getTexture();
154                 this._spriteFrame = spriteFrame;
155             } else {
156                 texture = cc.textureCache.addImage(textureFilename);
157             }
158 
159             var newConf = this._createFntConfig(texture, width, height, startChar);
160             newConf.atlasName = textureFilename;
161             self._config = newConf;
162 
163             var locIsLoaded = texture.isLoaded();
164             self._textureLoaded = locIsLoaded;
165             if (!locIsLoaded) {
166                 texture.addEventListener("load", function (sender) {
167                     var self1 = this;
168                     self1._textureLoaded = true;
169                     //reset the LabelBMFont
170                     self1.initWithTexture(sender, self1._initialString.length);
171                     self1.setString(self1._initialString, true);
172                     self1.dispatchEvent("load");
173                 }, self);
174             }
175         } else {
176             texture = new cc.Texture2D();
177             var image = new Image();
178             texture.initWithElement(image);
179             self._textureLoaded = false;
180         }
181 
182         if (self.initWithTexture(texture, theString.length)) {
183             self._alignment = cc.TEXT_ALIGNMENT_LEFT;
184             self._imageOffset = cc.p(0, 0);
185             self._width = -1;
186 
187             self._realOpacity = 255;
188             self._realColor = cc.color(255, 255, 255, 255);
189 
190             self._contentSize.width = 0;
191             self._contentSize.height = 0;
192 
193             self.setString(theString, true);
194             return true;
195         }
196         return false;
197     },
198 
199     setFntFile: function () {
200         cc.warn("setFntFile doesn't support with LabelAtlas.");
201     }
202 
203 });
204 
205 /**
206  * <p>
207  *     Please use new cc.LabelAtlas instead. <br />
208  *     Create a label atlas. <br />
209  *     It accepts two groups of parameters:                                                            <br/>
210  *         a) string, fntFile                                                                               <br/>
211  *         b) label, textureFilename, width, height, startChar                                              <br/>
212  * </p>
213  * @deprecated since v3.0 please use new cc.LabelAtlas
214  * @param {String} strText
215  * @param {String} charMapFile  charMapFile or fntFile
216  * @param {Number} [itemWidth=0]
217  * @param {Number} [itemHeight=0]
218  * @param {Number} [startCharMap=""]
219  * @return {cc.LabelAtlas} returns the LabelAtlas object on success
220  */
221 cc.LabelAtlas.create = function (strText, charMapFile, itemWidth, itemHeight, startCharMap) {
222     return new cc.LabelAtlas(strText, charMapFile, itemWidth, itemHeight, startCharMap);
223 };
