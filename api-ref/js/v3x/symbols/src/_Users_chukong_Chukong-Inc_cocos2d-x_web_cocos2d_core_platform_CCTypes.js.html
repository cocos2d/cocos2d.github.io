 27 /**
 28  * Color class, please use cc.color() to construct a color
 29  * @class cc.Color
 30  * @param {Number} r
 31  * @param {Number} g
 32  * @param {Number} b
 33  * @param {Number} a
 34  * @see cc.color
 35  */
 36 cc.Color = function (r, g, b, a) {
 37     r = r || 0;
 38     g = g || 0;
 39     b = b || 0;
 40     a = a || 0;
 41     this._val = ((r << 24) >>> 0) + (g << 16) + (b << 8) + a;
 42 };
 43 
 44 var _p = cc.Color.prototype;
 45 _p._getR = function () {
 46     return (this._val & 0xff000000) >>> 24;
 47 };
 48 _p._setR = function (value) {
 49     this._val = (this._val & 0x00ffffff) | ((value << 24) >>> 0);
 50 };
 51 _p._getG = function () {
 52     return (this._val & 0x00ff0000) >> 16;
 53 };
 54 _p._setG = function (value) {
 55     this._val = (this._val & 0xff00ffff) | (value << 16);
 56 };
 57 _p._getB = function () {
 58     return (this._val & 0x0000ff00) >> 8;
 59 };
 60 _p._setB = function (value) {
 61     this._val = (this._val & 0xffff00ff) | (value << 8);
 62 };
 63 _p._getA = function () {
 64     return this._val & 0x000000ff;
 65 };
 66 _p._setA = function (value) {
 67     this._val = (this._val & 0xffffff00) | value;
 68 };

 84 /**
 85  * Generate a color object based on multiple forms of parameters
 86  * @example
 87  *
 88  * // 1. All channels separately as parameters
 89  * var color1 = cc.color(255, 255, 255, 255);
 90  *
 91  * // 2. Convert a hex string to a color
 92  * var color2 = cc.color("#000000");
 93  *
 94  * // 3. An color object as parameter
 95  * var color3 = cc.color({r: 255, g: 255, b: 255, a: 255});
 96  *
 97  * Alpha channel is optional. Default value is 255
 98  *
 99  * @param {Number|String|cc.Color} r
100  * @param {Number} [g]
101  * @param {Number} [b]
102  * @param {Number} [a=255]
103  * @return {cc.Color}
104  */
105 cc.color = function (r, g, b, a) {
106     if (r === undefined)
107         return new cc.Color(0, 0, 0, 255);
108     if (typeof r === 'object')
109         return new cc.Color(r.r, r.g, r.b, (r.a == null) ? 255 : r.a);
110     if (typeof r === 'string')
111         return cc.hexToColor(r);
112     return new cc.Color(r, g, b, (a == null ? 255 : a));
113 };

116  * returns true if both ccColor3B are equal. Otherwise it returns false.
117  * @function
118  * @param {cc.Color} color1
119  * @param {cc.Color} color2
120  * @return {Boolean}  true if both ccColor3B are equal. Otherwise it returns false.
121  */
122 cc.colorEqual = function (color1, color2) {
123     return color1.r === color2.r && color1.g === color2.g && color1.b === color2.b;
124 };

787  * @function
788  * @param {String} hex
789  * @return {cc.Color}
790  */
791 cc.hexToColor = function (hex) {
792     hex = hex.replace(/^#?/, "0x");
793     var c = parseInt(hex);
794     var r = c >> 16;
795     var g = (c >> 8) % 256;
796     var b = c % 256;
797     return new cc.Color(r, g, b);
