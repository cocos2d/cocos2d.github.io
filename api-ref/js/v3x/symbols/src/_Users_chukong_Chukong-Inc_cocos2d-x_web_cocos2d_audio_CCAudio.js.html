 85 cc.Audio = cc.Class.extend({
 86     interruptPlay: false,
 87     src: null,
 88     _element: null,
 89     _AUDIO_TYPE: "AUDIO",
 90 
 91     ctor: function (url) {
 92         this.src = url;
 93     },
 94 
 95     setBuffer: function (buffer) {
 96         this._AUDIO_TYPE = "WEBAUDIO";
 97         this._element = new cc.Audio.WebAudio(buffer);
 98     },
 99 
100     setElement: function (element) {
101         this._AUDIO_TYPE = "AUDIO";
102         this._element = element;
103 
104         // Prevent partial browser from playing after the end does not reset the paused tag
105         // Will cause the player to judge the status of the error
106         element.addEventListener('ended', function () {
107             if (!element.loop) {
108                 element.paused = true;
109             }
110         });
111     },
112 
113     play: function (offset, loop) {
114         if (!this._element) {
115             this.interruptPlay = false;
116             return;
117         }
118         this._element.loop = loop;
119         this._element.play();
120         if (this._AUDIO_TYPE === 'AUDIO' && this._element.paused) {
121             this.stop();
122             cc.Audio.touchPlayList.push({ loop: loop, offset: offset, audio: this._element });
123         }
124 
125         if (cc.Audio.bindTouch === false && this._element.paused) {
126             cc.Audio.bindTouch = true;
127             // Listen to the touchstart body event and play the audio when necessary.
128             cc.game.canvas.addEventListener('touchstart', cc.Audio.touchStart);
129         }
130     },
131 
132     getPlaying: function () {
133         if (!this._element) return true;
134         return !this._element.paused;
135     },
136 
137     stop: function () {
138         if (!this._element) {
139             this.interruptPlay = true;
140             return;
141         }
142         this._element.pause();
143         try {
144             this._element.currentTime = 0;
145         } catch (err) {
146         }
147     },
148 
149     pause: function () {
150         if (!this._element) {
151             this.interruptPlay = true;
152             return;
153         }
154         this._element.pause();
155     },
156 
157     resume: function () {
158         if (!this._element) {
159             this.interruptPlay = false;
160             return;
161         }
162         this._element.play();
163     },
164 
165     setVolume: function (volume) {
166         if (!this._element) return;
167         this._element.volume = volume;
168     },
169 
170     getVolume: function () {
171         if (!this._element) return;
172         return this._element.volume;
173     },
174 
175     cloneNode: function () {
176         var audio = new cc.Audio(this.src);
177         if (this._AUDIO_TYPE === "AUDIO") {
178             var elem = document.createElement("audio");
179             var sources = elem.getElementsByTagName('source');
180             for (var i = 0; i < sources.length; i++) {
181                 elem.appendChild(sources[i]);
182             }
183             elem.src = this.src;
184             audio.setElement(elem);
185         } else {
186             audio.setBuffer(this._element.buffer);
187         }
188         return audio;
189     }

206 cc.Audio.WebAudio = function (buffer) {
207     this.buffer = buffer;
208     this.context = cc.Audio._context;
209 
210     var volume = this.context['createGain']();
211     volume['gain'].value = 1;
212     volume['connect'](this.context['destination']);
213     this._volume = volume;
214 
215     this._loop = false;
216 
217     // The time stamp on the audio time axis when the recording begins to play.
218     this._startTime = -1;
219     // Record the currently playing Source
220     this._currentSource = null;
221     // Record the time has been played
222     this.playedLength = 0;
223 
224     this._currextTimer = null;
225 };

227 cc.Audio.WebAudio.prototype = {
228     constructor: cc.Audio.WebAudio,
229 
230     get paused() {
231         // If the current audio is a loop, then paused is false
232         if (this._currentSource && this._currentSource.loop)
233             return false;
234 
235         // StartTime does not have value, as the default -1, it does not begin to play
236         if (this._startTime === -1)
237             return true;
238 
239         // currentTime - startTime > durationTime
240         return this.context.currentTime - this._startTime > this.buffer.duration;
241     },
242     set paused(bool) {
243     },
244 
245     get loop() {
246         return this._loop;
247     },
248     set loop(bool) {
249         return this._loop = bool;
250     },
251 
252     get volume() {
253         return this._volume['gain'].value;
254     },
255     set volume(num) {
256         return this._volume['gain'].value = num;
257     },
258 
259     get currentTime() {
260         return this.playedLength;
261     },
262     set currentTime(num) {
263         return this.playedLength = num;
264     },
265 
266     play: function (offset) {
267 
268         // If repeat play, you need to stop before an audio
269         if (this._currentSource && !this.paused) {
270             this._currentSource.stop(0);
271             this.playedLength = 0;
272         }
273 
274         var audio = this.context["createBufferSource"]();
275         audio.buffer = this.buffer;
276         audio["connect"](this._volume);
277         audio.loop = this._loop;
278 
279         this._startTime = this.context.currentTime;
280         offset = offset || this.playedLength;
281 
282         var duration = this.buffer.duration;
283         if (!this._loop) {
284             if (audio.start)
285                 audio.start(0, offset, duration - offset);
286             else if (audio["notoGrainOn"])
287                 audio["noteGrainOn"](0, offset, duration - offset);
288             else
289                 audio["noteOn"](0, offset, duration - offset);
290         } else {
291             if (audio.start)
292                 audio.start(0);
293             else if (audio["notoGrainOn"])
294                 audio["noteGrainOn"](0);
295             else
296                 audio["noteOn"](0);
297         }
298 
299         this._currentSource = audio;
300 
301         // If the current audio context time stamp is 0
302         // There may be a need to touch events before you can actually start playing audio
303         // So here to add a timer to determine whether the real start playing audio, if not, then the incoming touchPlay queue
304         if (this.context.currentTime === 0) {
305             var self = this;
306             clearTimeout(this._currextTimer);
307             this._currextTimer = setTimeout(function () {
308                 if (self.context.currentTime === 0) {
309                     cc.Audio.touchPlayList.push({
310                         offset: offset,
311                         audio: self
312                     });
313                 }
314             }, 10);
315         }
316     },
317     pause: function () {
318         // Record the time the current has been played
319         this.playedLength = this.context.currentTime - this._startTime;
320         //If the duration of playedLendth exceeds the audio, you should take the remainder
321         this.playedLength %= this.buffer.duration;
322         var audio = this._currentSource;
323         this._currentSource = null;
324         this._startTime = -1;
325         if (audio)
326             audio.stop(0);
327     }
328 };
