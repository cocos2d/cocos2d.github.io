 26 /**
 27  * The text atlas control of Cocos UI.
 28  * @class
 29  * @extends ccui.Widget
 30  *
 31  * @property {String}   string  - Content string of the label
 32  */
 33 ccui.TextAtlas = ccui.Widget.extend(/** @lends ccui.TextAtlas# */{
 34     _labelAtlasRenderer: null,
 35     _stringValue: "",
 36     _charMapFileName: "",
 37     _itemWidth: 0,
 38     _itemHeight: 0,
 39     _startCharMap: "",
 40     _className: "TextAtlas",
 41     _labelAtlasRendererAdaptDirty: null,
 42 
 43     /**
 44      * Allocates and initializes a UILabelAtlas.                  <br/>
 45      * Constructor of ccui.TextAtlas, override it to extend the construction behavior, remember to call "this._super()" in the extended "ctor" function.
 46      * @param {String} stringValue
 47      * @param {String} charMapFile
 48      * @param {number} itemWidth
 49      * @param {number} itemHeight
 50      * @param {String} startCharMap
 51      * @example
 52      * // example
 53      * var uiLabelAtlas = new ccui.TextAtlas();
 54      */
 55     ctor: function (stringValue, charMapFile, itemWidth, itemHeight, startCharMap) {
 56         ccui.Widget.prototype.ctor.call(this);
 57         if (startCharMap !== undefined) {
 58             this.setProperty(stringValue, charMapFile, itemWidth, itemHeight, startCharMap);
 59         }
 60     },
 61 
 62     _initRenderer: function () {
 63         this._labelAtlasRenderer = new cc.LabelAtlas();
 64         this._labelAtlasRenderer.setAnchorPoint(cc.p(0.5, 0.5));
 65         this.addProtectedChild(this._labelAtlasRenderer, ccui.TextAtlas.RENDERER_ZORDER, -1);
 66 
 67         this._labelAtlasRenderer.addEventListener('load', function () {
 68             this._updateContentSizeWithTextureSize(this._labelAtlasRenderer.getContentSize());
 69             this._findLayout();
 70         }, this);
 71     },
 72 
 73     /**
 74      * initializes the UILabelAtlas with a string, a char map file(the atlas), the width and height of each element and the starting char of the atlas
 75      * @param {String} stringValue
 76      * @param {String} charMapFile
 77      * @param {number} itemWidth
 78      * @param {number} itemHeight
 79      * @param {String} startCharMap
 80      */
 81     setProperty: function (stringValue, charMapFile, itemWidth, itemHeight, startCharMap) {
 82         this._stringValue = stringValue;
 83         this._charMapFileName = charMapFile;
 84         this._itemWidth = itemWidth;
 85         this._itemHeight = itemHeight;
 86         this._startCharMap = startCharMap;
 87 
 88         this._labelAtlasRenderer.initWithString(
 89             stringValue,
 90             this._charMapFileName,
 91             this._itemWidth,
 92             this._itemHeight,
 93             this._startCharMap[0]
 94         );
 95 
 96         this._updateContentSizeWithTextureSize(this._labelAtlasRenderer.getContentSize());
 97         this._labelAtlasRendererAdaptDirty = true;
 98     },
 99 
100     /**
101      * Sets string value for ui text atlas.
102      * @param {String} value
103      */
104     setString: function (value) {
105         if (value === this._labelAtlasRenderer.getString())
106             return;
107         this._stringValue = value;
108         this._labelAtlasRenderer.setString(value);
109         this._updateContentSizeWithTextureSize(this._labelAtlasRenderer.getContentSize());
110         this._labelAtlasRendererAdaptDirty = true;
111     },
112 
113     /**
114      * Sets string value for text atlas.
115      * @deprecated since v3.0, please use setString instead.
116      * @param {String} value
117      */
118     setStringValue: function (value) {
119         cc.log("Please use the setString");
120         this.setString(value);
121     },
122 
123     /**
124      * get string value for text atlas.
125      * @deprecated since v3.0, please use getString instead.
126      * @returns {String}
127      */
128     getStringValue: function () {
129         cc.log("Please use the getString");
130         return this.getString();
131     },
132 
133     /**
134      * get string value for ui text atlas.
135      * @returns {String}
136      */
137     getString: function () {
138         return this._labelAtlasRenderer.getString();
139     },
140 
141     /**
142      * Returns the length of string.
143      * @returns {*|Number|long|int}
144      */
145     getStringLength: function () {
146         return this._labelAtlasRenderer.getStringLength();
147     },
148 
149     _onSizeChanged: function () {
150         ccui.Widget.prototype._onSizeChanged.call(this);
151         this._labelAtlasRendererAdaptDirty = true;
152     },
153 
154     _adaptRenderers: function () {
155         if (this._labelAtlasRendererAdaptDirty) {
156             this._labelAtlasScaleChangedWithSize();
157             this._labelAtlasRendererAdaptDirty = false;
158         }
159     },
160 
161     /**
162      * Returns the renderer's content size
163      * @overrider
164      * @returns {cc.Size}
165      */
166     getVirtualRendererSize: function () {
167         return this._labelAtlasRenderer.getContentSize();
168     },
169 
170     /**
171      * Returns the renderer of ccui.TextAtlas.
172      * @returns {cc.Node}
173      */
174     getVirtualRenderer: function () {
175         return this._labelAtlasRenderer;
176     },
177 
178     _labelAtlasScaleChangedWithSize: function () {
179         var locRenderer = this._labelAtlasRenderer;
180         if (this._ignoreSize) {
181             locRenderer.setScale(1.0);
182         } else {
183             var textureSize = locRenderer.getContentSize();
184             if (textureSize.width <= 0.0 || textureSize.height <= 0.0) {
185                 locRenderer.setScale(1.0);
186                 return;
187             }
188             locRenderer.setScaleX(this._contentSize.width / textureSize.width);
189             locRenderer.setScaleY(this._contentSize.height / textureSize.height);
190         }
191         locRenderer.setPosition(this._contentSize.width / 2.0, this._contentSize.height / 2.0);
192     },
193 
194     /**
195      * Returns the "class name" of ccui.TextAtlas.
196      * @returns {string}
197      */
198     getDescription: function () {
199         return "LabelAtlas";
200     },
201 
202     _copySpecialProperties: function (labelAtlas) {
203         if (labelAtlas) {
204             this.setProperty(labelAtlas._stringValue, labelAtlas._charMapFileName, labelAtlas._itemWidth, labelAtlas._itemHeight, labelAtlas._startCharMap);
205         }
206     },
207 
208     _createCloneInstance: function () {
209         return new ccui.TextAtlas();
210     }
211 });

215 // Extended properties
216 /** @expose */
217 _p.string;
218 cc.defineGetterSetter(_p, "string", _p.getString, _p.setString);
219 
220 _p = null;

222 /**
223  * allocates and initializes a UILabelAtlas.
224  * @deprecated since v3.0, please use new ccui.TextAtlas() instead.
225  * @return {ccui.TextAtlas}
226  */
227 ccui.TextAtlas.create = function (stringValue, charMapFile, itemWidth, itemHeight, startCharMap) {
228     return new ccui.TextAtlas(stringValue, charMapFile, itemWidth, itemHeight, startCharMap);
229 };
