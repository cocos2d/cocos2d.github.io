 28 /**
 29  * <p>
 30  *     A cc.SpriteBatchNode can reference one and only one texture (one image file, one texture atlas).<br/>
 31  *     Only the cc.Sprites that are contained in that texture can be added to the cc.SpriteBatchNode.<br/>
 32  *     All cc.Sprites added to a cc.SpriteBatchNode are drawn in one WebGL draw call. <br/>
 33  *     If the cc.Sprites are not added to a cc.SpriteBatchNode then an WebGL draw call will be needed for each one, which is less efficient. <br/>
 34  *     <br/>
 35  *     Limitations:<br/>
 36  *       - The only object that is accepted as child (or grandchild, grand-grandchild, etc...) is cc.Sprite or any subclass of cc.Sprite. <br/>
 37  *          eg: particles, labels and layer can't be added to a cc.SpriteBatchNode. <br/>
 38  *       - Either all its children are Aliased or Antialiased. It can't be a mix. <br/>
 39  *          This is because "alias" is a property of the texture, and all the sprites share the same texture. </br>
 40  * </p>
 41  * @class
 42  * @extends cc.Node
 43  *
 44  * @param {String|cc.Texture2D} fileImage
 45  * @example
 46  *
 47  * // 1. create a SpriteBatchNode with image path
 48  * var spriteBatchNode = new cc.SpriteBatchNode("res/animations/grossini.png");
 49  *
 50  * // 2. create a SpriteBatchNode with texture
 51  * var texture = cc.textureCache.addImage("res/animations/grossini.png");
 52  * var spriteBatchNode = new cc.SpriteBatchNode(texture);
 53  *
 54  * @property {cc.TextureAtlas}  textureAtlas    - The texture atlas
 55  * @property {Array}            descendants     - <@readonly> Descendants of sprite batch node
 56  */
 57 cc.SpriteBatchNode = cc.Node.extend(/** @lends cc.SpriteBatchNode# */{
 58     _blendFunc: null,
 59     // all descendants: chlidren, gran children, etc...
 60     _texture: null,
 61     _className: "SpriteBatchNode",
 62 
 63     ctor: function (fileImage) {
 64         cc.Node.prototype.ctor.call(this);
 65         this._blendFunc = new cc.BlendFunc(cc.BLEND_SRC, cc.BLEND_DST);
 66 
 67         var texture2D;
 68         if (cc.isString(fileImage)) {
 69             texture2D = cc.textureCache.getTextureForKey(fileImage);
 70             if (!texture2D)
 71                 texture2D = cc.textureCache.addImage(fileImage);
 72         } else if (fileImage instanceof cc.Texture2D)
 73             texture2D = fileImage;
 74 
 75         texture2D && this.initWithTexture(texture2D);
 76     },
 77 
 78     /**
 79      * <p>
 80      *    Same as addChild
 81      * </p>
 82      * @param {cc.Sprite} child
 83      * @param {Number} z zOrder
 84      * @param {Number} aTag
 85      * @return {cc.SpriteBatchNode}
 86      * @deprecated since v3.12
 87      */
 88     addSpriteWithoutQuad: function (child, z, aTag) {
 89         this.addChild(child, z, aTag);
 90         return this;
 91     },

101     },

120     /**
121      * <p>
122      *    Initializes a cc.SpriteBatchNode with a file image (.png, .jpeg, .pvr, etc) and a capacity of children.<br/>
123      *    The capacity will be increased in 33% in runtime if it run out of space.<br/>
124      *    The file will be loaded using the TextureMgr.<br/>
125      *    Please pass parameters to constructor to initialize the sprite batch node, do not call this function yourself.
126      * </p>
127      * @param {String} fileImage
128      * @param {Number} capacity
129      * @return {Boolean}
130      */
131     initWithFile: function (fileImage, capacity) {
132         var texture2D = cc.textureCache.getTextureForKey(fileImage);
133         if (!texture2D)
134             texture2D = cc.textureCache.addImage(fileImage);
135         return this.initWithTexture(texture2D, capacity);
136     },
137 
138     /**
139      * <p>
140      *    initializes a cc.SpriteBatchNode with a file image (.png, .jpeg, .pvr, etc) and a capacity of children.<br/>
141      *    The capacity will be increased in 33% in runtime if it run out of space.<br/>
142      *    The file will be loaded using the TextureMgr.<br/>
143      *    Please pass parameters to constructor to initialize the sprite batch node, do not call this function yourself.
144      * </p>
145      * @param {String} fileImage
146      * @param {Number} capacity
147      * @return {Boolean}
148      */
149     init: function (fileImage, capacity) {
150         var texture2D = cc.textureCache.getTextureForKey(fileImage);
151         if (!texture2D)
152             texture2D = cc.textureCache.addImage(fileImage);
153         return this.initWithTexture(texture2D, capacity);
154     },
155 
156     /**
157      * Do nothing
158      * @deprecated since v3.12
159      */
160     increaseAtlasCapacity: function () {
161     },
162 
163     /**
164      * Removes a child given a certain index. It will also cleanup the running actions depending on the cleanup parameter.
165      * @warning Removing a child from a cc.SpriteBatchNode is very slow
166      * @param {Number} index
167      * @param {Boolean} doCleanup
168      */
169     removeChildAtIndex: function (index, doCleanup) {
170         this.removeChild(this._children[index], doCleanup);
171     },
172 
173     /**
174      * Do nothing
175      * @param {cc.Sprite} pobParent
176      * @param {Number} index
177      * @return {Number}
178      * @deprecated since v3.12

180     rebuildIndexInOrder: function (pobParent, index) {
181         return index;
182     },

184     /**
185      * Returns highest atlas index in child
186      * @param {cc.Sprite} sprite
187      * @return {Number}
188      * @deprecated since v3.12
189      */
190     highestAtlasIndexInChild: function (sprite) {
191         var children = sprite.children;
192         if (!children || children.length === 0)
193             return sprite.zIndex;
194         else
195             return this.highestAtlasIndexInChild(children[children.length - 1]);
196     },

204     lowestAtlasIndexInChild: function (sprite) {
205         var children = sprite.children;
206         if (!children || children.length === 0)
207             return sprite.zIndex;
208         else
209             return this.lowestAtlasIndexInChild(children[children.length - 1]);
210     },

213      * Returns index for child
214      * @param {cc.Sprite} sprite
215      * @return {Number}
216      * @deprecated since v3.12

218     atlasIndexForChild: function (sprite) {
219         return sprite.zIndex;
220     },

223      * Sprites use this to start sortChildren, don't call this manually
224      * @param {Boolean} reorder

227     reorderBatch: function (reorder) {
228         this._reorderChildDirty = reorder;
229     },
230 
231     /**
232      * Sets the source and destination blending function for the texture
233      * @param {Number | cc.BlendFunc} src
234      * @param {Number} dst
235      */
236     setBlendFunc: function (src, dst) {
237         if (dst === undefined)
238             this._blendFunc = src;
239         else
240             this._blendFunc = {src: src, dst: dst};
241     },
242 
243     /**
244      * Returns the blending function used for the texture
245      * @return {cc.BlendFunc}
246      */
247     getBlendFunc: function () {
248         return new cc.BlendFunc(this._blendFunc.src, this._blendFunc.dst);
249     },
250 
251     /**
252      * <p>
253      *   Updates a quad at a certain index into the texture atlas. The CCSprite won't be added into the children array.                 <br/>
254      *   This method should be called only when you are dealing with very big AtlasSrite and when most of the cc.Sprite won't be updated.<br/>
255      *   For example: a tile map (cc.TMXMap) or a label with lots of characters (BitmapFontAtlas)<br/>
256      * </p>
257      * @function
258      * @param {cc.Sprite} sprite
259      * @param {Number} index
260      */
261     updateQuadFromSprite: function (sprite, index) {
262         cc.assert(sprite, cc._LogInfos.CCSpriteBatchNode_updateQuadFromSprite_2);
263         if (!(sprite instanceof cc.Sprite)) {
264             cc.log(cc._LogInfos.CCSpriteBatchNode_updateQuadFromSprite);
265             return;
266         }
267 
268         //
269         // update the quad directly. Don't add the sprite to the scene graph
270         //
271         sprite.dirty = true;
272         // UpdateTransform updates the textureAtlas quad
273         sprite._renderCmd.transform(this._renderCmd, true);
274     },
275 
276     /**
277      * <p>
278      *    Same as addChild(sprite, index)
279      * </p>
280      * @function
281      * @param {cc.Sprite} sprite
282      * @param {Number} index
283      * @deprecated since v3.12
284      */
285     insertQuadFromSprite: function (sprite, index) {
286         this.addChild(sprite, index);
287     },
288 
289     /**
290      * Same as addChild(sprite, index)
291      * @param {cc.Sprite} sprite The child sprite
292      * @param {Number} index The insert index
293      * @deprecated since v3.12
294      */
295     insertChild: function (sprite, index) {
296         this.addChild(sprite, index);
297     },
298 
299     /**
300      * Add child at the end
301      * @function
302      * @param {cc.Sprite} sprite
303      */
304     appendChild: function (sprite) {
305         this.sortAllChildren();
306         var lastLocalZOrder = this._children[this._children.length - 1]._localZOrder;
307         this.addChild(sprite.lastLocalZOrder + 1);
308     },
309 
310     /**
311      * Same as removeChild
312      * @function
313      * @param {cc.Sprite} sprite
314      * @param {Boolean} [cleanup=true]  true if all running actions and callbacks on the child node will be cleanup, false otherwise.
315      * @deprecated since v3.12
316      */
317     removeSpriteFromAtlas: function (sprite, cleanup) {
318         this.removeChild(sprite, cleanup);
319     },
320 
321     /**
322      * Set the texture property
323      * @function
324      * @param {cc.Texture2D} tex
325      * @return {Boolean}
326      */
327     initWithTexture: function (tex) {
328         this.setTexture(tex);
329         return true;
330     },
331 
332     // CCTextureProtocol
333     /**
334      * Returns texture of the sprite batch node
335      * @function
336      * @return {cc.Texture2D}
337      */
338     getTexture: function () {
339         return this._texture;
340     },
341 
342     /**
343      * Sets the texture of the sprite batch node.
344      * @function
345      * @param {cc.Texture2D} texture
346      */
347     setTexture: function (texture) {
348         this._texture = texture;
349 
350         if (texture._textureLoaded) {
351             var i, children = this._children, len = children.length;
352             for (i = 0; i < len; ++i) {
353                 children[i].setTexture(texture);
354             }
355         }
356         else {
357             texture.addEventListener("load", function () {
358                 var i, children = this._children, len = children.length;
359                 for (i = 0; i < len; ++i) {
360                     children[i].setTexture(texture);
361                 }
362             }, this);
363         }
364     },
365 
366     setShaderProgram: function (newShaderProgram) {
367         this._renderCmd.setShaderProgram(newShaderProgram);
368         var i, children = this._children, len = children.length;
369         for (i = 0; i < len; ++i) {
370             children[i].setShaderProgram(newShaderProgram);
371         }
372     },
373 
374     /**
375      * Add child to the sprite batch node (override addChild of cc.Node)
376      * @function
377      * @override
378      * @param {cc.Sprite} child
379      * @param {Number} [zOrder]
380      * @param {Number} [tag]
381      */
382     addChild: function (child, zOrder, tag) {
383         cc.assert(child !== undefined, cc._LogInfos.CCSpriteBatchNode_addChild_3);
384 
385         if (!this._isValidChild(child))
386             return;
387 
388         zOrder = (zOrder === undefined) ? child.zIndex : zOrder;
389         tag = (tag === undefined) ? child.tag : tag;
390         cc.Node.prototype.addChild.call(this, child, zOrder, tag);
391         
392         // Apply shader
393         if (this._renderCmd._shaderProgram) {
394             child.shaderProgram = this._renderCmd._shaderProgram;
395         }
396     },
397 
398     _isValidChild: function (child) {
399         if (!(child instanceof cc.Sprite)) {
400             cc.log(cc._LogInfos.Sprite_addChild_4);
401             return false;
402         }
403         if (child.texture !== this._texture) {
404             cc.log(cc._LogInfos.Sprite_addChild_5);
405             return false;
406         }
407         return true;
408     }
409 });
