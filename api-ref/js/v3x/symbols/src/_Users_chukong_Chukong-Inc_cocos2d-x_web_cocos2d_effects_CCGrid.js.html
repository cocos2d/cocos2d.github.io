 28 /**
 29  * Base class for cc.Grid
 30  * @class
 31  * @extends cc.Class
 32  */
 33 cc.GridBase = cc.Class.extend(/** @lends cc.GridBase# */{
 34     _active: false,
 35     _reuseGrid: 0,
 36     _gridSize: null,
 37     _gridRect: null,
 38     _texture: null,
 39     _step: null,
 40     _grabber: null,
 41     _isTextureFlipped: false,
 42     _shaderProgram: null,
 43     _directorProjection: 0,
 44 
 45     _dirty: false,
 46 
 47     /**
 48      * create one cc.GridBase Object
 49      * Constructor of cc.GridBase
 50      * @param {cc.Size} gridSize
 51      * @param {cc.Texture2D} [texture=]
 52      * @param {Boolean} [flipped=]
 53      * @param {cc.Rect} rect
 54      */
 55     ctor: function (gridSize, texture, flipped, rect) {
 56         cc.sys._checkWebGLRenderMode();
 57         this._active = false;
 58         this._reuseGrid = 0;
 59         this._gridSize = null;
 60         this._gridRect = new cc.rect();
 61         this._texture = null;
 62         this._step = cc.p(0, 0);
 63         this._grabber = null;
 64         this._isTextureFlipped = false;
 65         this._shaderProgram = null;
 66         this._directorProjection = 0;
 67         this._dirty = false;
 68 
 69         if (gridSize !== undefined)
 70             this.initWithSize(gridSize, texture, flipped, rect);
 71     },
 72 
 73     /**
 74      * whether or not the grid is active
 75      * @return {Boolean}
 76      */
 77     isActive: function () {
 78         return this._active;
 79     },
 80 
 81     /**
 82      * whether or not the grid is active
 83      * @param {Number} active
 84      */
 85     setActive: function (active) {
 86         this._active = active;
 87         if (!active) {
 88             var director = cc.director;
 89             var proj = director.getProjection();
 90             director.setProjection(proj);
 91         }
 92     },

102      * set number of times that the grid will be reused

167     /**
168      * set whether or not the texture is flipped
169      * @param {Boolean} flipped
170      */
171     setTextureFlipped: function (flipped) {
172         if (this._isTextureFlipped !== flipped) {
173             this._isTextureFlipped = flipped;
174             this.calculateVertexPoints();
175         }

178     /**
179      *
180      * @param {cc.Size} gridSize
181      * @param {cc.Texture2D} [texture=]
182      * @param {Boolean} [flipped=false]
183      * @param {cc.Rect} [rect=]
184      * @returns {boolean}
185      */
186     initWithSize: function (gridSize, texture, flipped, rect) {
187         if (!texture) {
188             var director = cc.director;
189             var winSize = director.getWinSizeInPixels();
190 
191             var POTWide = cc.NextPOT(winSize.width);
192             var POTHigh = cc.NextPOT(winSize.height);
193 
194             var data = new Uint8Array(POTWide * POTHigh * 4);
195             if (!data) {
196                 cc.log("cocos2d: CCGrid: not enough memory.");
197                 return false;
198             }
199 
200             texture = new cc.Texture2D();
201             // we only use rgba8888
202             texture.initWithData(data, cc.Texture2D.PIXEL_FORMAT_RGBA8888, POTWide, POTHigh, winSize);
203             if (!texture) {
204                 cc.log("cocos2d: CCGrid: error creating texture");
205                 return false;
206             }
207         }
208 
209         flipped = flipped || false;
210 
211         this._active = false;
212         this._reuseGrid = 0;
213         this._gridSize = gridSize;
214         this._texture = texture;
215         this._isTextureFlipped = flipped;
216         if (rect === undefined || cc._rectEqualToZero(rect)) {
217             var size = this._texture.getContentSize();
218             rect = new cc.rect(0, 0, size.width, size.height);
219         }
220 
221         this._gridRect = rect;
222 
223         this._step.x = this._gridRect.width / gridSize.width;
224         this._step.y = this._gridRect.height / gridSize.height;
225 
226         this._grabber = new cc.Grabber();
227         if (!this._grabber)
228             return false;
229         this._grabber.grab(this._texture);
230         this._shaderProgram = cc.shaderCache.programForKey(cc.SHADER_POSITION_TEXTURE);
231         this.calculateVertexPoints();
232         return true;
233     },
234 
235     beforeDraw: function () {
236         // save projection
237         this._directorProjection = cc.director.getProjection();
238 
239         //this.set2DProjection();    //TODO why?
240         var size = cc.director.getWinSizeInPixels();
241         gl.viewport(0, 0, size.width, size.height);
242         this._grabber.beforeRender(this._texture);
243     },
244 
245     afterDraw: function (target) {
246         this._grabber.afterRender(this._texture);
247 
248         // restore projection
249         //cc.director.setProjection(this._directorProjection);
250         cc.director.setViewport();
251 
252         cc.glBindTexture2D(this._texture);
253         this.beforeBlit();
254         this.blit(target);
255         this.afterBlit();
256     },
257 
258     beforeBlit: function () {
259     },
260 
261     afterBlit: function () {
262     },
263 
264     blit: function () {
265         cc.log("cc.GridBase.blit(): Shall be overridden in subclass.");
266     },
267 
268     reuse: function () {
269         cc.log("cc.GridBase.reuse(): Shall be overridden in subclass.");
270     },
271 
272     calculateVertexPoints: function () {
273         cc.log("cc.GridBase.calculateVertexPoints(): Shall be overridden in subclass.");
274     },
275 
276     set2DProjection: function () {
277         var winSize = cc.director.getWinSizeInPixels();
278 
279         var gl = cc._renderContext;
280         gl.viewport(0, 0, winSize.width, winSize.height);
281         cc.kmGLMatrixMode(cc.KM_GL_PROJECTION);
282         cc.kmGLLoadIdentity();
283 
284         var orthoMatrix = cc.math.Matrix4.createOrthographicProjection(0, winSize.width, 0, winSize.height, -1, 1);
285         cc.kmGLMultMatrix(orthoMatrix);
286 
287         cc.kmGLMatrixMode(cc.KM_GL_MODELVIEW);
288         cc.kmGLLoadIdentity();
289         cc.setProjectionMatrixDirty()
290     }
291 });

306 /**
307  * cc.Grid3D is a 3D grid implementation. Each vertex has 3 dimensions: x,y,z
308  * @class
309  * @extends cc.GridBase
310  */
311 cc.Grid3D = cc.GridBase.extend(/** @lends cc.Grid3D# */{
312     _texCoordinates: null,
313     _vertices: null,
314     _originalVertices: null,
315     _indices: null,
316 
317     _texCoordinateBuffer: null,
318     _verticesBuffer: null,
319     _indicesBuffer: null,
320 
321     _needDepthTestForBlit: false,
322     _oldDepthTestValue: false,
323     _oldDepthWriteValue: false,
324 
325     /**
326      * create one Grid3D object
327      * Constructor of cc.Grid3D
328      * @param {cc.Size} gridSize
329      * @param {cc.Texture2D} [texture=]
330      * @param {Boolean} [flipped=]
331      * @param {cc.Rect} [rect=]
332      */
333     ctor: function (gridSize, texture, flipped, rect) {
334         cc.GridBase.prototype.ctor.call(this);
335         this._texCoordinates = null;
336         this._vertices = null;
337         this._originalVertices = null;
338         this._indices = null;
339 
340         this._texCoordinateBuffer = null;
341         this._verticesBuffer = null;
342         this._indicesBuffer = null;
343 
344         this._matrix = new cc.math.Matrix4();
345         this._matrix.identity();
346 
347         if (gridSize !== undefined)
348             this.initWithSize(gridSize, texture, flipped, rect);
349     },
350 
351     /**
352      * returns the vertex at a given position      <br/>
353      * It will be deprecated in future, please use getVertex instead.
354      * @param {cc.Point} pos
355      * @return {cc.Vertex3F}
356      */
357     vertex: function (pos) {
358         return this.getVertex(pos);
359     },
360 
361     /**
362      * returns the vertex at a given position
363      * @param {cc.Point} pos
364      * @return {cc.Vertex3F}
365      */
366     getVertex: function (pos) {
367         if (pos.x !== (0 | pos.x) || pos.y !== (0 | pos.y))
368             cc.log("cc.Grid3D.vertex() : Numbers must be integers");
369         var index = 0 | ((pos.x * (this._gridSize.height + 1) + pos.y) * 3);
370         var locVertices = this._vertices;
371         return new cc.Vertex3F(locVertices[index], locVertices[index + 1], locVertices[index + 2]);
372     },
373 
374     /**
375      * returns the original (non-transformed) vertex at a given position             <br/>
376      * It will be deprecated in future, please use getOriginalVertex instead.
377      * @param {cc.Point} pos
378      * @return {cc.Vertex3F}
379      */
380     originalVertex: function (pos) {
381         return this.getOriginalVertex(pos);
382     },
383 
384     /**
385      * returns the original (non-transformed) vertex at a given position
386      * @param {cc.Point} pos
387      * @return {cc.Vertex3F}
388      */
389     getOriginalVertex: function (pos) {
390         if (pos.x !== (0 | pos.x) || pos.y !== (0 | pos.y))
391             cc.log("cc.Grid3D.originalVertex() : Numbers must be integers");
392         var index = 0 | ((pos.x * (this._gridSize.height + 1) + pos.y) * 3);
393         var locOriginalVertices = this._originalVertices;
394         return new cc.Vertex3F(locOriginalVertices[index], locOriginalVertices[index + 1], locOriginalVertices[index + 2]);
395     },
396 
397     /**
398      * sets a new vertex at a given position
399      * @param {cc.Point} pos
400      * @param {cc.Vertex3F} vertex
401      */
402     setVertex: function (pos, vertex) {
403         if (pos.x !== (0 | pos.x) || pos.y !== (0 | pos.y))
404             cc.log("cc.Grid3D.setVertex() : Numbers must be integers");
405         var index = 0 | ((pos.x * (this._gridSize.height + 1) + pos.y) * 3);
406         var vertArray = this._vertices;
407         vertArray[index] = vertex.x;
408         vertArray[index + 1] = vertex.y;
409         vertArray[index + 2] = vertex.z;
410         this._dirty = true;
411     },
412 
413     beforeBlit: function () {
414         if (this._needDepthTestForBlit) {
415             var gl = cc._renderContext;
416             this._oldDepthTestValue = gl.isEnabled(gl.DEPTH_TEST);
417             this._oldDepthWriteValue = gl.getParameter(gl.DEPTH_WRITEMASK);
418             //CHECK_GL_ERROR_DEBUG();
419             gl.enable(gl.DEPTH_TEST);
420             gl.depthMask(true);
421         }
422     },
423 
424     afterBlit: function () {
425         if (this._needDepthTestForBlit) {
426             var gl = cc._renderContext;
427             if (this._oldDepthTestValue)
428                 gl.enable(gl.DEPTH_TEST);
429             else
430                 gl.disable(gl.DEPTH_TEST);
431             gl.depthMask(this._oldDepthWriteValue);
432         }
433     },
434 
435     blit: function (target) {
436         var n = this._gridSize.width * this._gridSize.height;
437 
438         var wt = target._renderCmd._worldTransform;
439         this._matrix.mat[0] = wt.a;
440         this._matrix.mat[4] = wt.c;
441         this._matrix.mat[12] = wt.tx;
442         this._matrix.mat[1] = wt.b;
443         this._matrix.mat[5] = wt.d;
444         this._matrix.mat[13] = wt.ty;
445 
446         this._shaderProgram.use();
447         //this._shaderProgram.setUniformsForBuiltins();
448         this._shaderProgram._setUniformForMVPMatrixWithMat4(this._matrix);
449 
450         var gl = cc._renderContext, locDirty = this._dirty;
451 
452         gl.enableVertexAttribArray(cc.VERTEX_ATTRIB_POSITION);
453         gl.enableVertexAttribArray(cc.VERTEX_ATTRIB_TEX_COORDS);
454         //
455         // Attributes
456         //
457         // position
458         gl.bindBuffer(gl.ARRAY_BUFFER, this._verticesBuffer);
459         if (locDirty)
460             gl.bufferData(gl.ARRAY_BUFFER, this._vertices, gl.DYNAMIC_DRAW);
461         gl.vertexAttribPointer(cc.VERTEX_ATTRIB_POSITION, 3, gl.FLOAT, false, 0, 0);
462 
463         // texCoords
464         gl.bindBuffer(gl.ARRAY_BUFFER, this._texCoordinateBuffer);
465         if (locDirty)
466             gl.bufferData(gl.ARRAY_BUFFER, this._texCoordinates, gl.DYNAMIC_DRAW);
467         gl.vertexAttribPointer(cc.VERTEX_ATTRIB_TEX_COORDS, 2, gl.FLOAT, false, 0, 0);
468 
469         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this._indicesBuffer);
470         if (locDirty)
471             gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, this._indices, gl.STATIC_DRAW);
472         gl.drawElements(gl.TRIANGLES, n * 6, gl.UNSIGNED_SHORT, 0);
473         if (locDirty)
474             this._dirty = false;
475         cc.incrementGLDraws(1);
476     },
477 
478     reuse: function () {
479         if (this._reuseGrid > 0) {
480             var locOriginalVertices = this._originalVertices, locVertices = this._vertices;
481             for (var i = 0, len = this._vertices.length; i < len; i++)
482                 locOriginalVertices[i] = locVertices[i];
483             --this._reuseGrid;
484         }
485     },
486 
487     calculateVertexPoints: function () {
488         var gl = cc._renderContext;
489 
490         var width = this._texture.pixelsWidth;
491         var height = this._texture.pixelsHeight;
492         var imageH = this._texture.getContentSizeInPixels().height;
493         var locGridSize = this._gridSize;
494 
495         var numOfPoints = (locGridSize.width + 1) * (locGridSize.height + 1);
496         this._vertices = new Float32Array(numOfPoints * 3);
497         this._texCoordinates = new Float32Array(numOfPoints * 2);
498         this._indices = new Uint16Array(locGridSize.width * locGridSize.height * 6);
499 
500         if (this._verticesBuffer)
501             gl.deleteBuffer(this._verticesBuffer);
502         this._verticesBuffer = gl.createBuffer();
503         if (this._texCoordinateBuffer)
504             gl.deleteBuffer(this._texCoordinateBuffer);
505         this._texCoordinateBuffer = gl.createBuffer();
506         if (this._indicesBuffer)
507             gl.deleteBuffer(this._indicesBuffer);
508         this._indicesBuffer = gl.createBuffer();
509 
510         var x, y, i, locIndices = this._indices, locTexCoordinates = this._texCoordinates;
511         var locIsTextureFlipped = this._isTextureFlipped, locVertices = this._vertices;
512         for (x = 0; x < locGridSize.width; ++x) {
513             for (y = 0; y < locGridSize.height; ++y) {
514                 var idx = (y * locGridSize.width) + x;
515                 var x1 = x * this._step.x + this._gridRect.x;
516                 var x2 = x1 + this._step.x;
517                 var y1 = y * this._step.y + this._gridRect.y;
518                 var y2 = y1 + this._step.y;
519 
520                 var a = (x * (locGridSize.height + 1) + y);
521                 var b = ((x + 1) * (locGridSize.height + 1) + y);
522                 var c = ((x + 1) * (locGridSize.height + 1) + (y + 1));
523                 var d = (x * (locGridSize.height + 1) + (y + 1));
524 
525                 locIndices[idx * 6] = a;
526                 locIndices[idx * 6 + 1] = b;
527                 locIndices[idx * 6 + 2] = d;
528                 locIndices[idx * 6 + 3] = b;
529                 locIndices[idx * 6 + 4] = c;
530                 locIndices[idx * 6 + 5] = d;
531 
532                 var l1 = [a * 3, b * 3, c * 3, d * 3];
533                 var e = {x: x1, y: y1, z: 0};   //new cc.Vertex3F(x1, y1, 0);
534                 var f = {x: x2, y: y1, z: 0};   //new cc.Vertex3F(x2, y1, 0);
535                 var g = {x: x2, y: y2, z: 0};   // new cc.Vertex3F(x2, y2, 0);
536                 var h = {x: x1, y: y2, z: 0};   //new cc.Vertex3F(x1, y2, 0);
537 
538                 var l2 = [e, f, g, h];
539                 var tex1 = [a * 2, b * 2, c * 2, d * 2];
540                 var tex2 = [cc.p(x1, y1), cc.p(x2, y1), cc.p(x2, y2), cc.p(x1, y2)];
541                 for (i = 0; i < 4; ++i) {
542                     locVertices[l1[i]] = l2[i].x;
543                     locVertices[l1[i] + 1] = l2[i].y;
544                     locVertices[l1[i] + 2] = l2[i].z;
545                     locTexCoordinates[tex1[i]] = tex2[i].x / width;
546                     if (locIsTextureFlipped)
547                         locTexCoordinates[tex1[i] + 1] = (imageH - tex2[i].y) / height;
548                     else
549                         locTexCoordinates[tex1[i] + 1] = tex2[i].y / height;
550                 }
551             }
552         }
553         this._originalVertices = new Float32Array(this._vertices);
554 
555         gl.bindBuffer(gl.ARRAY_BUFFER, this._verticesBuffer);
556         gl.bufferData(gl.ARRAY_BUFFER, this._vertices, gl.DYNAMIC_DRAW);
557         gl.bindBuffer(gl.ARRAY_BUFFER, this._texCoordinateBuffer);
558         gl.bufferData(gl.ARRAY_BUFFER, this._texCoordinates, gl.DYNAMIC_DRAW);
559         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this._indicesBuffer);
560         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, this._indices, gl.STATIC_DRAW);
561         this._dirty = true;
562     },
563 
564     setNeedDepthTestForBlit: function (needDepthTest) {
565         this._needDepthTestForBlit = needDepthTest;
566     },
567 
568     getNeedDepthTestForBlit: function () {
569         return this._needDepthTestForBlit;
570     }
571 });

605      * @param {cc.Texture2D} [texture=]
606      * @param {Boolean} [flipped=]
607      */
608     ctor: function (gridSize, texture, flipped, rect) {
609         cc.GridBase.prototype.ctor.call(this);
610         this._texCoordinates = null;
611         this._vertices = null;
612         this._originalVertices = null;
613         this._indices = null;
614 
615         this._texCoordinateBuffer = null;
616         this._verticesBuffer = null;
617         this._indicesBuffer = null;
618 
619         this._matrix = new cc.math.Matrix4();
620         this._matrix.identity();
621 
622         if (gridSize !== undefined)
623             this.initWithSize(gridSize, texture, flipped, rect);
624     },
625 
626     /**
627      * returns the tile at the given position    <br/>
628      * It will be deprecated in future, please use getTile instead.
629      * @param {cc.Point} pos
630      * @return {cc.Quad3}
631      */
632     tile: function (pos) {
633         return this.getTile(pos);

635 
636     /**
637      * returns the tile at the given position
638      * @param {cc.Point} pos
639      * @return {cc.Quad3}
640      */
641     getTile: function (pos) {
642         if (pos.x !== (0 | pos.x) || pos.y !== (0 | pos.y))
643             cc.log("cc.TiledGrid3D.tile() : Numbers must be integers");
644 
645         var idx = (this._gridSize.height * pos.x + pos.y) * 4 * 3;
646         var locVertices = this._vertices;
647         return new cc.Quad3(new cc.Vertex3F(locVertices[idx], locVertices[idx + 1], locVertices[idx + 2]),
648             new cc.Vertex3F(locVertices[idx + 3], locVertices[idx + 4], locVertices[idx + 5]),
649             new cc.Vertex3F(locVertices[idx + 6], locVertices[idx + 7], locVertices[idx + 8]),
650             new cc.Vertex3F(locVertices[idx + 9], locVertices[idx + 10], locVertices[idx + 11]));
651     },
652 
653     /**
654      * returns the original tile (untransformed) at the given position
655      * @param {cc.Point} pos
656      * @return {cc.Quad3}
657      */
658     getOriginalTile: function (pos) {
659         if (pos.x !== (0 | pos.x) || pos.y !== (0 | pos.y))
660             cc.log("cc.TiledGrid3D.originalTile() : Numbers must be integers");
661 
662         var idx = (this._gridSize.height * pos.x + pos.y) * 4 * 3;
663         var locOriginalVertices = this._originalVertices;
664         return new cc.Quad3(new cc.Vertex3F(locOriginalVertices[idx], locOriginalVertices[idx + 1], locOriginalVertices[idx + 2]),
665             new cc.Vertex3F(locOriginalVertices[idx + 3], locOriginalVertices[idx + 4], locOriginalVertices[idx + 5]),
666             new cc.Vertex3F(locOriginalVertices[idx + 6], locOriginalVertices[idx + 7], locOriginalVertices[idx + 8]),
667             new cc.Vertex3F(locOriginalVertices[idx + 9], locOriginalVertices[idx + 10], locOriginalVertices[idx + 11]));
668     },
669 
670     /**
671      * returns the original tile (untransformed) at the given position.      <br/>
672      * It will be deprecated in future, please use getOriginalTile instead.
673      * @param {cc.Point} pos
674      * @return {cc.Quad3}
675      */
676     originalTile: function (pos) {
677         return this.getOriginalTile(pos);
678     },
679 
680     /**
681      * sets a new tile
682      * @param {cc.Point} pos
683      * @param {cc.Quad3} coords
684      */
685     setTile: function (pos, coords) {
686         if (pos.x !== (0 | pos.x) || pos.y !== (0 | pos.y))
687             cc.log("cc.TiledGrid3D.setTile() : Numbers must be integers");
688 
689         var idx = (this._gridSize.height * pos.x + pos.y) * 12;
690         var locVertices = this._vertices;
691         locVertices[idx] = coords.bl.x;
692         locVertices[idx + 1] = coords.bl.y;
693         locVertices[idx + 2] = coords.bl.z;
694         locVertices[idx + 3] = coords.br.x;
695         locVertices[idx + 4] = coords.br.y;
696         locVertices[idx + 5] = coords.br.z;
697         locVertices[idx + 6] = coords.tl.x;
698         locVertices[idx + 7] = coords.tl.y;
699         locVertices[idx + 8] = coords.tl.z;
700         locVertices[idx + 9] = coords.tr.x;
701         locVertices[idx + 10] = coords.tr.y;
702         locVertices[idx + 11] = coords.tr.z;
703         this._dirty = true;
704     },
705 
706     blit: function (target) {
707         var n = this._gridSize.width * this._gridSize.height;
708 
709         var wt = target._renderCmd._worldTransform;
710         this._matrix.mat[0] = wt.a;
711         this._matrix.mat[4] = wt.c;
712         this._matrix.mat[12] = wt.tx;
713         this._matrix.mat[1] = wt.b;
714         this._matrix.mat[5] = wt.d;
715         this._matrix.mat[13] = wt.ty;
716 
717         this._shaderProgram.use();
718         this._shaderProgram._setUniformForMVPMatrixWithMat4(this._matrix);
719         //this._shaderProgram.setUniformsForBuiltins();
720 
721         //
722         // Attributes
723         //
724         var gl = cc._renderContext, locDirty = this._dirty;
725         gl.enableVertexAttribArray(cc.VERTEX_ATTRIB_POSITION);
726         gl.enableVertexAttribArray(cc.VERTEX_ATTRIB_TEX_COORDS);
727 
728         // position
729         gl.bindBuffer(gl.ARRAY_BUFFER, this._verticesBuffer);
730         if (locDirty)
731             gl.bufferData(gl.ARRAY_BUFFER, this._vertices, gl.DYNAMIC_DRAW);
732         gl.vertexAttribPointer(cc.VERTEX_ATTRIB_POSITION, 3, gl.FLOAT, false, 0, this._vertices);
733 
734         // texCoords
735         gl.bindBuffer(gl.ARRAY_BUFFER, this._texCoordinateBuffer);
736         if (locDirty)
737             gl.bufferData(gl.ARRAY_BUFFER, this._texCoordinates, gl.DYNAMIC_DRAW);
738         gl.vertexAttribPointer(cc.VERTEX_ATTRIB_TEX_COORDS, 2, gl.FLOAT, false, 0, this._texCoordinates);
739 
740         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this._indicesBuffer);
741         if (locDirty)
742             gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, this._indices, gl.STATIC_DRAW);
743         gl.drawElements(gl.TRIANGLES, n * 6, gl.UNSIGNED_SHORT, 0);
744         if (locDirty)
745             this._dirty = false;
746         cc.incrementGLDraws(1);
747     },
748 
749     reuse: function () {
750         if (this._reuseGrid > 0) {
751             var locVertices = this._vertices, locOriginalVertices = this._originalVertices;
752             for (var i = 0; i < locVertices.length; i++)
753                 locOriginalVertices[i] = locVertices[i];
754             --this._reuseGrid;
755         }
756     },
757 
758     calculateVertexPoints: function () {
759         var width = this._texture.pixelsWidth;
760         var height = this._texture.pixelsHeight;
761         var imageH = this._texture.getContentSizeInPixels().height;
762         var locGridSize = this._gridSize;
763 
764         var numQuads = locGridSize.width * locGridSize.height;
765         this._vertices = new Float32Array(numQuads * 12);
766         this._texCoordinates = new Float32Array(numQuads * 8);
767         this._indices = new Uint16Array(numQuads * 6);
768 
769         var gl = cc._renderContext;
770         if (this._verticesBuffer)
771             gl.deleteBuffer(this._verticesBuffer);
772         this._verticesBuffer = gl.createBuffer();
773         if (this._texCoordinateBuffer)
774             gl.deleteBuffer(this._texCoordinateBuffer);
775         this._texCoordinateBuffer = gl.createBuffer();
776         if (this._indicesBuffer)
777             gl.deleteBuffer(this._indicesBuffer);
778         this._indicesBuffer = gl.createBuffer();
779 
780         var x, y, i = 0;
781         var locStep = this._step, locVertices = this._vertices, locTexCoords = this._texCoordinates, locIsTextureFlipped = this._isTextureFlipped;
782         for (x = 0; x < locGridSize.width; x++) {
783             for (y = 0; y < locGridSize.height; y++) {
784                 var x1 = x * locStep.x;
785                 var x2 = x1 + locStep.x;
786                 var y1 = y * locStep.y;
787                 var y2 = y1 + locStep.y;
788 
789                 locVertices[i * 12] = x1;
790                 locVertices[i * 12 + 1] = y1;
791                 locVertices[i * 12 + 2] = 0;
792                 locVertices[i * 12 + 3] = x2;
793                 locVertices[i * 12 + 4] = y1;
794                 locVertices[i * 12 + 5] = 0;
795                 locVertices[i * 12 + 6] = x1;
796                 locVertices[i * 12 + 7] = y2;
797                 locVertices[i * 12 + 8] = 0;
798                 locVertices[i * 12 + 9] = x2;
799                 locVertices[i * 12 + 10] = y2;
800                 locVertices[i * 12 + 11] = 0;
801 
802                 var newY1 = y1;
803                 var newY2 = y2;
804 
805                 if (locIsTextureFlipped) {
806                     newY1 = imageH - y1;
807                     newY2 = imageH - y2;
808                 }
809 
810                 locTexCoords[i * 8] = x1 / width;
811                 locTexCoords[i * 8 + 1] = newY1 / height;
812                 locTexCoords[i * 8 + 2] = x2 / width;
813                 locTexCoords[i * 8 + 3] = newY1 / height;
814                 locTexCoords[i * 8 + 4] = x1 / width;
815                 locTexCoords[i * 8 + 5] = newY2 / height;
816                 locTexCoords[i * 8 + 6] = x2 / width;
817                 locTexCoords[i * 8 + 7] = newY2 / height;
818                 i++;
819             }
820         }
821 
822         var locIndices = this._indices;
823         for (x = 0; x < numQuads; x++) {
824             locIndices[x * 6 + 0] = (x * 4 + 0);
825             locIndices[x * 6 + 1] = (x * 4 + 1);
826             locIndices[x * 6 + 2] = (x * 4 + 2);
827 
828             locIndices[x * 6 + 3] = (x * 4 + 1);
829             locIndices[x * 6 + 4] = (x * 4 + 2);
830             locIndices[x * 6 + 5] = (x * 4 + 3);
831         }
832         this._originalVertices = new Float32Array(this._vertices);
833 
834         gl.bindBuffer(gl.ARRAY_BUFFER, this._verticesBuffer);
835         gl.bufferData(gl.ARRAY_BUFFER, this._vertices, gl.DYNAMIC_DRAW);
836         gl.bindBuffer(gl.ARRAY_BUFFER, this._texCoordinateBuffer);
837         gl.bufferData(gl.ARRAY_BUFFER, this._texCoordinates, gl.DYNAMIC_DRAW);
838         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this._indicesBuffer);
839         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, this._indices, gl.DYNAMIC_DRAW);
840         this._dirty = true;
841     }
842 });
