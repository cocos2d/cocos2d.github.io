 25 /**
 26  * @class
 27  * @extends ccui.Widget
 28  * @brief A View that displays web pages.
 29  *
 30  * @note WebView displays web pages based on DOM element
 31  * WebView will be rendered above all other graphical elements.
 32  *
 33  * @property {String}   path - The url to be shown in the web view
 34  */
 35 ccui.WebView = ccui.Widget.extend(/** @lends ccui.WebView# */{
 36 
 37     ctor: function (path) {
 38         ccui.Widget.prototype.ctor.call(this);
 39         this._EventList = {};
 40         if (path)
 41             this.loadURL(path);
 42     },
 43 
 44     visit: function () {
 45         var cmd = this._renderCmd,
 46             div = cmd._div,
 47             container = cc.container,
 48             eventManager = cc.eventManager;
 49         if (this._visible) {
 50             container.appendChild(div);
 51             if (this._listener === null)
 52                 this._listener = eventManager.addCustomListener(cc.game.EVENT_RESIZE, function () {
 53                     cmd.resize();
 54                 });
 55         } else {
 56             var hasChild = false;
 57             if ('contains' in container) {
 58                 hasChild = container.contains(div);
 59             } else {
 60                 hasChild = container.compareDocumentPosition(div) % 16;
 61             }
 62             if (hasChild)
 63                 container.removeChild(div);
 64             var list = eventManager._listenersMap[cc.game.EVENT_RESIZE].getFixedPriorityListeners();
 65             eventManager._removeListenerInVector(list, cmd._listener);
 66             cmd._listener = null;
 67         }
 68         cmd.updateStatus();
 69         cmd.resize(cc.view);
 70     },
 71 
 72     setJavascriptInterfaceScheme: function (scheme) {
 73     },
 74     loadData: function (data, MIMEType, encoding, baseURL) {
 75     },
 76     loadHTMLString: function (string, baseURL) {
 77     },
 78 
 79 
 80     /**
 81      * Load an URL
 82      * @param {String} url
 83      */
 84     loadURL: function (url) {
 85         this._renderCmd.updateURL(url);
 86         this._dispatchEvent(ccui.WebView.EventType.LOADING);
 87     },
 88 
 89     /**
 90      * Stop loading
 91      */
 92     stopLoading: function () {
 93         cc.log("Web does not support loading");
 94     },
 95 
 96     /**
 97      * Reload the WebView
 98      */
 99     reload: function () {
100         var iframe = this._renderCmd._iframe;
101         if (iframe) {
102             var win = iframe.contentWindow;
103             if (win && win.location)
104                 win.location.reload();
105         }
106     },
107 
108     /**
109      * Determine whether to go back
110      */
111     canGoBack: function () {
112         cc.log("Web does not support query history");
113         return true;
114     },
115 
116     /**
117      * Determine whether to go forward
118      */
119     canGoForward: function () {
120         cc.log("Web does not support query history");
121         return true;
122     },
123 
124     /**
125      * go back
126      */
127     goBack: function () {
128         try {
129             if (ccui.WebView._polyfill.closeHistory)
130                 return cc.log("The current browser does not support the GoBack");
131             var iframe = this._renderCmd._iframe;
132             if (iframe) {
133                 var win = iframe.contentWindow;
134                 if (win && win.location)
135                     try {
136                         win.history.back.call(win);
137                     } catch (error) {
138                         win.history.back();
139                     }
140             }
141         } catch (err) {
142             cc.log(err);
143         }
144     },
145 
146     /**
147      * go forward
148      */
149     goForward: function () {
150         try {
151             if (ccui.WebView._polyfill.closeHistory)
152                 return cc.log("The current browser does not support the GoForward");
153             var iframe = this._renderCmd._iframe;
154             if (iframe) {
155                 var win = iframe.contentWindow;
156                 if (win && win.location)
157                     try {
158                         win.history.forward.call(win);
159                     } catch (error) {
160                         win.history.forward();
161                     }
162             }
163         } catch (err) {
164             cc.log(err);
165         }
166     },
167 
168     /**
169      * In the webview execution within a period of js string
170      * @param {String} str
171      */
172     evaluateJS: function (str) {
173         var iframe = this._renderCmd._iframe;
174         if (iframe) {
175             var win = iframe.contentWindow;
176             try {
177                 win.eval(str);
178                 this._dispatchEvent(ccui.WebView.EventType.JS_EVALUATED);
179             } catch (err) {
180                 console.error(err);
181             }
182         }
183     },
184 
185     /**
186      * Limited scale
187      */
188     setScalesPageToFit: function () {
189         cc.log("Web does not support zoom");
190     },
191 
192     /**
193      * The binding event
194      * @param {ccui.WebView.EventType} event
195      * @param {Function} callback
196      */
197     setEventListener: function (event, callback) {
198         this._EventList[event] = callback;
199     },
200 
201     /**
202      * Delete events
203      * @param {ccui.WebView.EventType} event
204      */
205     removeEventListener: function (event) {
206         this._EventList[event] = null;
207     },
208 
209     _dispatchEvent: function (event) {
210         var callback = this._EventList[event];
211         if (callback)
212             callback.call(this, this, this._renderCmd._iframe.src);
213     },
214 
215     _createRenderCmd: function () {
216         return new ccui.WebView.RenderCmd(this);
217     },
218 
219     /**
220      * Set the contentSize
221      * @param {Number} w
222      * @param {Number} h
223      */
224     setContentSize: function (w, h) {
225         ccui.Widget.prototype.setContentSize.call(this, w, h);
226         if (h === undefined) {
227             h = w.height;
228             w = w.width;
229         }
230         this._renderCmd.changeSize(w, h);
231     },
232 
233     /**
234      * remove node
235      */
236     cleanup: function () {
237         this._renderCmd.removeDom();
238         this.stopAllActions();
239         this.unscheduleAllCallbacks();
240     }
241 });

279     var RenderCmd = null;
280     if (cc._renderType === cc.game.RENDER_TYPE_WEBGL) {
281         RenderCmd = cc.Node.WebGLRenderCmd;

286     ccui.WebView.RenderCmd = function (node) {
287         this._rootCtor(node);
288 
289         this._div = null;
290         this._iframe = null;
291 
292         if (polyfill.enableDiv) {
293             this._div = document.createElement("div");
294             this._div.style["-webkit-overflow"] = "auto";
295             this._div.style["-webkit-overflow-scrolling"] = "touch";
296             this._iframe = document.createElement("iframe");
297             this._iframe.style["width"] = "100%";
298             this._iframe.style["height"] = "100%";
299             this._div.appendChild(this._iframe);
300         } else {
301             this._div = this._iframe = document.createElement("iframe");
302         }
303 
304         if (polyfill.enableBG)
305             this._div.style["background"] = "#FFF";
306 
307         this._iframe.addEventListener("load", function () {
308             node._dispatchEvent(ccui.WebView.EventType.LOADED);
309         });
310         this._iframe.addEventListener("error", function () {
311             node._dispatchEvent(ccui.WebView.EventType.ERROR);
312         });
313         this._div.style["background"] = "#FFF";
314         this._div.style.height = "200px";
315         this._div.style.width = "300px";
316         this._div.style.overflow = "scroll";
317         this._div.style["border"] = "none";
318         this._listener = null;
319         this.initStyle();
320     };

322     var proto = ccui.WebView.RenderCmd.prototype = Object.create(RenderCmd.prototype);
323     proto.constructor = ccui.WebView.RenderCmd;
324 
325     proto.transform = function (parentCmd, recursive) {
326         this.originTransform(parentCmd, recursive);
327         this.updateMatrix(this._worldTransform, cc.view._scaleX, cc.view._scaleY);
328     };
329 
330     proto.updateStatus = function () {
331         polyfill.devicePixelRatio = cc.view.isRetinaEnabled();
332         var flags = cc.Node._dirtyFlags, locFlag = this._dirtyFlag;
333         if (locFlag & flags.transformDirty) {
334             //update the transform
335             this.transform(this.getParentRenderCmd(), true);
336             this.updateMatrix(this._worldTransform, cc.view._scaleX, cc.view._scaleY);
337             this._dirtyFlag = this._dirtyFlag & cc.Node._dirtyFlags.transformDirty ^ this._dirtyFlag;
338         }
339 
340         if (locFlag & flags.orderDirty) {
341             this._dirtyFlag = this._dirtyFlag & flags.orderDirty ^ this._dirtyFlag;
342         }
343     };
344 
345     proto.resize = function (view) {
346         view = view || cc.view;
347         var node = this._node,
348             eventManager = cc.eventManager;
349         if (node._parent && node._visible)
350             this.updateMatrix(this._worldTransform, view._scaleX, view._scaleY);
351         else {
352             var list = eventManager._listenersMap[cc.game.EVENT_RESIZE].getFixedPriorityListeners();
353             eventManager._removeListenerInVector(list, this._listener);
354             this._listener = null;
355         }
356     };
357 
358     proto.updateMatrix = function (t, scaleX, scaleY) {
359         var node = this._node;
360         if (polyfill.devicePixelRatio) {
361             var dpr = cc.view.getDevicePixelRatio();
362             scaleX = scaleX / dpr;
363             scaleY = scaleY / dpr;
364         }
365         if (this._loaded === false) return;
366         var containerStyle = cc.game.container.style,
367             offsetX = parseInt(containerStyle.paddingLeft),
368             offsetY = parseInt(containerStyle.paddingBottom),
369             cw = node._contentSize.width,
370             ch = node._contentSize.height;
371         var a = t.a * scaleX,
372             b = t.b,
373             c = t.c,
374             d = t.d * scaleY,
375             tx = offsetX + t.tx * scaleX - cw / 2 + cw * node._scaleX / 2 * scaleX,
376             ty = offsetY + t.ty * scaleY - ch / 2 + ch * node._scaleY / 2 * scaleY;
377         var matrix = "matrix(" + a + "," + b + "," + c + "," + d + "," + tx + "," + -ty + ")";
378         this._div.style["transform"] = matrix;
379         this._div.style["-webkit-transform"] = matrix;
380     };
381 
382     proto.initStyle = function () {
383         if (!this._div)  return;
384         var div = this._div;
385         div.style.position = "absolute";
386         div.style.bottom = "0px";
387         div.style.left = "0px";
388     };
389 
390     proto.updateURL = function (url) {
391         var iframe = this._iframe;
392         iframe.src = url;
393         var self = this;
394         var cb = function () {
395             self._loaded = true;
396             iframe.removeEventListener("load", cb);
397         };
398         iframe.addEventListener("load", cb);
399     };
400 
401     proto.changeSize = function (w, h) {
402         var div = this._div;
403         if (div) {
404             div.style["width"] = w + "px";
405             div.style["height"] = h + "px";
406         }
407     };
408 
409     proto.removeDom = function () {
410         var div = this._div;
411         if (div) {
412             var hasChild = false;
413             if ('contains' in cc.container) {
414                 hasChild = cc.container.contains(div);
415             } else {
416                 hasChild = cc.container.compareDocumentPosition(div) % 16;
417             }
418             if (hasChild)
419                 cc.container.removeChild(div);
420         }
421     };
