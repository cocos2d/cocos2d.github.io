 27 /**
 28  * <p>cc.Sprite is a 2d image ( http://en.wikipedia.org/wiki/Sprite_(computer_graphics) )  <br/>
 29  *
 30  * cc.Sprite can be created with an image, or with a sub-rectangle of an image.  <br/>
 31  *
 32  * If the parent or any of its ancestors is a cc.SpriteBatchNode then the following features/limitations are valid   <br/>
 33  *    - Features when the parent is a cc.BatchNode: <br/>
 34  *        - MUCH faster rendering, specially if the cc.SpriteBatchNode has many children. All the children will be drawn in a single batch.  <br/>
 35  *
 36  *    - Limitations   <br/>
 37  *        - Camera is not supported yet (eg: CCOrbitCamera action doesn't work)  <br/>
 38  *        - GridBase actions are not supported (eg: CCLens, CCRipple, CCTwirl) <br/>
 39  *        - The Alias/Antialias property belongs to CCSpriteBatchNode, so you can't individually set the aliased property.  <br/>
 40  *        - The Blending function property belongs to CCSpriteBatchNode, so you can't individually set the blending function property. <br/>
 41  *        - Parallax scroller is not supported, but can be simulated with a "proxy" sprite.        <br/>
 42  *
 43  *  If the parent is an standard cc.Node, then cc.Sprite behaves like any other cc.Node:      <br/>
 44  *    - It supports blending functions    <br/>
 45  *    - It supports aliasing / antialiasing    <br/>
 46  *    - But the rendering will be slower: 1 draw per children.   <br/>
 47  *
 48  * The default anchorPoint in cc.Sprite is (0.5, 0.5). </p>
 49  * @class
 50  * @extends cc.Node
 51  *
 52  * @param {String|cc.SpriteFrame|HTMLImageElement|cc.Texture2D} fileName  The string which indicates a path to image file, e.g., "scene1/monster.png".
 53  * @param {cc.Rect} [rect]  Only the contents inside rect of pszFileName's texture will be applied for this sprite.
 54  * @param {Boolean} [rotated] Whether or not the texture rectangle is rotated.
 55  * @example
 56  *
 57  * 1.Create a sprite with image path and rect
 58  * var sprite1 = new cc.Sprite("res/HelloHTML5World.png");
 59  * var sprite2 = new cc.Sprite("res/HelloHTML5World.png",cc.rect(0,0,480,320));
 60  *
 61  * 2.Create a sprite with a sprite frame name. Must add "#" before frame name.
 62  * var sprite = new cc.Sprite('#grossini_dance_01.png');
 63  *
 64  * 3.Create a sprite with a sprite frame
 65  * var spriteFrame = cc.spriteFrameCache.getSpriteFrame("grossini_dance_01.png");
 66  * var sprite = new cc.Sprite(spriteFrame);
 67  *
 68  * 4.Create a sprite with an existing texture contained in a CCTexture2D object
 69  *      After creation, the rect will be the size of the texture, and the offset will be (0,0).
 70  * var texture = cc.textureCache.addImage("HelloHTML5World.png");
 71  * var sprite1 = new cc.Sprite(texture);
 72  * var sprite2 = new cc.Sprite(texture, cc.rect(0,0,480,320));
 73  *
 74  * @property {Boolean}              dirty               - Indicates whether the sprite needs to be updated.
 75  * @property {Boolean}              flippedX            - Indicates whether or not the sprite is flipped on x axis.
 76  * @property {Boolean}              flippedY            - Indicates whether or not the sprite is flipped on y axis.
 77  * @property {Number}               offsetX             - <@readonly> The offset position on x axis of the sprite in texture. Calculated automatically by editors like Zwoptex.
 78  * @property {Number}               offsetY             - <@readonly> The offset position on x axis of the sprite in texture. Calculated automatically by editors like Zwoptex.
 79  * @property {Number}               atlasIndex          - The index used on the TextureAtlas.
 80  * @property {cc.Texture2D}         texture             - Texture used to render the sprite.
 81  * @property {Boolean}              textureRectRotated  - <@readonly> Indicate whether the texture rectangle is rotated.
 82  * @property {cc.TextureAtlas}      textureAtlas        - The weak reference of the cc.TextureAtlas when the sprite is rendered using via cc.SpriteBatchNode.
 83  * @property {cc.SpriteBatchNode}   batchNode           - The batch node object if this sprite is rendered by cc.SpriteBatchNode.
 84  * @property {cc.V3F_C4B_T2F_Quad}  quad                - <@readonly> The quad (tex coords, vertex coords and color) information.

420     _softInit: function (fileName, rect, rotated) {
421         if (fileName === undefined)
422             cc.Sprite.prototype.init.call(this);
423         else if (typeof fileName === 'string') {
424             if (fileName[0] === "#") {
425                 // Init with a sprite frame name
426                 var frameName = fileName.substr(1, fileName.length - 1);
427                 var spriteFrame = cc.spriteFrameCache.getSpriteFrame(frameName);
428                 if (spriteFrame)
429                     this.initWithSpriteFrame(spriteFrame);
430                 else
431                     cc.log("%s does not exist", fileName);
432             } else {
433                 // Init  with filename and rect
434                 cc.Sprite.prototype.init.call(this, fileName, rect);
435             }
436         } else if (typeof fileName === "object") {
437             if (fileName instanceof cc.Texture2D) {
438                 // Init  with texture and rect
439                 this.initWithTexture(fileName, rect, rotated);
440             } else if (fileName instanceof cc.SpriteFrame) {
441                 // Init with a sprite frame
442                 this.initWithSpriteFrame(fileName);
443             } else if ((fileName instanceof HTMLImageElement) || (fileName instanceof HTMLCanvasElement)) {
444                 // Init with a canvas or image element
445                 var texture2d = new cc.Texture2D();
446                 texture2d.initWithElement(fileName);
447                 texture2d.handleLoadedTexture();
448                 this.initWithTexture(texture2d);
449             }
450         }
451     },

514     /**
515      * <p>
516      *     Initializes a sprite with an image filename.<br/>
517      *
518      *     This method will find pszFilename from local file system, load its content to CCTexture2D,<br/>
519      *     then use CCTexture2D to create a sprite.<br/>
520      *     After initialization, the rect used will be the size of the image. The offset will be (0,0).<br/>
521      *     Please pass parameters to the constructor to initialize the sprite, do not call this function yourself.
522      * </p>
523      * @param {String} filename The path to an image file in local file system
524      * @param {cc.Rect} rect The rectangle assigned the content area from texture.
525      * @return {Boolean} true if the sprite is initialized properly, false otherwise.
526      */
527     initWithFile: function (filename, rect) {
528         cc.assert(filename, cc._LogInfos.Sprite_initWithFile);
529 
530         var tex = cc.textureCache.getTextureForKey(filename);
531         if (!tex) {
532             tex = cc.textureCache.addImage(filename);
533         }
534 
535         if (!tex.isLoaded()) {
536             this._loader.clear();
537             this._loader.once(tex, function () {
538                 this.initWithFile(filename, rect);
539                 this.dispatchEvent("load");
540             }, this);
541             return false;
542         }
543 
544         if (!rect) {
545             var size = tex.getContentSize();
546             rect = cc.rect(0, 0, size.width, size.height);
547         }
548         return this.initWithTexture(tex, rect);
549     },
550 
551     /**
552      * Initializes a sprite with a texture and a rect in points, optionally rotated.  <br/>
553      * After initialization, the rect used will be the size of the texture, and the offset will be (0,0).<br/>
554      * Please pass parameters to the constructor to initialize the sprite, do not call this function yourself.
555      * @function
556      * @param {cc.Texture2D|HTMLImageElement|HTMLCanvasElement} texture A pointer to an existing CCTexture2D object. You can use a CCTexture2D object for many sprites.
557      * @param {cc.Rect} [rect] Only the contents inside rect of this texture will be applied for this sprite.
558      * @param {Boolean} [rotated] Whether or not the texture rectangle is rotated.
559      * @param {Boolean} [counterclockwise=true] Whether or not the texture rectangle rotation is counterclockwise (texture package is counterclockwise, spine is clockwise).
560      * @return {Boolean} true if the sprite is initialized properly, false otherwise.
561      */
562     initWithTexture: function (texture, rect, rotated, counterclockwise) {
563         var _t = this;
564         cc.assert(arguments.length !== 0, cc._LogInfos.CCSpriteBatchNode_initWithTexture);
565         this._loader.clear();
566 
567         _t._textureLoaded = texture.isLoaded();
568         if (!_t._textureLoaded) {
569             this._loader.once(texture, function () {
570                 this.initWithTexture(texture, rect, rotated, counterclockwise);
571                 this.dispatchEvent("load");
572             }, this);
573             return false;
574         }
575 
576         rotated = rotated || false;
577         texture = this._renderCmd._handleTextureForRotatedTexture(texture, rect, rotated, counterclockwise);
578 
579         if (!cc.Node.prototype.init.call(_t))
580             return false;
581 
582         _t._batchNode = null;
583         _t._recursiveDirty = false;
584         _t.dirty = false;
585         _t._opacityModifyRGB = true;
586 
587         _t._blendFunc.src = cc.BLEND_SRC;
588         _t._blendFunc.dst = cc.BLEND_DST;
589 
590         _t._flippedX = _t._flippedY = false;
591 
592         // zwoptex default values
593         _t._offsetPosition.x = 0;
594         _t._offsetPosition.y = 0;
595         _t._hasChildren = false;
596 
597         _t._rectRotated = rotated;
598         if (rect) {
599             _t._rect.x = rect.x;
600             _t._rect.y = rect.y;
601             _t._rect.width = rect.width;
602             _t._rect.height = rect.height;
603         }
604 
605         if (!rect)
606             rect = cc.rect(0, 0, texture.width, texture.height);
607 
608         this._renderCmd._checkTextureBoundary(texture, rect, rotated);
609 
610         _t.setTexture(texture);
611         _t.setTextureRect(rect, rotated);
612 
613         // by default use "Self Render".
614         // if the sprite is added to a batchnode, then it will automatically switch to "batchnode Render"
615         _t.setBatchNode(null);
616         return true;
617     },

764     /**
765      * Sets the texture of sprite
766      * @function
767      * @param {cc.Texture2D|String} texture
768      */
769     setTexture: function (texture) {
770         if (!texture)
771             return this._renderCmd._setTexture(null);
772 
773         //CCSprite.cpp 327 and 338
774         var isFileName = (typeof texture === 'string');
775 
776         if (isFileName)
777             texture = cc.textureCache.addImage(texture);
778 
779         this._loader.clear();
780         if (!texture._textureLoaded) {
781             // wait for the load to be set again
782             this._loader.once(texture, function () {
783                 this.setTexture(texture);
784                 this.dispatchEvent("load");
785             }, this);
786             return false;
787         }
788 
789         this._renderCmd._setTexture(texture);
790         if (isFileName)
791             this._changeRectWithTexture(texture);
792         this.setColor(this._realColor);
793         this._textureLoaded = true;
794     },

797         var contentSize = texture._contentSize;
798         var rect = cc.rect(
799             0, 0,
800             contentSize.width, contentSize.height
801         );
802         this.setTextureRect(rect);
803     },
