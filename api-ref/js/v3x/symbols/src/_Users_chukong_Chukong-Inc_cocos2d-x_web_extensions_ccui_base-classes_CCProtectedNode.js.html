 25 /**
 26  * A class inhert from cc.Node, use for saving some protected children in other list.
 27  * @class
 28  * @extends cc.Node
 29  */
 30 cc.ProtectedNode = cc.Node.extend(/** @lends cc.ProtectedNode# */{
 31     _protectedChildren: null,
 32     _reorderProtectedChildDirty: false,
 33 
 34     _insertProtectedChild: function (child, z) {
 35         this._reorderProtectedChildDirty = true;
 36         this._protectedChildren.push(child);
 37         child._setLocalZOrder(z);
 38     },
 39 
 40     /**
 41      * Constructor function, override it to extend the construction behavior, remember to call "this._super()" in the extended "ctor" function.
 42      * @function
 43      */
 44     ctor: function () {
 45         cc.Node.prototype.ctor.call(this);
 46         this._protectedChildren = [];
 47     },
 48 
 49     visit: function (parent) {
 50         // quick return if not visible
 51         if (!this._visible)
 52             return;
 53 
 54         var renderer = cc.renderer, cmd = this._renderCmd;
 55         var i, children = this._children, len = children.length, child;
 56         var j, pChildren = this._protectedChildren, pLen = pChildren.length, pChild;
 57 
 58         cmd.visit(parent && parent._renderCmd);
 59 
 60         var locGrid = this.grid;
 61         if (locGrid && locGrid._active)
 62             locGrid.beforeDraw();
 63 
 64         if (this._reorderChildDirty) this.sortAllChildren();
 65         if (this._reorderProtectedChildDirty) this.sortAllProtectedChildren();
 66 
 67         // draw children zOrder < 0
 68         for (i = 0; i < len; i++) {
 69             child = children[i];
 70             if (child._localZOrder < 0) {
 71                 child.visit(this);
 72             }
 73             else {
 74                 break;
 75             }
 76         }
 77         for (j = 0; j < pLen; j++) {
 78             pChild = pChildren[j];
 79             if (pChild && pChild._localZOrder < 0) {
 80                 cmd._changeProtectedChild(pChild);
 81                 pChild.visit(this);
 82             }
 83             else
 84                 break;
 85         }
 86 
 87         renderer.pushRenderCommand(cmd);
 88 
 89         for (; i < len; i++) {
 90             children[i].visit(this);
 91         }
 92         for (; j < pLen; j++) {
 93             pChild = pChildren[j];
 94             if (!pChild) continue;
 95             cmd._changeProtectedChild(pChild);
 96             pChild.visit(this);
 97         }
 98 
 99         if (locGrid && locGrid._active)
100             locGrid.afterDraw(this);
101 
102         cmd._dirtyFlag = 0;
103     },
104 
105     /**
106      * <p>
107      *  Adds a child to the container with z order and tag                                                                         <br/>
108      *  If the child is added to a 'running' node, then 'onEnter' and 'onEnterTransitionDidFinish' will be called immediately.     <br/>
109      *  </p>
110      * @param {cc.Node} child  A child node
111      * @param {Number} [localZOrder]  Z order for drawing priority. Please refer to `setLocalZOrder(int)`
112      * @param {Number} [tag]  An integer to identify the node easily. Please refer to `setTag(int)`
113      */
114     addProtectedChild: function (child, localZOrder, tag) {
115         cc.assert(child != null, "child must be non-nil");
116         cc.assert(!child.parent, "child already added. It can't be added again");
117 
118         localZOrder = localZOrder || child.getLocalZOrder();
119         if (tag)
120             child.setTag(tag);
121 
122         this._insertProtectedChild(child, localZOrder);
123         child.setParent(this);
124         child.setOrderOfArrival(cc.s_globalOrderOfArrival);
125 
126         if (this._running) {
127             child._performRecursive(cc.Node._stateCallbackType.onEnter);
128             // prevent onEnterTransitionDidFinish to be called twice when a node is added in onEnter
129             if (this._isTransitionFinished)
130                 child._performRecursive(cc.Node._stateCallbackType.onEnterTransitionDidFinish);
131         }
132         if (this._cascadeColorEnabled)
133             this._renderCmd.setCascadeColorEnabledDirty();
134         if (this._cascadeOpacityEnabled)
135             this._renderCmd.setCascadeOpacityEnabledDirty();
136     },
137 
138     /**
139      * Gets a child from the container with its tag
140      * @param {Number} tag An identifier to find the child node.
141      * @return {cc.Node} a Node object whose tag equals to the input parameter
142      */
143     getProtectedChildByTag: function (tag) {
144         cc.assert(tag !== cc.NODE_TAG_INVALID, "Invalid tag");
145         var locChildren = this._protectedChildren;
146         for (var i = 0, len = locChildren.length; i < len; i++)
147             if (locChildren.getTag() === tag)
148                 return locChildren[i];
149         return null;
150     },
151 
152     /**
153      * Removes a child from the container. It will also cleanup all running actions depending on the cleanup parameter.
154      * @param {cc.Node} child  The child node which will be removed.
155      * @param {Boolean} [cleanup=true] true if all running actions and callbacks on the child node will be cleanup, false otherwise.
156      */
157     removeProtectedChild: function (child, cleanup) {
158         if (cleanup == null)
159             cleanup = true;
160         var locChildren = this._protectedChildren;
161         if (locChildren.length === 0)
162             return;
163         var idx = locChildren.indexOf(child);
164         if (idx > -1) {
165             if (this._running) {
166                 child._performRecursive(cc.Node._stateCallbackType.onExitTransitionDidStart);
167                 child._performRecursive(cc.Node._stateCallbackType.onExit);
168             }
169 
170             // If you don't do cleanup, the child's actions will not get removed and the
171             // its scheduledSelectors_ dict will not get released!
172             if (cleanup)
173                 child._performRecursive(cc.Node._stateCallbackType.cleanup);
174 
175             // set parent nil at the end
176             child.setParent(null);
177             locChildren.splice(idx, 1);
178         }
179     },
180 
181     /**
182      * Removes a child from the container by tag value.                                    <br/>
183      * It will also cleanup all running actions depending on the cleanup parameter
184      * @param {Number} tag
185      * @param {Boolean} [cleanup=true]
186      */
187     removeProtectedChildByTag: function (tag, cleanup) {
188         cc.assert(tag !== cc.NODE_TAG_INVALID, "Invalid tag");
189 
190         if (cleanup == null)
191             cleanup = true;
192 
193         var child = this.getProtectedChildByTag(tag);
194 
195         if (child == null)
196             cc.log("cocos2d: removeChildByTag(tag = %d): child not found!", tag);
197         else
198             this.removeProtectedChild(child, cleanup);
199     },
200 
201     /**
202      * Removes all children from the container with a cleanup.
203      * @see cc.ProtectedNode#removeAllProtectedChildrenWithCleanup
204      */
205     removeAllProtectedChildren: function () {
206         this.removeAllProtectedChildrenWithCleanup(true);
207     },
208 
209     /**
210      * Removes all children from the container, and do a cleanup to all running actions depending on the cleanup parameter.
211      * @param {Boolean} [cleanup=true] true if all running actions on all children nodes should be cleanup, false otherwise.
212      */
213     removeAllProtectedChildrenWithCleanup: function (cleanup) {
214         if (cleanup == null)
215             cleanup = true;
216         var locChildren = this._protectedChildren;
217         // not using detachChild improves speed here
218         for (var i = 0, len = locChildren.length; i < len; i++) {
219             var child = locChildren[i];
220             // IMPORTANT:
221             //  -1st do onExit
222             //  -2nd cleanup
223             if (this._running) {
224                 child._performRecursive(cc.Node._stateCallbackType.onExitTransitionDidStart);
225                 child._performRecursive(cc.Node._stateCallbackType.onExit);
226             }
227 
228             if (cleanup)
229                 child._performRecursive(cc.Node._stateCallbackType.cleanup);
230             // set parent nil at the end
231             child.setParent(null);
232         }
233         locChildren.length = 0;
234     },
235 
236     /**
237      * Reorders a child according to a new z value.
238      * @param {cc.Node} child An already added child node. It MUST be already added.
239      * @param {Number} localZOrder Z order for drawing priority. Please refer to setLocalZOrder(int)
240      */
241     reorderProtectedChild: function (child, localZOrder) {
242         cc.assert(child != null, "Child must be non-nil");
243         this._reorderProtectedChildDirty = true;
244         child.setOrderOfArrival(cc.s_globalOrderOfArrival++);
245         child._setLocalZOrder(localZOrder);
246     },
247 
248     /**
249      * <p>
250      *     Sorts the children array once before drawing, instead of every time when a child is added or reordered.       <br/>
251      *     This approach can improves the performance massively.                                                         <br/>
252      *     @note Don't call this manually unless a child added needs to be removed in the same frame
253      * </p>
254      */
255     sortAllProtectedChildren: function () {
256         if (this._reorderProtectedChildDirty) {
257             var _children = this._protectedChildren;
258 
259             // insertion sort
260             var i, j, len = _children.length, tmp;
261             for (i = 1; i < len; i++) {
262                 tmp = _children[i];
263                 j = i - 1;
264 
265                 //continue moving element downwards while zOrder is smaller or when zOrder is the same but mutatedIndex is smaller
266                 while (j >= 0) {
267                     if (tmp._localZOrder < _children[j]._localZOrder) {
268                         _children[j + 1] = _children[j];
269                     } else if (tmp._localZOrder === _children[j]._localZOrder && tmp.arrivalOrder < _children[j].arrivalOrder) {
270                         _children[j + 1] = _children[j];
271                     } else
272                         break;
273                     j--;
274                 }
275                 _children[j + 1] = tmp;
276             }
277 
278             //don't need to check children recursively, that's done in visit of each child
279             this._reorderProtectedChildDirty = false;
280         }
281     },
282 
283     _changePosition: function () {
284     },
285 
286     _createRenderCmd: function () {
287         if (cc._renderType === cc.game.RENDER_TYPE_CANVAS)
288             return new cc.ProtectedNode.CanvasRenderCmd(this);
289         else
290             return new cc.ProtectedNode.WebGLRenderCmd(this);
291     }
292 });
293 
294 /**
295  * create a cc.ProtectedNode object;
296  * @deprecated since v3.0, please use new cc.ProtectedNode() instead.
297  * @return cc.ProtectedNode
298  */
299 cc.ProtectedNode.create = function () {
300     return new cc.ProtectedNode();
301 };
