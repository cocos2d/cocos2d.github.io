 25 /**
 26  * Timeline Frame.
 27  * base class
 28  * @class
 29  */
 30 ccs.Frame = ccs.Class.extend({
 31 
 32     _frameIndex: null,
 33     _tween: null,
 34     _timeline: null,
 35     _node: null,
 36     _tweenType: null,
 37     _easingParam: null,
 38     _enterWhenPassed: null,
 39 
 40     ctor: function () {
 41         this._frameIndex = 0;
 42         this._tween = true;
 43         this._timeline = null;
 44         this._node = null;
 45         this._enterWhenPassed = false;
 46         this._easingParam = [];
 47     },
 48 
 49     _emitEvent: function () {
 50         if (this._timeline) {
 51             this._timeline.getActionTimeline()._emitFrameEvent(this);
 52         }
 53     },
 54 
 55     _cloneProperty: function (frame) {
 56         this._frameIndex = frame.getFrameIndex();
 57         this._tween = frame.isTween();
 58         this._tweenType = frame.getTweenType();
 59         this.setEasingParams(frame.getEasingParams());
 60     },
 61 
 62     /**
 63      * Set the frame index
 64      * @param {number} frameIndex
 65      */
 66     setFrameIndex: function (frameIndex) {
 67         this._frameIndex = frameIndex;
 68     },
 69 
 70     /**
 71      * Get the frame index
 72      * @returns {null}
 73      */
 74     getFrameIndex: function () {
 75         return this._frameIndex;
 76     },
 77 
 78     /**
 79      * Set timeline
 80      * @param timeline
 81      */
 82     setTimeline: function (timeline) {
 83         this._timeline = timeline;
 84     },
 85 
 86     /**
 87      * Get timeline
 88      * @param timeline
 89      * @returns {ccs.timeline}
 90      */
 91     getTimeline: function (timeline) {
 92         return this._timeline;
 93     },
 94 
 95     /**
 96      * Set Node
 97      * @param {cc.Node} node
 98      */
 99     setNode: function (node) {
100         this._node = node;
101     },
102 
103     /**
104      * gets the Node
105      * @return node
106      */
107     getNode: function () {
108         return this._node;
109     },
110 
111     /**
112      * set tween
113      * @param tween
114      */
115     setTween: function (tween) {
116         this._tween = tween;
117     },
118 
119     /**
120      * Gets the tween
121      * @returns {boolean | null}
122      */
123     isTween: function () {
124         return this._tween;
125     },
126 
127     /**
128      * the execution of the callback
129      * @override
130      * @param {ccs.Frame} nextFrame
131      */
132     onEnter: function (nextFrame) { // = 0
133     },
134 
135     /**
136      * Each frame logic
137      * @override
138      * @param {number} percent
139      */
140     apply: function (percent) {
141         if (!this._tween)
142             return;
143         if (this._tweenType !== ccs.FrameEaseType.TWEEN_EASING_MAX && this._tweenType !== ccs.FrameEaseType.LINEAR)
144             percent = this.tweenPercent(percent);
145         this._onApply(percent);
146     },
147 
148     _onApply: function (percent) {
149 
150     },
151 
152     /**
153      * to copy object with deep copy.
154      * returns a clone of action.
155      * @override
156      * @return {ccs.Frame}
157      */
158     clone: function () { // = 0
159     },
160 
161     tweenPercent: function (percent) {
162         var func = ccs.Frame.tweenToMap[this._tweenType];
163         if (func)
164             return func(percent, this._easingParam);
165         else
166             return percent;
167     },
168 
169     setEasingParams: function (easingParams) {
170         if (easingParams) {
171             this._easingParam.length = 0;
172             for (var i = 0; i < easingParams.length; i++)
173                 this._easingParam[i] = easingParams[i];
174         }
175     },
176 
177     getEasingParams: function () {
178         return this._easingParam;
179     },
180 
181     setTweenType: function (tweenType) {
182         this._tweenType = tweenType;
183     },
184 
185     getTweenType: function () {
186         return this._tweenType;
187     },
188 
189     isEnterWhenPassed: function () {
190         return this._enterWhenPassed;
191     }
192 });
193 
194 ccs.Frame.tweenToMap = {
195     "-1": function (time, easingParam) {
196         if (easingParam) {
197             var tt = 1 - time;
198             return easingParam[1] * tt * tt * tt + 3 * easingParam[3] * time * tt * tt + 3 * easingParam[5] * time * time * tt + easingParam[7] * time * time * time;
199         }
200         return time;
201     },
202     1: cc._easeSineInObj.easing,//Sine_EaseIn
203     2: cc._easeSineOutObj.easing,//Sine_EaseOut
204     3: cc._easeSineInOutObj.easing,//Sine_EaseInOut
205 
206     4: cc._easeQuadraticActionIn.easing,//Quad_EaseIn
207     5: cc._easeQuadraticActionOut.easing,//Quad_EaseOut
208     6: cc._easeQuadraticActionInOut.easing,//Quad_EaseInOut
209 
210     7: cc._easeCubicActionIn.easing, //Cubic_EaseIn
211     8: cc._easeCubicActionOut.easing,//Cubic_EaseOut
212     9: cc._easeCubicActionInOut.easing,//Cubic_EaseInOut
213 
214     10: cc._easeCubicActionIn.easing,//Cubic_EaseIn
215     11: cc._easeCubicActionOut.easing,//Cubic_EaseOut
216     12: cc._easeCubicActionInOut.easing,//Cubic_EaseInOut
217 
218     13: cc._easeQuinticActionIn.easing,//Quint_EaseIn
219     14: cc._easeQuinticActionOut.easing,//Quint_EaseOut
220     15: cc._easeQuinticActionInOut.easing,//Quint_EaseInOut
221 
222     16: cc._easeExponentialInObj.easing,//Expo_EaseIn
223     17: cc._easeExponentialOutObj.easing,//Expo_EaseOut
224     18: cc._easeExponentialInOutObj.easing,//Expo_EaseInOut
225 
226     19: cc._easeCircleActionIn.easing,//Circ_EaseIn
227     20: cc._easeCircleActionOut.easing,//Circ_EaseOut
228     21: cc._easeCircleActionInOut.easing,//Circ_EaseInOut
229 
230     22: function (time, easingParam) {
231         var period = 0.3;
232         easingParam != null && ( period = easingParam[0] );
233         return cc.easeElasticIn(period).easing(time);
234     },//Elastic_EaesIn
235     23: function (time, easingParam) {
236         var period = 0.3;
237         easingParam != null && ( period = easingParam[0] );
238         return cc.easeElasticOut(period).easing(time);
239     },//Elastic_EaesOut
240     24: function (time, easingParam) {
241         var period = 0.3;
242         easingParam != null && ( period = easingParam[0] );
243         return cc.easeElasticInOut(period).easing(time);
244     },//Elastic_EaesInOut
245 
246     25: cc._easeBackInObj.easing, //Back_EaseIn
247     26: cc._easeBackOutObj.easing, //Back_EaseOut
248     27: cc._easeBackInOutObj.easing, //Back_EaseInOut
249 
250     28: cc._easeBounceInObj.easing, //Bounce_EaseIn
251     29: cc._easeBounceOutObj.easing, //Bounce_EaseOut
252     30: cc._easeBounceInOutObj.easing //Bounce_EaseInOut
253 };

321 /**
322  * Texture frame
323  * @class
324  * @extend ccs.Frame
325  */
326 ccs.TextureFrame = ccs.Frame.extend({
327 
328     _sprite: null,
329     _textureName: null,
330 
331     ctor: function () {
332         ccs.Frame.prototype.ctor.call(this);
333 
334         this._textureName = "";
335     },
336 
337     /**
338      * Set the node element to draw texture
339      * @param {cc.Node} node
340      */
341     setNode: function (node) {
342         ccs.Frame.prototype.setNode.call(this, node);
343         this._sprite = node;
344     },
345 
346     /**
347      * the execution of the callback
348      * @param {ccs.Frame} nextFrame
349      */
350     onEnter: function (nextFrame) {
351         if (this._sprite) {
352             var spriteBlendFunc = this._sprite.getBlendFunc();
353             var spriteFrame = cc.spriteFrameCache._spriteFrames[this._textureName];
354             if (spriteFrame != null)
355                 this._sprite.setSpriteFrame(spriteFrame);
356             else
357                 this._sprite.setTexture(this._textureName);
358 
359             if (this._sprite.getBlendFunc() !== spriteBlendFunc)
360                 this._sprite.setBlendFunc(spriteBlendFunc);
361         }
362 
363     },
364 
365     /**
366      * to copy object with deep copy.
367      * returns a clone of action.
368      * @return {ccs.TextureFrame}
369      */
370     clone: function () {
371         var frame = new ccs.TextureFrame();
372         frame.setTextureName(this._textureName);
373         frame._cloneProperty(this);
374         return frame;
375     },
376 
377     /**
378      * Set the texture name
379      * @param {string} textureName
380      */
381     setTextureName: function (textureName) {
382         this._textureName = textureName;
383     },
384 
385     /**
386      * Gets the Texture name
387      * @returns {null}
388      */
389     getTextureName: function () {
390         return this._textureName;
391     }
392 
393 });
394

396  * Create the Texture frame

398  * @deprecated v3.0, please use new ccs.TextureFrame() instead.
399  * @returns {ccs.TextureFrame}
400  */
401 ccs.TextureFrame.create = function () {
402     return new ccs.TextureFrame();

980 ccs.InnerActionFrame = ccs.Frame.extend({
981 
982     _innerActionType: null,
983     _startFrameIndex: null,
984 
985     _endFrameIndex: 0,
986     _singleFrameIndex: 0,
987     _enterWithName: null,
988     _animationName: "",
989 
990     ctor: function () {
991         ccs.Frame.prototype.ctor.call(this);
992 
993         this._enterWithName = false;
994         this._innerActionType = ccs.InnerActionType.LoopAction;
995         this._startFrameIndex = 0;
996     },
997 
998     /**
999      * the execution of the callback
1000      * @param {ccs.Frame} nextFrame
1001      */
1002     onEnter: function (nextFrame) {
1003         if (!this._node)  return;
1004         var innerActiontimeline = this._node.getActionByTag(this._node.getTag());
1005         if (!innerActiontimeline) return;
1006         if (ccs.InnerActionType.SingleFrame === this._innerActionType) {
1007             innerActiontimeline.gotoFrameAndPause(this._singleFrameIndex);
1008             return;
1009         }
1010 
1011         var innerStart = this._startFrameIndex;
1012         var innerEnd = this._endFrameIndex;
1013         if (this._enterWithName) {
1014             if (this._animationName === "-- ALL --") {
1015                 innerStart = 0;
1016                 innerEnd = innerActiontimeline.getDuration();
1017             } else if (innerActiontimeline.isAnimationInfoExists(this._animationName)) {
1018                 var info = innerActiontimeline.getAnimationInfo(this._animationName);
1019                 innerStart = info.startIndex;
1020                 innerEnd = info.endIndex;
1021             } else {
1022                 cc.log("Animation %s not exists!", this._animationName);
1023             }
1024         }
1025 
1026         var duration = this._timeline.getActionTimeline().getDuration();
1027         var odddiff = duration - this._frameIndex - innerEnd + innerStart;
1028         if (odddiff < 0) {
1029             innerEnd += odddiff;
1030         }
1031 
1032         if (ccs.InnerActionType.NoLoopAction === this._innerActionType) {
1033             innerActiontimeline.gotoFrameAndPlay(innerStart, innerEnd, false);
1034         } else if (ccs.InnerActionType.LoopAction === this._innerActionType) {
1035             innerActiontimeline.gotoFrameAndPlay(innerStart, innerEnd, true);
1036         }
1037     },
1038

1040         this._animationName = animationName;
1041     },
1042 
1043     setSingleFrameIndex: function (frameIndex) {

1045     },

1048         return this._startFrameIndex;
1049     },
1050

1067         frame.setStartFrameIndex(this._startFrameIndex);
1068         frame.setEnterWithName(this._enterWithName);
