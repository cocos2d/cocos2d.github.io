 25 /**
 26  * timeline object
 27  * @class
 28  * @extend ccs.Class
 29  */
 30 ccs.Timeline = ccs.Class.extend({
 31 
 32     //{ccs.Frame}
 33     _frames: null,
 34     //{ccs.Frame}
 35     _currentKeyFrame: null,
 36     //{Number}
 37     _currentKeyFrameIndex: null,
 38     //{Number}
 39     _fromIndex: null,
 40     //{Number}
 41     _toIndex: null,
 42     //{Number}
 43     _betweenDuration: null,
 44     //{Number}
 45     _actionTag: null,
 46     //{ccs.ActionTimeline}
 47     _ActionTimeline: null,
 48     //{cc.Node}
 49     _node: null,
 50 
 51     ctor: function () {
 52         this._frames = [];
 53         this._currentKeyFrame = null;
 54         this._currentKeyFrameIndex = 0;
 55         this._fromIndex = 0;
 56         this._toIndex = 0;
 57         this._betweenDuration = 0;
 58         this._actionTag = 0;
 59         this._ActionTimeline = null;
 60         this._node = null;
 61     },
 62 
 63     _gotoFrame: function (frameIndex) {
 64         if (this._frames.length === 0)
 65             return;
 66 
 67         this._binarySearchKeyFrame(frameIndex);
 68         this._apply(frameIndex);
 69     },
 70 
 71     _stepToFrame: function (frameIndex) {
 72         if (this._frames.length === 0)
 73             return;
 74 
 75         this._updateCurrentKeyFrame(frameIndex);
 76         this._apply(frameIndex);
 77     },
 78 
 79     /**
 80      * Get the frame list
 81      * @returns {ccs.Frame}
 82      */
 83     getFrames: function () {
 84         return this._frames;
 85     },
 86 
 87     /**
 88      * push frame to frame list
 89      * @param {ccs.Frame} frame
 90      */
 91     addFrame: function (frame) {
 92         this._frames.push(frame);
 93         frame.setTimeline(this)
 94     },
 95 
 96     /**
 97      * insert the frame to frame list
 98      * @param {ccs.Frame} frame
 99      * @param {Number} index
100      */
101     insertFrame: function (frame, index) {
102         this._frames.splice(index, 0, frame);
103         frame.setTimeline(this);
104 
105     },
106 
107     /**
108      * remove frame
109      * @param {ccs.Frame} frame
110      */
111     removeFrame: function (frame) {
112         cc.arrayRemoveObject(this._frames, frame);
113         frame.setTimeline(null);
114     },
115 
116     /**
117      * Set the action tag
118      * @param {Number} tag
119      */
120     setActionTag: function (tag) {
121         this._actionTag = tag;
122     },
123 
124     /**
125      * Gets the action tag
126      * return {Number}
127      */
128     getActionTag: function () {
129         return this._actionTag;
130     },
131 
132     /**
133      * Set the node
134      * @param {cc.Node} node
135      */
136     setNode: function (node) {
137         for (var i = 0; i < this._frames.length; i++) {
138             var frame = this._frames[i];
139             frame.setNode(node);
140         }
141     },
142 
143     /**
144      * Gets the node
145      * return {cc.Node}
146      */
147     getNode: function () {
148         return this._node;
149     },
150 
151     /**
152      * Set the action timeline
153      * @param {ccs.ActionTimeline} action
154      */
155     setActionTimeline: function (action) {
156         this._ActionTimeline = action;
157     },
158 
159     /**
160      * get the action timeline
161      * return {cc.Action}
162      */
163     getActionTimeline: function () {
164         return this._ActionTimeline;
165     },
166 
167     /**
168      * to copy object with deep copy.
169      * returns a clone of action.
170      * @return {ccs.Timeline}
171      */
172     clone: function () {
173         var timeline = new ccs.Timeline();
174         timeline._actionTag = this._actionTag;
175 
176         for (var i = 0; i < this._frames.length; i++) {
177             var frame = this._frames[i];
178             var newFrame = frame.clone();
179             timeline.addFrame(newFrame);
180         }
181 
182         return timeline;
183 
184     },
185 
186     _apply: function (frameIndex) {
187         if (this._currentKeyFrame) {
188             var currentPercent = this._betweenDuration <= 0 ? 0 : (frameIndex - this._currentKeyFrameIndex) / this._betweenDuration;
189             this._currentKeyFrame.apply(currentPercent);
190         }
191     },
192 
193     _binarySearchKeyFrame: function (frameIndex) {
194         var from = null;
195         var to = null;
196 
197         var length = this._frames.length;
198         var needEnterFrame = false;
199 
200         do {
201             if (frameIndex < this._frames[0].getFrameIndex()) {
202                 if (this._currentKeyFrameIndex >= this._frames[0].getFrameIndex())
203                     needEnterFrame = true;
204 
205                 this._fromIndex = 0;
206                 this._toIndex = 0;
207 
208                 from = to = this._frames[0];
209                 this._currentKeyFrameIndex = 0;
210                 this._betweenDuration = this._frames[0].getFrameIndex();
211                 break;
212             } else if (frameIndex >= this._frames[length - 1].getFrameIndex()) {
213                 this._fromIndex = length - 1;
214                 this._toIndex = 0;
215 
216                 from = to = this._frames[length - 1];
217                 this._currentKeyFrameIndex = this._frames[length - 1].getFrameIndex();
218                 this._betweenDuration = 0;
219                 break;
220             }
221 
222             var target = -1;
223             var low = 0,
224                 high = length - 1,
225                 mid = 0;
226             while (low <= high) {
227                 mid = Math.ceil(( low + high ) / 2);
228                 if (frameIndex >= this._frames[mid].getFrameIndex() && frameIndex < this._frames[mid + 1].getFrameIndex()) {
229                     target = mid;
230                     break;
231                 }
232                 if (this._frames[mid].getFrameIndex() > frameIndex)
233                     high = mid - 1;
234                 else
235                     low = mid + 1;
236             }
237 
238             this._fromIndex = target;
239 
240             if (length > 1)
241                 this._toIndex = (target + 1) | 0;
242             else
243                 this._toIndex = (target) | 0;
244 
245             from = this._frames[this._fromIndex];
246             to = this._frames[this._toIndex];
247 
248             from = this._frames[target];
249             to = this._frames[target + 1];
250 
251             if (target === 0 && this._currentKeyFrameIndex < from.getFrameIndex())
252                 needEnterFrame = true;
253 
254             this._currentKeyFrameIndex = from.getFrameIndex();
255             this._betweenDuration = to.getFrameIndex() - from.getFrameIndex();
256         } while (0);
257 
258         if (needEnterFrame || this._currentKeyFrame != from) {
259             this._currentKeyFrame = from;
260             this._currentKeyFrame.onEnter(to);
261         }
262 
263     },
264 
265     _updateCurrentKeyFrame: function (frameIndex) {
266         if (frameIndex > 60)
267             var a = 0;
268         //! If play to current frame's front or back, then find current frame again
269         if (frameIndex < this._currentKeyFrameIndex || frameIndex >= this._currentKeyFrameIndex + this._betweenDuration) {
270             var from = null;
271             var to = null;
272 
273             do
274             {
275                 var length = this._frames.length;
276 
277                 if (frameIndex < this._frames[0].getFrameIndex()) {
278                     from = to = this._frames[0];
279                     this._currentKeyFrameIndex = 0;
280                     this._betweenDuration = this._frames[0].getFrameIndex();
281                     break;
282                 }
283                 else if (frameIndex >= this._frames[length - 1].getFrameIndex()) {
284                     var lastFrameIndex = this._frames[length - 1].getFrameIndex();
285                     if (this._currentKeyFrameIndex >= lastFrameIndex)
286                         return;
287                     frameIndex = lastFrameIndex;
288                 }
289 
290                 do {
291                     this._fromIndex = this._toIndex;
292                     from = this._frames[this._fromIndex];
293                     this._currentKeyFrameIndex = from.getFrameIndex();
294 
295                     this._toIndex = this._fromIndex + 1;
296                     if (this._toIndex >= length) {
297                         this._toIndex = 0;
298                     }
299 
300                     to = this._frames[this._toIndex];
301 
302                     if (frameIndex === from.getFrameIndex())
303                         break;
304                     if (frameIndex > from.getFrameIndex() && frameIndex < to.getFrameIndex())
305                         break;
306                     if (from.isEnterWhenPassed())
307                         from.onEnter(to);
308                 } while (true);
309 
310                 this._betweenDuration = to.getFrameIndex() - from.getFrameIndex();
311 
312             } while (0);
313 
314             this._currentKeyFrame = from;
315             this._currentKeyFrame.onEnter(to);
316         }
317     }
318 
319 });
320 
321 /**
322  * Create the Timeline
323  *
324  * @deprecated v3.0, please use new ccs.Timeline() instead.
325  * @returns {ccs.Timeline}
326  */
327 ccs.Timeline.create = function () {
328     return new ccs.Timeline();
329 };
