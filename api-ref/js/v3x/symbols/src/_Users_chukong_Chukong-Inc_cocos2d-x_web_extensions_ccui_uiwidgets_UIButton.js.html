 77     /**
 78      * Allocates and initializes a UIButton.
 79      * Constructor of ccui.Button. override it to extend the construction behavior, remember to call "this._super()" in the extended "ctor" function.
 80      * @param {String} normalImage
 81      * @param {String} [selectedImage=""]
 82      * @param {String} [disableImage=""]
 83      * @param {Number} [texType=ccui.Widget.LOCAL_TEXTURE]

 85      * // example
 86      * var uiButton = new ccui.Button();

 88     ctor: function (normalImage, selectedImage, disableImage, texType) {
 89         this._capInsetsNormal = cc.rect(0, 0, 0, 0);
 90         this._normalTextureSize = cc.size(0, 0);
 91         ccui.Widget.prototype.ctor.call(this);
 92         this.setTouchEnabled(true);
 93 
 94         this._normalLoader = new cc.Sprite.LoadManager();
 95         this._clickedLoader = new cc.Sprite.LoadManager();
 96         this._disabledLoader = new cc.Sprite.LoadManager();
 97 
 98         if (normalImage) {
 99             this.loadTextures(normalImage, selectedImage,disableImage, texType);
100         }

121     /**
122      * Sets if button is using scale9 renderer.
123      * @param {Boolean} able true that using scale9 renderer, false otherwise.
124      */
125     setScale9Enabled: function (able) {
126         if (this._scale9Enabled === able)
127             return;
128 
129         this._brightStyle = ccui.Widget.BRIGHT_STYLE_NONE;
130         this._scale9Enabled = able;
131 
132         if (this._scale9Enabled) {
133             this._buttonScale9Renderer.setRenderingType(ccui.Scale9Sprite.RenderingType.SLICED);
134         } else {
135             this._buttonScale9Renderer.setRenderingType(ccui.Scale9Sprite.RenderingType.SIMPLE);
136         }
137 
138         if (this._scale9Enabled) {
139             var ignoreBefore = this._ignoreSize;
140             this.ignoreContentAdaptWithSize(false);
141             this._prevIgnoreSize = ignoreBefore;
142         } else {
143             this.ignoreContentAdaptWithSize(this._prevIgnoreSize);
144         }
145         this.setCapInsets(this._capInsetsNormal);
146 
147         this.setBright(this._bright);
148 
149         this._normalTextureAdaptDirty = true;

205     _createSpriteFrameWithFile: function (file) {
206         var texture = cc.textureCache.getTextureForKey(file);
207         if (!texture) {
208             texture = cc.textureCache.addImage(file);
209         }
210         if(!texture._textureLoaded) {
211             return texture;
212         }
213 
214         var textureSize = texture.getContentSize();
215         var rect = cc.rect(0, 0, textureSize.width, textureSize.height);
216         return new cc.SpriteFrame(texture, rect);
217     },

219     _createSpriteFrameWithName: function (name) {
220         var frame = cc.spriteFrameCache.getSpriteFrame(name);
221         if (frame == null) {
222             cc.log("ccui.Scale9Sprite.initWithSpriteFrameName(): can't find the sprite frame by spriteFrameName");
223             return null;
224         }
225 
226         return frame;
227     },

229     /**
230      * Load normal state texture for button.
231      * @param {String} normal normal state of texture's filename.
232      * @param {ccui.Widget.LOCAL_TEXTURE|ccui.Widget.PLIST_TEXTURE} texType
233      */
234     loadTextureNormal: function (normal, texType) {
235         if (!normal) return;
236 
237         texType = texType || ccui.Widget.LOCAL_TEXTURE;
238         this._normalFileName = normal;
239         this._normalTexType = texType;
240 
241         var normalSpriteFrame;
242         switch (this._normalTexType){
243             case ccui.Widget.LOCAL_TEXTURE:
244               normalSpriteFrame = this._createSpriteFrameWithFile(normal);
245               break;
246             case ccui.Widget.PLIST_TEXTURE:
247               if (normal[0] === "#") {
248                   normal = normal.substr(1, normal.length - 1);
249               }
250               normalSpriteFrame = this._createSpriteFrameWithName(normal);
251               break;
252           default:
253               break;
254         }
255 
256         if(!normalSpriteFrame) {
257             return;
258         }
259 
260         if(!normalSpriteFrame._textureLoaded) {
261             this._normalLoader.clear();
262             this._normalLoader.once(normalSpriteFrame, function () {
263                 this.loadTextureNormal(this._normalFileName, this._normalTexType);
264             }, this);
265             return;
266         }
267 
268         this._normalTextureLoaded = normalSpriteFrame._textureLoaded;
269         this._buttonNormalSpriteFrame = normalSpriteFrame;
270         this._buttonScale9Renderer.setSpriteFrame(normalSpriteFrame);
271         if (this._scale9Enabled){
272             this._buttonScale9Renderer.setCapInsets(this._capInsetsNormal);
273         }
274 
275         // FIXME: https://github.com/cocos2d/cocos2d-x/issues/12249
276         if(!this._ignoreSize &&  cc.sizeEqualToSize(this._customSize, cc.size(0, 0))) {
277             this._customSize = this._buttonScale9Renderer.getContentSize();
278         }
279 
280         this._normalTextureSize = this._buttonScale9Renderer.getContentSize();
281         this._updateChildrenDisplayedRGBA();
282         if (this._unifySize){
283             if (this._scale9Enabled){
284                 this._buttonScale9Renderer.setCapInsets(this._capInsetsNormal);
285                 this._updateContentSizeWithTextureSize(this._getNormalSize());
286             }
287         }else {
288             this._updateContentSizeWithTextureSize(this._normalTextureSize);
289         }
290 
291         this._normalTextureAdaptDirty = true;
292         this._findLayout();
293     },

295     /**
296      * Load selected state texture for button.
297      * @param {String} selected selected state of texture's filename.
298      * @param {ccui.Widget.LOCAL_TEXTURE|ccui.Widget.PLIST_TEXTURE} texType
299      */
300     loadTexturePressed: function (selected, texType) {
301         if (!selected)
302             return;
303         texType = texType || ccui.Widget.LOCAL_TEXTURE;
304         this._clickedFileName = selected;
305         this._pressedTexType = texType;
306 
307         var clickedSpriteFrame;
308         switch (this._pressedTexType) {
309           case ccui.Widget.LOCAL_TEXTURE:
310               clickedSpriteFrame = this._createSpriteFrameWithFile(selected);
311               break;
312           case ccui.Widget.PLIST_TEXTURE:
313               if (selected[0] === "#") {
314                   selected = selected.substr(1, selected.length - 1);
315               }
316               clickedSpriteFrame = this._createSpriteFrameWithName(selected);
317               break;
318           default:
319               break;
320         }
321 
322         if(!clickedSpriteFrame) return;
323 
324         if(!clickedSpriteFrame._textureLoaded) {
325             this._clickedLoader.clear();
326             this._clickedLoader.once(clickedSpriteFrame, function () {
327                 this.loadTexturePressed(this._clickedFileName, this._pressedTexType);
328             }, this);
329             return;
330         }
331 
332         this._buttonClickedSpriteFrame = clickedSpriteFrame;
333         this._updateChildrenDisplayedRGBA();
334 
335         this._pressedTextureLoaded = true;
336     },

391     /**
392      * Sets capinsets for button, if button is using scale9 renderer.
393      * @param {cc.Rect} capInsets
394      */
395     setCapInsetsNormalRenderer: function (capInsets) {
396         if(!capInsets || !this._scale9Enabled)
397             return;
398 
399         var x = capInsets.x, y = capInsets.y;
400         var width = capInsets.width, height = capInsets.height;
401         if (this._normalTextureSize.width < width){
402             x = 0;
403             width = 0;
404         }
405         if (this._normalTextureSize.height < height){
406             y = 0;
407             height = 0;
408         }
409 
410         var locInsets = this._capInsetsNormal;
411         locInsets.x = x;
412         locInsets.y = y;
413         locInsets.width = width;
414         locInsets.height = height;
415 
416         this._capInsetsNormal = locInsets;
417         this._buttonScale9Renderer.setCapInsets(locInsets);
418     },

460     _onPressStateChangedToNormal: function () {
461         this._buttonScale9Renderer.setSpriteFrame(this._buttonNormalSpriteFrame);
462 
463         this._buttonScale9Renderer.setState( ccui.Scale9Sprite.state.NORMAL);
464 
465         if (this._pressedTextureLoaded) {
466             if (this.pressedActionEnabled){
467                 this._buttonScale9Renderer.stopAllActions();
468                 this._buttonScale9Renderer.setScale(1.0);
469 
470                 if(this._titleRenderer) {
471                     this._titleRenderer.stopAllActions();
472 
473                     if (this._unifySize){
474                         var zoomTitleAction = cc.scaleTo(ccui.Button.ZOOM_ACTION_TIME_STEP, 1, 1);
475                         this._titleRenderer.runAction(zoomTitleAction);
476                     }else{
477                         this._titleRenderer.setScaleX(1);
478                         this._titleRenderer.setScaleY(1);
479                     }
480                 }
481 
482             }
483         } else {
484             this._buttonScale9Renderer.stopAllActions();
485             this._buttonScale9Renderer.setScale(1.0);
486 
487             if (this._scale9Enabled) {
488                 this._buttonScale9Renderer.setColor(cc.color.WHITE);
489             }
490 
491             if(this._titleRenderer) {
492                 this._titleRenderer.stopAllActions();
493 
494                 this._titleRenderer.setScaleX(1);
495                 this._titleRenderer.setScaleY(1);
496             }
497         }
498     },
499 
500     _onPressStateChangedToPressed: function () {
501         this._buttonScale9Renderer.setState(ccui.Scale9Sprite.state.NORMAL);
502 
503         if (this._pressedTextureLoaded) {
504             this._buttonScale9Renderer.setSpriteFrame(this._buttonClickedSpriteFrame);
505 
506             if (this.pressedActionEnabled) {
507                 this._buttonScale9Renderer.stopAllActions();
508 
509                 var zoomAction = cc.scaleTo(ccui.Button.ZOOM_ACTION_TIME_STEP,
510                                             1.0 + this._zoomScale,
511                                             1.0 + this._zoomScale);
512                 this._buttonScale9Renderer.runAction(zoomAction);
513 
514                 if(this._titleRenderer) {
515                     this._titleRenderer.stopAllActions();
516                     this._titleRenderer.runAction(cc.scaleTo(ccui.Button.ZOOM_ACTION_TIME_STEP,
517                                                              1 + this._zoomScale,
518                                                              1 + this._zoomScale));
519                 }
520             }
521         } else {
522             this._buttonScale9Renderer.setSpriteFrame(this._buttonClickedSpriteFrame);
523 
524             this._buttonScale9Renderer.stopAllActions();
525             this._buttonScale9Renderer.setScale(1.0 + this._zoomScale, 1.0 + this._zoomScale);
526 
527             if (this._titleRenderer) {
528                 this._titleRenderer.stopAllActions();
529                 this._titleRenderer.setScaleX(1 + this._zoomScale);
530                 this._titleRenderer.setScaleY(1 + this._zoomScale);
531             }
532         }
533     },
