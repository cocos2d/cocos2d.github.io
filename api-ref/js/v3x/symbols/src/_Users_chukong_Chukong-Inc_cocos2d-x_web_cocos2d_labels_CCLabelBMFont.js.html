 33 /**
 34  * <p>cc.LabelBMFont is a subclass of cc.SpriteBatchNode.</p>
 35  *
 36  * <p>Features:<br/>
 37  * <ul><li>- Treats each character like a cc.Sprite. This means that each individual character can be:</li>
 38  * <li>- rotated</li>
 39  * <li>- scaled</li>
 40  * <li>- translated</li>
 41  * <li>- tinted</li>
 42  * <li>- change the opacity</li>
 43  * <li>- It can be used as part of a menu item.</li>
 44  * <li>- anchorPoint can be used to align the "label"</li>
 45  * <li>- Supports AngelCode text format</li></ul></p>
 46  *
 47  * <p>Limitations:<br/>
 48  * - All inner characters are using an anchorPoint of (0.5, 0.5) and it is not recommend to change it
 49  * because it might affect the rendering</p>
 50  *
 51  * <p>cc.LabelBMFont implements the protocol cc.LabelProtocol, like cc.Label and cc.LabelAtlas.<br/>
 52  * cc.LabelBMFont has the flexibility of cc.Label, the speed of cc.LabelAtlas and all the features of cc.Sprite.<br/>
 53  * If in doubt, use cc.LabelBMFont instead of cc.LabelAtlas / cc.Label.</p>
 54  *
 55  * <p>Supported editors:<br/>
 56  * http://glyphdesigner.71squared.com/ (Commercial, Mac OS X)<br/>
 57  * http://www.n4te.com/hiero/hiero.jnlp (Free, Java)<br/>
 58  * http://slick.cokeandcode.com/demos/hiero.jnlp (Free, Java)<br/>
 59  * http://www.angelcode.com/products/bmfont/ (Free, Windows only)</p>
 60  * @class
 61  * @extends cc.SpriteBatchNode
 62  *
 63  * @property {String}   string          - Content string of label
 64  * @property {Number}   textAlign       - Horizontal Alignment of label, cc.TEXT_ALIGNMENT_LEFT|cc.TEXT_ALIGNMENT_CENTER|cc.TEXT_ALIGNMENT_RIGHT
 65  * @property {Number}   boundingWidth   - Width of the bounding box of label, the real content width is limited by boundingWidth
 66  *
 67  * @param {String} str
 68  * @param {String} fntFile
 69  * @param {Number} [width=-1]
 70  * @param {Number} [alignment=cc.TEXT_ALIGNMENT_LEFT]
 71  * @param {cc.Point} [imageOffset=cc.p(0,0)]
 72  *
 73  * @example
 74  * // Example 01
 75  * var label1 = new cc.LabelBMFont("Test case", "test.fnt");
 76  *
 77  * // Example 02
 78  * var label2 = new cc.LabelBMFont("test case", "test.fnt", 200, cc.TEXT_ALIGNMENT_LEFT);
 79  *
 80  * // Example 03
 81  * var label3 = new cc.LabelBMFont("This is a \n test case", "test.fnt", 200, cc.TEXT_ALIGNMENT_LEFT, cc.p(0,0));
 82  */

211     /**
212      * init a bitmap font atlas with an initial string and the FNT file
213      * @param {String} str
214      * @param {String} fntFile
215      * @param {Number} [width=-1]
216      * @param {Number} [alignment=cc.TEXT_ALIGNMENT_LEFT]
217      * @param {cc.Point} [imageOffset=cc.p(0,0)]
218      * @return {Boolean}
219      */
220     initWithString: function (str, fntFile, width, alignment, imageOffset) {
221         var self = this, theString = str || "";
222 
223         if (self._config)
224             cc.log("cc.LabelBMFont.initWithString(): re-init is no longer supported");
225 
226         var texture;
227         if (fntFile) {
228             var newConf = cc.loader.getRes(fntFile);
229             if (!newConf) {
230                 newConf = cc.FntFrameCache[fntFile] || cc.FntFrameCache[cc.path.basename(fntFile)];
231                 if(!newConf) {
232                     cc.log("cc.LabelBMFont.initWithString(): Impossible to create font. Please check file");
233                     return false;
234                 }
235             }
236 
237             self._config = newConf;
238             self._fntFile = fntFile;
239             var spriteFrameBaseName = newConf.atlasName;
240             var spriteFrame = cc.spriteFrameCache.getSpriteFrame(spriteFrameBaseName) || cc.spriteFrameCache.getSpriteFrame(cc.path.basename(spriteFrameBaseName));
241             if(spriteFrame) {
242                 texture = spriteFrame.getTexture();
243                 this._spriteFrame = spriteFrame;
244             } else {
245                 texture = cc.textureCache.addImage(newConf.atlasName);
246             }
247             var locIsLoaded = texture.isLoaded();
248             self._textureLoaded = locIsLoaded;
249             if (!locIsLoaded) {
250                 texture.addEventListener("load", function (sender) {
251                     var self1 = this;
252                     self1._textureLoaded = true;
253                     //reset the LabelBMFont
254                     self1.initWithTexture(sender, self1._initialString.length);
255                     self1.setString(self1._initialString, true);
256                     self1.dispatchEvent("load");
257                 }, self);
258             }
259         } else {
260             texture = new cc.Texture2D();
261             var image = new Image();
262             texture.initWithElement(image);
263             self._textureLoaded = false;
264         }
265 
266         if (self.initWithTexture(texture, theString.length)) {
267             self._alignment = alignment || cc.TEXT_ALIGNMENT_LEFT;
268             self._imageOffset = imageOffset || cc.p(0, 0);
269             self._width = (width === undefined) ? -1 : width;
270 
271             self._realOpacity = 255;
272             self._realColor = cc.color(255, 255, 255, 255);
273 
274             self._contentSize.width = 0;
275             self._contentSize.height = 0;
276 
277             self.setAnchorPoint(0.5, 0.5);
278 
279             self.setString(theString, true);
280 
281             return true;
282         }
283         return false;
284     },

286     /**
287      * updates the font chars based on the string to render
288      */
289     createFontChars: function () {
290         var locStr = this._string;
291         var stringLen = locStr ? locStr.length : 0;
292 
293         var self = this;
294         var cmd = this._renderCmd;
295         var locTexture = cmd._texture || this._texture;
296 
297         var nextFontPositionX = 0;
298 
299         var tmpSize = cc.size(0, 0);
300 
301         var longestLine = 0;
302 
303         var quantityOfLines = 1;
304 
305 
306         var i, locCfg = self._config, locKerningDict = locCfg.kerningDict,
307             locCommonH = locCfg.commonHeight, locFontDict = locCfg.fontDefDictionary;
308         for (i = 0; i < stringLen - 1; i++) {
309             if (locStr.charCodeAt(i) === 10) quantityOfLines++;
310         }
311 
312         var totalHeight = locCommonH * quantityOfLines;
313         var nextFontPositionY = -(locCommonH - locCommonH * quantityOfLines);
314 
315         var prev = -1;
316         var fontDef;
317         for (i = 0; i < stringLen; i++) {
318             var key = locStr.charCodeAt(i);
319             if (key === 0) continue;
320 
321             if (key === 10) {
322                 //new line
323                 nextFontPositionX = 0;
324                 nextFontPositionY -= locCfg.commonHeight;
325                 continue;
326             }
327 
328             var kerningAmount = locKerningDict[(prev << 16) | (key & 0xffff)] || 0;
329             fontDef = locFontDict[key];
330             if (!fontDef) {
331                 cc.log("cocos2d: LabelBMFont: character not found " + locStr[i]);
332 
333                 fontDef = {
334                     rect: {
335                         x: 0,
336                         y: 0,
337                         width: 0,
338                         height: 0
339                     },
340                     xOffset: 0,
341                     yOffset: 0,
342                     xAdvance: 0
343                 };
344             }
345 
346             var rect = cc.rect(fontDef.rect.x, fontDef.rect.y, fontDef.rect.width, fontDef.rect.height);
347             rect = cc.rectPixelsToPoints(rect);
348             rect.x += self._imageOffset.x;
349             rect.y += self._imageOffset.y;
350 
351             var isRotated = false;
352             if(this._spriteFrame) {
353                 var textureWidth = locTexture.width;
354                 var spriteFrameRect = this._spriteFrame._rect;
355                 if(!this._spriteFrame._rotated) {
356                     rect.x = rect.x + spriteFrameRect.x;
357                     rect.y = rect.y + spriteFrameRect.y;
358                 } else {
359                     isRotated = true;
360                     var originalX = rect.x;
361                     rect.x = rect.y + spriteFrameRect.x;
362                     rect.y = originalX + spriteFrameRect.y;
363                 }
364             }
365 
366             var fontChar = self.getChildByTag(i);
367 
368             if (!fontChar) {
369                 fontChar = new cc.Sprite();
370                 fontChar.initWithTexture(locTexture, rect, isRotated);
371                 fontChar._newTextureWhenChangeColor = true;
372                 this.addChild(fontChar, 0, i);
373             } else {
374                 cmd._updateCharTexture(fontChar, rect, key, isRotated);
375             }
376 
377             // Apply label properties
378             fontChar.opacityModifyRGB = this._opacityModifyRGB;
379             cmd._updateCharColorAndOpacity(fontChar);
380 
381             var yOffset = locCfg.commonHeight - fontDef.yOffset;
382             var fontPos = cc.p(nextFontPositionX + fontDef.xOffset + fontDef.rect.width * 0.5 + kerningAmount,
383                 nextFontPositionY + yOffset - rect.height * 0.5 * cc.contentScaleFactor());
384             fontChar.setPosition(cc.pointPixelsToPoints(fontPos));
385 
386             // update kerning
387             nextFontPositionX += fontDef.xAdvance + kerningAmount;
388             prev = key;
389 
390             if (longestLine < nextFontPositionX)
391                 longestLine = nextFontPositionX;
392         }
393 
394         //If the last character processed has an xAdvance which is less that the width of the characters image, then we need
395         // to adjust the width of the string to take this into account, or the character will overlap the end of the bounding box
396         if (fontDef && fontDef.xAdvance < fontDef.rect.width)
397             tmpSize.width = longestLine - fontDef.xAdvance + fontDef.rect.width;
398         else
399             tmpSize.width = longestLine;
400         tmpSize.height = totalHeight;
401         self.setContentSize(cc.sizePixelsToPoints(tmpSize));
402     },

467     // calc the text all with in a line
468     _getCharsWidth: function (startIndex, endIndex) {
469         if (endIndex <= 0) {
470             return 0;
471         }
472         var curTextFirstSprite = this.getChildByTag(startIndex);
473         var curTextLastSprite = this.getChildByTag(startIndex + endIndex);
474         return this._getLetterPosXLeft(curTextLastSprite) - this._getLetterPosXLeft(curTextFirstSprite);
475     },

477     _checkWarp: function (strArr, i, maxWidth, initStringWrapNum) {
478         var self = this;
479         var text = strArr[i];
480         var curLength = 0;
481         for (var strArrIndex = 0; strArrIndex < i; strArrIndex++) {
482             curLength += strArr[strArrIndex].length;
483         }
484 
485         curLength = curLength + i - initStringWrapNum; // add the wrap line num
486 
487         var allWidth = self._getCharsWidth(curLength, strArr[i].length - 1);
488 
489         if (allWidth > maxWidth && text.length > 1) {
490             var fuzzyLen = text.length * ( maxWidth / allWidth ) | 0;
491             var tmpText = text.substr(fuzzyLen);
492             var width = allWidth - this._getCharsWidth(curLength + fuzzyLen, tmpText.length - 1);
493             var sLine;
494             var pushNum = 0;
495 
496             //Increased while cycle maximum ceiling. default 100 time
497             var checkWhile = 0;
498 
499             //Exceeded the size
500             while (width > maxWidth && checkWhile++ < 100) {
501                 fuzzyLen *= maxWidth / width;
502                 fuzzyLen = fuzzyLen | 0;
503                 tmpText = text.substr(fuzzyLen);
504                 width = allWidth - this._getCharsWidth(curLength + fuzzyLen, tmpText.length - 1);
505             }
506 
507             checkWhile = 0;
508 
509             //Find the truncation point
510             while (width < maxWidth && checkWhile++ < 100) {
511                 if (tmpText) {
512                     var exec = cc.LabelTTF._wordRex.exec(tmpText);
513                     pushNum = exec ? exec[0].length : 1;
514                     sLine = tmpText;
515                 }
516                 if (self._lineBreakWithoutSpaces) {
517                     pushNum = 0;
518                 }
519                 fuzzyLen = fuzzyLen + pushNum;
520                 tmpText = text.substr(fuzzyLen);
521                 width = allWidth - this._getCharsWidth(curLength + fuzzyLen, tmpText.length - 1);
522             }
523 
524             fuzzyLen -= pushNum;
525             if (fuzzyLen === 0) {
526                 fuzzyLen = 1;
527                 sLine = sLine.substr(1);
528             }
529 
530             var sText = text.substr(0, fuzzyLen), result;
531 
532             //symbol in the first
533             if (cc.LabelTTF.wrapInspection) {
534                 if (cc.LabelTTF._symbolRex.test(sLine || tmpText)) {
535                     result = cc.LabelTTF._lastWordRex.exec(sText);
536                     pushNum = result ? result[0].length : 0;
537                     if (self._lineBreakWithoutSpaces) {
538                         pushNum = 0;
539                     }
540                     fuzzyLen -= pushNum;
541 
542                     sLine = text.substr(fuzzyLen);
543                     sText = text.substr(0, fuzzyLen);
544                 }
545             }
546 
547             //To judge whether a English words are truncated
548             if (cc.LabelTTF._firsrEnglish.test(sLine)) {
549                 result = cc.LabelTTF._lastEnglish.exec(sText);
550                 if (result && sText !== result[0]) {
551                     pushNum = result[0].length;
552                     if (self._lineBreakWithoutSpaces) {
553                         pushNum = 0;
554                     }
555                     fuzzyLen -= pushNum;
556                     sLine = text.substr(fuzzyLen);
557                     sText = text.substr(0, fuzzyLen);
558                 }
559             }
560             strArr[i] = sLine || tmpText;
561             strArr.splice(i, 0, sText);
562         }
563     },

565     /**
566      * Update Label. <br />
567      * Update this Label display string and more...
568      */
569     updateLabel: function () {
570         var self = this;
571         self.string = self._initialString;
572 
573         var i, j, characterSprite;
574         // process string
575         // Step 1: Make multiline
576         if (self._width > 0) {
577             var stringArr = self.string.split('\n');
578             var wrapString = "";
579             var newWrapNum = 0;
580             var oldArrLength = 0;
581             for (i = 0; i < stringArr.length; i++) {
582                 oldArrLength = stringArr.length;
583                 this._checkWarp(stringArr, i, self._width * this._scaleX, newWrapNum);
584                 if (oldArrLength < stringArr.length) {
585                     newWrapNum++;
586                 }
587                 if (i > 0) {
588                     wrapString += "\n";
589                 }
590                 wrapString += stringArr[i];
591             }
592             wrapString = wrapString + String.fromCharCode(0);
593             self._setString(wrapString, false);
594         }
595         // Step 2: Make alignment
596         if (self._alignment !== cc.TEXT_ALIGNMENT_LEFT) {
597             i = 0;
598 
599             var lineNumber = 0;
600             var strlen = self._string.length;
601             var last_line = [];
602 
603             for (var ctr = 0; ctr < strlen; ctr++) {
604                 if (self._string[ctr].charCodeAt(0) === 10 || self._string[ctr].charCodeAt(0) === 0) {
605                     var lineWidth = 0;
606                     var line_length = last_line.length;
607                     // if last line is empty we must just increase lineNumber and work with next line
608                     if (line_length === 0) {
609                         lineNumber++;
610                         continue;
611                     }
612                     var index = i + line_length - 1 + lineNumber;
613                     if (index < 0) continue;
614 
615                     var lastChar = self.getChildByTag(index);
616                     if (lastChar == null)
617                         continue;
618                     lineWidth = lastChar.getPositionX() + lastChar._getWidth() / 2;
619 
620                     var shift = 0;
621                     switch (self._alignment) {
622                         case cc.TEXT_ALIGNMENT_CENTER:
623                             shift = self.width / 2 - lineWidth / 2;
624                             break;
625                         case cc.TEXT_ALIGNMENT_RIGHT:
626                             shift = self.width - lineWidth;
627                             break;
628                         default:
629                             break;
630                     }
631 
632                     if (shift !== 0) {
633                         for (j = 0; j < line_length; j++) {
634                             index = i + j + lineNumber;
635                             if (index < 0) continue;
636                             characterSprite = self.getChildByTag(index);
637                             if (characterSprite)
638                                 characterSprite.x += shift;
639                         }
640                     }
641 
642                     i += line_length;
643                     lineNumber++;
644 
645                     last_line.length = 0;
646                     continue;
647                 }
648                 last_line.push(self._string[i]);
649             }
650         }
651     },

833         var len = str.length;
834 
835         if (len <= 0)
836             return;
837 
838         var last_index = len - 1;
839 
840         // Only start trimming if the last character is whitespace..
841         if (this._isspace_unicode(str[last_index])) {
842             for (var i = last_index - 1; i >= 0; --i) {
843                 if (this._isspace_unicode(str[i])) {
844                     last_index = i;
845                 }
846                 else {
847                     break;
848                 }
849             }
850             this._utf8_trim_from(str, last_index);
851         }

898 var _fntLoader = {
899     FNT_HEAD: /fntframes [^\n]*(\n|$)/gi,
900     FNT_FRAME_NAME: /fntframe [^\n]*(\n|$)/gi,
901     INFO_EXP: /info [^\n]*(\n|$)/gi,
902     COMMON_EXP: /common [^\n]*(\n|$)/gi,
903     PAGE_EXP: /page [^\n]*(\n|$)/gi,
904     CHAR_EXP: /char [^\n]*(\n|$)/gi,
905     KERNING_EXP: /kerning [^\n]*(\n|$)/gi,
906     ITEM_EXP: /\w+=[^ \r\n]+/gi,
907     INT_EXP: /^[\-]?\d+$/,
908 
909     _parseStrToObj: function (str) {
910         var arr = str.match(this.ITEM_EXP);
911         var obj = {};
912         if (arr) {
913             for (var i = 0, li = arr.length; i < li; i++) {
914                 var tempStr = arr[i];
915                 var index = tempStr.indexOf("=");
916                 var key = tempStr.substring(0, index);
917                 var value = tempStr.substring(index + 1);
918                 if (value.match(this.INT_EXP)) value = parseInt(value);
919                 else if (value[0] === '"') value = value.substring(1, value.length - 1);
920                 obj[key] = value;
921             }
922         }
923         return obj;
924     },
925 
926     _parseFntContent: function (fnt, fntStr, url, useAtlas) {
927         var self = this;
928         //common
929         var commonObj = self._parseStrToObj(fntStr.match(self.COMMON_EXP)[0]);
930         fnt.commonHeight = commonObj["lineHeight"];
931         if (cc._renderType === cc.game.RENDER_TYPE_WEBGL) {
932             var texSize = cc.configuration.getMaxTextureSize();
933             if (commonObj["scaleW"] > texSize.width || commonObj["scaleH"] > texSize.height)
934                 cc.log("cc.LabelBMFont._parseCommonArguments(): page can't be larger than supported");
935         }
936         if (commonObj["pages"] !== 1) cc.log("cc.LabelBMFont._parseCommonArguments(): only supports 1 page");
937 
938         //page
939         var pageObj = self._parseStrToObj(fntStr.match(self.PAGE_EXP)[0]);
940         if (pageObj["id"] !== 0) cc.log("cc.LabelBMFont._parseImageFileName() : file could not be found");
941         if(!useAtlas) {
942             fnt.atlasName = cc.path.changeBasename(url, pageObj["file"]);
943         } else {
944             fnt.atlasName = cc.path.join(cc.path.dirname(useAtlas.path) + pageObj["file"]);
945         }
946 
947         //char
948         var charLines = fntStr.match(self.CHAR_EXP);
949         var fontDefDictionary = fnt.fontDefDictionary = {};
950         for (var i = 0, li = charLines.length; i < li; i++) {
951             var charObj = self._parseStrToObj(charLines[i]);
952             var charId = charObj["id"];
953             fontDefDictionary[charId] = {
954                 rect: {x: charObj["x"], y: charObj["y"], width: charObj["width"], height: charObj["height"]},
955                 xOffset: charObj["xoffset"],
956                 yOffset: charObj["yoffset"],
957                 xAdvance: charObj["xadvance"]
958             };
959         }
960 
961         //kerning
962         var kerningDict = fnt.kerningDict = {};
963         var kerningLines = fntStr.match(self.KERNING_EXP);
964         if (kerningLines) {
965             for (var i = 0, li = kerningLines.length; i < li; i++) {
966                 var kerningObj = self._parseStrToObj(kerningLines[i]);
967                 kerningDict[(kerningObj["first"] << 16) | (kerningObj["second"] & 0xffff)] = kerningObj["amount"];
968             }
969         }
970 
971         return fnt;
972     },

974     /**
975      * Parse Fnt string.
976      * @param fntStr
977      * @param url
978      * @returns {{}}
979      */
980     parseFnt: function (fntStr, url) {
981         var self = this, fnt = {};
982         var headString = fntStr.match(self.FNT_HEAD);
983         if(headString) {
984             var headObj = self._parseStrToObj(headString[0]);
985             if(headObj && headObj.count) {
986                 fntStr = fntStr.substr(headString[0].length);
987                 var fntFrames = fntStr.split("----");
988                 for(var i = 0; i < headObj.count; ++i) {
989                     var contentString = fntFrames[i];
990                     var frameNameStr = contentString.match(self.FNT_FRAME_NAME);
991                     if(frameNameStr) {
992                         var frameName = self._parseStrToObj(frameNameStr[0]);
993                         if(frameName && frameName.name) {
994                             fnt = {};
995                             var realFntPathKey = cc.path.join(cc.path.dirname(url), frameName.name);
996                             cc.FntFrameCache[realFntPathKey] = this._parseFntContent(fnt, contentString.substr(frameNameStr[0].length), url, {path: frameName.name});
997                         }
998                     }
999                 }
1000             }
1001         } else {
1002             fnt = this._parseFntContent(fnt, fntStr, url);
1003         }
1004         return fnt;
1005     },
