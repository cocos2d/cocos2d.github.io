118 /**
119  * ActionTimeline
120  * @class
121  * @extend cc.Action
122  *
123  * @property gotoFrameAndPlay
124  * @property gotoFrameAndPause
125  */
126 ccs.ActionTimeline = cc.Action.extend({
127 
128     _timelineMap: null,
129     _timelineList: null,
130     _duration: 0,
131     _time: null,
132     _timeSpeed: 1,
133     _frameInternal: 1 / 60,
134     _playing: false,
135     _currentFrame: 0,
136     _startFrame: 0,
137     _endFrame: 0,
138     _loop: null,
139     _frameEventListener: null,
140     _animationInfos: null,
141     _lastFrameListener: null,
142 
143     ctor: function () {
144         cc.Action.prototype.ctor.call(this);
145         this._timelineMap = {};
146         this._timelineList = [];
147         this._animationInfos = {};
148         this.init();
149     },
150 
151     _gotoFrame: function (frameIndex) {
152         var size = this._timelineList.length;
153         for (var i = 0; i < size; i++) {
154             this._timelineList[i]._gotoFrame(frameIndex);
155         }
156     },
157 
158     _stepToFrame: function (frameIndex) {
159         var size = this._timelineList.length;
160         for (var i = 0; i < size; i++) {
161             this._timelineList[i]._stepToFrame(frameIndex);
162         }
163     },
164 
165     //emit frame event, call it when enter a frame
166     _emitFrameEvent: function (frame) {
167         if (this._frameEventListener) {
168             this._frameEventListener(frame);
169         }
170     },
171 
172     init: function () {
173         return true;
174     },
175 
176     /**
177      * Goto the specified frame index, and start playing from this index.
178      * @param startIndex The animation will play from this index.
179      * @param [endIndex=] The animation will end at this index.
180      * @param [currentFrameIndex=] set current frame index.
181      * @param [loop=] Whether or not the animation need loop.
182      */
183     gotoFrameAndPlay: function (startIndex, endIndex, currentFrameIndex, loop) {
184         //Consolidation parameters
185         var i = 0,
186             argLen = arguments.length;
187         var num = [],
188             bool;
189         for (i; i < argLen; i++) {
190             if (typeof arguments[i] === "boolean") {
191                 bool = arguments[i];
192             } else {
193                 num.push(arguments[i]);
194             }
195         }
196         startIndex = num[0];
197         endIndex = num[1] !== undefined ? num[1] : this._duration;
198         currentFrameIndex = num[2] || startIndex;
199         loop = bool != null ? bool : true;
200 
201         this._startFrame = startIndex;
202         this._endFrame = endIndex;
203         this._currentFrame = currentFrameIndex;
204         this._loop = loop;
205         this._time = this._currentFrame * this._frameInternal;
206 
207         this.resume();
208         this._gotoFrame(this._currentFrame);
209     },
210 
211     /**
212      * Goto the specified frame index, and pause at this index.
213      * @param startIndex The animation will pause at this index.
214      */
215     gotoFrameAndPause: function (startIndex) {
216         this._startFrame = this._currentFrame = startIndex;
217         this._time = this._currentFrame * this._frameInternal;
218 
219         this.pause();
220         this._gotoFrame(this._currentFrame);
221     },
222 
223     /**
224      * Pause the animation.
225      */
226     pause: function () {
227         this._playing = false;
228     },
229 
230     /**
231      * Resume the animation.
232      */
233     resume: function () {
234         this._playing = true;
235     },
236 
237     /**
238      * Whether or not Action is playing.
239      */
240     isPlaying: function () {
241         return this._playing;
242     },
243 
244     /**
245      * Set the animation speed, this will speed up or slow down the speed.
246      * @param {number} speed
247      */
248     setTimeSpeed: function (speed) {
249         this._timeSpeed = speed;
250     },
251 
252     /**
253      * Get current animation speed.
254      * @returns {number}
255      */
256     getTimeSpeed: function () {
257         return this._timeSpeed;
258     },
259 
260     /**
261      * duration of the whole action
262      * @param {number} duration
263      */
264     setDuration: function (duration) {
265         this._duration = duration;
266     },
267 
268     /**
269      * Get current animation duration.
270      * @returns {number}
271      */
272     getDuration: function () {
273         return this._duration;
274     },
275 
276     /**
277      * Start frame index of this action
278      * @returns {number}
279      */
280     getStartFrame: function () {
281         return this._startFrame;
282     },
283 
284     /**
285      * End frame of this action.
286      * When action play to this frame, if action is not loop, then it will stop,
287      * or it will play from start frame again.
288      * @returns {number}
289      */
290     getEndFrame: function () {
291         return this._endFrame;
292     },
293 
294     /**
295      * Set current frame index, this will cause action plays to this frame.
296      */
297     setCurrentFrame: function (frameIndex) {
298         if (frameIndex >= this._startFrame && frameIndex <= this._endFrame) {
299             this._currentFrame = frameIndex;
300             this._time = this._currentFrame * this._frameInternal;
301         } else {
302             cc.log("frame index is not between start frame and end frame");
303         }
304 
305     },
306 
307     /**
308      * Get current frame.
309      * @returns {number}
310      */
311     getCurrentFrame: function () {
312         return this._currentFrame;
313     },
314 
315     /**
316      * add Timeline to ActionTimeline
317      * @param {ccs.Timeline} timeline
318      */
319     addTimeline: function (timeline) {
320         var tag = timeline.getActionTag();
321         if (!this._timelineMap[tag]) {
322             this._timelineMap[tag] = [];
323         }
324 
325         if (this._timelineMap[tag].indexOf(timeline) === -1) {
326             this._timelineList.push(timeline);
327             this._timelineMap[tag].push(timeline);
328             timeline.setActionTimeline(this);
329         }
330 
331     },
332 
333     /**
334      * remove Timeline to ActionTimeline
335      * @param {ccs.Timeline} timeline
336      */
337     removeTimeline: function (timeline) {
338         var tag = timeline.getActionTag();
339         if (this._timelineMap[tag]) {
340             if (this._timelineMap[tag].some(function (item) {
341                     if (item === timeline)
342                         return true;
343                 })) {
344                 cc.arrayRemoveObject(this._timelineMap[tag], timeline);
345                 cc.arrayRemoveObject(this._timelineList, timeline);
346                 timeline.setActionTimeline(null);
347             }
348         }
349     },
350 
351     /**
352      * Gets the timeline list
353      * @returns {array | null}
354      */
355     getTimelines: function () {
356         return this._timelineList;
357     },
358 
359     /**
360      * Set the Frame event
361      * @param {function} listener
362      */
363     setFrameEventCallFunc: function (listener) {
364         this._frameEventListener = listener;
365     },
366 
367     /**
368      * remove event
369      */
370     clearFrameEventCallFunc: function () {
371         this._frameEventListener = null;
372     },
373 
374     /**
375      * Clone this timeline
376      * @returns {ccs.ActionTimeline}
377      */
378     clone: function () {
379         var newAction = new ccs.ActionTimeline();
380         newAction.setDuration(this._duration);
381         newAction.setTimeSpeed(this._timeSpeed);
382 
383         for (var a in this._timelineMap) {
384             var timelines = this._timelineMap[a];
385             for (var b in timelines) {
386                 var timeline = timelines[b];
387                 var newTimeline = timeline.clone();
388                 newAction.addTimeline(newTimeline);
389             }
390         }
391 
392         return newAction;
393 
394     },
395 
396     /**
397      * Reverse is not defined;
398      * @returns {null}
399      */
400     reverse: function () {
401         return null;
402     },
403 
404     /**
405      * Stepping of this time line.
406      * @param {number} delta
407      */
408     step: function (delta) {
409         if (!this._playing || this._timelineMap.length === 0 || this._duration === 0) {
410             return;
411         }
412 
413         this._time += delta * this._timeSpeed;
414         var endoffset = this._time - this._endFrame * this._frameInternal;
415 
416         if (endoffset < this._frameInternal) {
417             this._currentFrame = Math.floor(this._time / this._frameInternal);
418             this._stepToFrame(this._currentFrame);
419             if (endoffset >= 0 && this._lastFrameListener)
420                 this._lastFrameListener();
421         } else {
422             this._playing = this._loop;
423             if (!this._playing) {
424                 this._time = this._endFrame * this._frameInternal;
425                 if (this._currentFrame != this._endFrame) {
426                     this._currentFrame = this._endFrame;
427                     this._stepToFrame(this._currentFrame);
428                     if (this._lastFrameListener)
429                         this._lastFrameListener();
430                 }
431             } else
432                 this.gotoFrameAndPlay(this._startFrame, this._endFrame, this._loop);
433         }
434 
435     },
436 
437     _foreachNodeDescendant: function (parent, callback) {
438         callback(parent);
439 
440         var children = parent.getChildren();
441         for (var i = 0; i < children.length; i++) {
442             var child = children[i];
443             this._foreachNodeDescendant(child, callback);
444         }
445     },
446 
447     /**
448      * start with node.
449      * @param {cc.Node} target
450      */
451     startWithTarget: function(target){
452         cc.Action.prototype.startWithTarget.call(this, target);
453 
454         var self = this;
455         var callback = function(child){
456             var data = child.getComponent("ComExtensionData");
457 
458             if(data) {
459                 var actionTag = data.getActionTag();
460                 if(self._timelineMap[actionTag]) {
461                     var timelines = self._timelineMap[actionTag];
462                     for (var i=0; i<timelines.length; i++) {
463                         var timeline = timelines[i];
464                         timeline.setNode(child);
465                     }
466                 }
467             }
468         };
469 
470         this._foreachNodeDescendant(target, callback);
471     },
472 
473     /**
474      * Whether or not complete
475      * @returns {boolean}
476      */
477     isDone: function () {
478         return false;
479     },
480 
481     /**
482      * @param {String} name
483      * @param {Boolean} loop
484      */
485     play: function (name, loop) {
486         var info = this._animationInfos[name];
487         if (!info)
488             return cc.log("Can't find animation info for %s", name);
489 
490         this.gotoFrameAndPlay(info.startIndex, info.endIndex, loop);
491     },
492 
493     /**
494      * Add animationInfo
495      * @param {Object} info
496      */
497     addAnimationInfo: function (info) {
498         this._animationInfos[info.name] = info;
499     },
500 
501     /**
502      * Remove animationInfo
503      * @param {String} name
504      */
505     removeAnimationInfo: function (name) {
506         delete this._animationInfos[name];
507     },
508 
509     isAnimationInfoExists: function (name) {
510         return this._animationInfos[name];
511     },
512 
513     getAnimationInfo: function (name) {
514         return this._animationInfos[name];
515     },
516 
517     setLastFrameCallFunc: function (listener) {
518         this._lastFrameListener = listener;
519     },
520 
521     clearLastFrameCallFunc: function () {
522         this._lastFrameListener = null;
523     }
524 });
