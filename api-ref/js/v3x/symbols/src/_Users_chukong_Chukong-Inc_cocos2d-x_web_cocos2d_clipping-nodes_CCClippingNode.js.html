 28 /**
 29  * the value of stencil bits.
 30  * @type Number
 31  */
 32 cc.stencilBits = -1;
 33 
 34 /**
 35  * <p>
 36  *     cc.ClippingNode is a subclass of cc.Node.                                                            <br/>
 37  *     It draws its content (children) clipped using a stencil.                                               <br/>
 38  *     The stencil is an other cc.Node that will not be drawn.                                               <br/>
 39  *     The clipping is done using the alpha part of the stencil (adjusted with an alphaThreshold).
 40  * </p>
 41  * @class
 42  * @extends cc.Node
 43  * @param {cc.Node} [stencil=null]
 44  *
 45  * @property {Number}   alphaThreshold  - Threshold for alpha value.
 46  * @property {Boolean}  inverted        - Indicate whether in inverted mode.
 47  * @property {cc.Node}  stencil         - he cc.Node to use as a stencil to do the clipping.
 48  */
 49 cc.ClippingNode = cc.Node.extend(/** @lends cc.ClippingNode# */{
 50     inverted: false,
 51     _alphaThreshold: 0,
 52 
 53     _stencil: null,
 54     _className: "ClippingNode",
 55 
 56     _originStencilProgram: null,
 57 
 58     /**
 59      * Constructor function, override it to extend the construction behavior, remember to call "this._super()" in the extended "ctor" function.
 60      * @param {cc.Node} [stencil=null]
 61      */
 62     ctor: function (stencil) {
 63         stencil = stencil || null;
 64         cc.Node.prototype.ctor.call(this);
 65         this._stencil = stencil;
 66         if (stencil) {
 67             this._originStencilProgram = stencil.getShaderProgram();
 68         }
 69         this.alphaThreshold = 1;
 70         this.inverted = false;
 71         this._renderCmd.initStencilBits();
 72     },
 73 
 74     /**
 75      * <p>
 76      *     Event callback that is invoked every time when node enters the 'stage'.                                   <br/>
 77      *     If the CCNode enters the 'stage' with a transition, this event is called when the transition starts.        <br/>
 78      *     During onEnter you can't access a "sister/brother" node.                                                    <br/>
 79      *     If you override onEnter, you must call its parent's onEnter function with this._super().
 80      * </p>
 81      * @function
 82      */
 83     onEnter: function () {
 84         cc.Node.prototype.onEnter.call(this);
 85         if (this._stencil)
 86             this._stencil._performRecursive(cc.Node._stateCallbackType.onEnter);
 87     },
 88 
 89     /**
 90      * <p>
 91      *     Event callback that is invoked when the node enters in the 'stage'.                                                        <br/>
 92      *     If the node enters the 'stage' with a transition, this event is called when the transition finishes.                       <br/>
 93      *     If you override onEnterTransitionDidFinish, you shall call its parent's onEnterTransitionDidFinish with this._super()
 94      * </p>
 95      * @function
 96      */
 97     onEnterTransitionDidFinish: function () {
 98         cc.Node.prototype.onEnterTransitionDidFinish.call(this);
 99         if (this._stencil)
100             this._stencil._performRecursive(cc.Node._stateCallbackType.onEnterTransitionDidFinish);
101     },
102 
103     /**
104      * <p>
105      *     callback that is called every time the node leaves the 'stage'.  <br/>
106      *     If the node leaves the 'stage' with a transition, this callback is called when the transition starts. <br/>
107      *     If you override onExitTransitionDidStart, you shall call its parent's onExitTransitionDidStart with this._super()
108      * </p>
109      * @function
110      */
111     onExitTransitionDidStart: function () {
112         this._stencil._performRecursive(cc.Node._stateCallbackType.onExitTransitionDidStart);
113         cc.Node.prototype.onExitTransitionDidStart.call(this);
114     },
115 
116     /**
117      * <p>
118      * callback that is called every time the node leaves the 'stage'. <br/>
119      * If the node leaves the 'stage' with a transition, this callback is called when the transition finishes. <br/>
120      * During onExit you can't access a sibling node.                                                             <br/>
121      * If you override onExit, you shall call its parent's onExit with this._super().
122      * </p>
123      * @function
124      */
125     onExit: function () {
126         this._stencil._performRecursive(cc.Node._stateCallbackType.onExit);
127         cc.Node.prototype.onExit.call(this);
128     },
129 
130     visit: function (parent) {
131         if (!this._visible)
132             return;
133 
134         this._renderCmd.clippingVisit(parent && parent._renderCmd);
135     },
136 
137     _visitChildren: function () {
138         var renderer = cc.renderer;
139         if (this._reorderChildDirty) {
140             this.sortAllChildren();
141         }
142         var children = this._children, child;
143         for (var i = 0, len = children.length; i < len; i++) {
144             child = children[i];
145             if (child && child._visible) {
146                 child.visit(this);
147             }
148         }
149         this._renderCmd._dirtyFlag = 0;
150     },
151 
152     /**
153      * <p>
154      * The alpha threshold.                                                                                   <br/>
155      * The content is drawn only where the stencil have pixel with alpha greater than the alphaThreshold.     <br/>
156      * Should be a float between 0 and 1.                                                                     <br/>
157      * This default to 1 (so alpha test is disabled).
158      * </P>
159      * @return {Number}
160      */
161     getAlphaThreshold: function () {
162         return this._alphaThreshold;
163     },
164 
165     /**
166      * set alpha threshold.
167      * @param {Number} alphaThreshold
168      */
169     setAlphaThreshold: function (alphaThreshold) {
170         if (alphaThreshold === 1 && alphaThreshold !== this._alphaThreshold) {
171             // should reset program used by _stencil
172             this._renderCmd.resetProgramByStencil();
173         }
174         this._alphaThreshold = alphaThreshold;
175     },
176 
177     /**
178      * <p>
179      *     Inverted. If this is set to YES,                                                                 <br/>
180      *     the stencil is inverted, so the content is drawn where the stencil is NOT drawn.                 <br/>
181      *     This default to NO.
182      * </p>
183      * @return {Boolean}
184      */
185     isInverted: function () {
186         return this.inverted;
187     },
188 
189     /**
190      * set whether or not invert of stencil
191      * @param {Boolean} inverted
192      */
193     setInverted: function (inverted) {
194         this.inverted = inverted;
195     },
196 
197     /**
198      * The cc.Node to use as a stencil to do the clipping.                                   <br/>
199      * The stencil node will be retained. This default to nil.
200      * @return {cc.Node}
201      */
202     getStencil: function () {
203         return this._stencil;
204     },
205 
206     /**
207      * Set stencil.
208      * @function
209      * @param {cc.Node} stencil
210      */
211     setStencil: function (stencil) {
212         if (this._stencil === stencil)
213             return;
214         if (stencil)
215             this._originStencilProgram = stencil.getShaderProgram();
216         this._renderCmd.setStencil(stencil);
217     },
218 
219     _createRenderCmd: function () {
220         if (cc._renderType === cc.game.RENDER_TYPE_CANVAS)
221             return new cc.ClippingNode.CanvasRenderCmd(this);
222         else
223             return new cc.ClippingNode.WebGLRenderCmd(this);
224     }
225 });
226 
227 var _p = cc.ClippingNode.prototype;

229 // Extended properties
230 /** @expose */
231 _p.stencil;
232 cc.defineGetterSetter(_p, "stencil", _p.getStencil, _p.setStencil);
233 /** @expose */
234 _p.alphaThreshold;
235 cc.defineGetterSetter(_p, "alphaThreshold", _p.getAlphaThreshold, _p.setAlphaThreshold);

239  * Creates and initializes a clipping node with an other node as its stencil. <br/>
240  * The stencil node will be retained.

242  * @param {cc.Node} [stencil=null]
243  * @return {cc.ClippingNode}
244  * @example
245  * //example
246  * new cc.ClippingNode(stencil);
247  */
248 cc.ClippingNode.create = function (stencil) {
249     return new cc.ClippingNode(stencil);
