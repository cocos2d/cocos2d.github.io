226     _refreshIndicatorPosition: function () {
227         if (this._indicator) {
228             var contentSize = this.getContentSize();
229             var posX = contentSize.width * this._indicatorPositionAsAnchorPoint.x;
230             var posY = contentSize.height * this._indicatorPositionAsAnchorPoint.y;
231             this._indicator.setPosition(cc.p(posX, posY));
232         }
233     },

235     _handleReleaseLogic: function (touchPoint) {
236 
237         ccui.ScrollView.prototype._handleReleaseLogic.call(this, touchPoint);
238 
239         if (this._items.length <= 0)
240             return;
241 
242         var touchMoveVelocity = this._flattenVectorByDirection(this._calculateTouchMoveVelocity());
243 
244         var INERTIA_THRESHOLD = 500;
245         if (cc.pLength(touchMoveVelocity) < INERTIA_THRESHOLD) {
246             this._startMagneticScroll();
247         }
248         else {
249             // Handle paging by inertia force.
250             var currentPage = this.getItem(this._curPageIdx);
251             var destination = this._calculateItemDestination(cc.p(0.5, 0.5), currentPage, cc.p(0.5, 0.5));
252             var deltaToCurrentPage = cc.pSub(destination, this.getInnerContainerPosition());
253             deltaToCurrentPage = this._flattenVectorByDirection(deltaToCurrentPage);
254 
255             // If the direction of displacement to current page and the direction of touch are same, just start magnetic scroll to the current page.
256             // Otherwise, move to the next page of touch direction.
257             if (touchMoveVelocity.x * deltaToCurrentPage.x > 0 || touchMoveVelocity.y * deltaToCurrentPage.y > 0) {
258                 this._startMagneticScroll();
259             }
260             else {
261                 if (touchMoveVelocity.x < 0 || touchMoveVelocity.y > 0) {
262                     ++this._curPageIdx;
263                 }
264                 else {
265                     --this._curPageIdx;
266                 }
267                 this._curPageIdx = Math.min(this._curPageIdx, this._items.length);
268                 this._curPageIdx = Math.max(this._curPageIdx, 0);
269                 this.scrollToItem(this._curPageIdx);
270             }
271         }
272

279     _pageTurningEvent: function () {
280         if (this._pageViewEventSelector) {
281             if (this._pageViewEventListener)
282                 this._pageViewEventSelector.call(this._pageViewEventListener, this, ccui.PageView.EVENT_TURNING);
283             else
284                 this._pageViewEventSelector(this, ccui.PageView.EVENT_TURNING);
285         }
286         if (this._ccEventCallback)
287             this._ccEventCallback(this, ccui.PageView.EVENT_TURNING);
288     },

391     /**
392      * Toggle page indicator enabled.
393      * @param {boolean} enabled True if enable page indicator, false otherwise.
394      */
395     setIndicatorEnabled: function (enabled) {
396         if (enabled == (this._indicator !== null)) {
397             return;
398         }
399 
400         if (!enabled) {
401             this.removeProtectedChild(this._indicator);
402             this._indicator = null;
403         }
404         else {
405             this._indicator = new ccui.PageViewIndicator();
406             this._indicator.setDirection(this.getDirection());
407             this.addProtectedChild(this._indicator, 10000);
408             this.setIndicatorSelectedIndexColor(cc.color(100, 100, 255));
409             this._refreshIndicatorPosition();
410         }
411     },
412
