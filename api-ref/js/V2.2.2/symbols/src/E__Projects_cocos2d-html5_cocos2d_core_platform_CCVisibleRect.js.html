 28 /**
 29  *
 30  * @type Object
 31  */
 32 cc.VisibleRect = {
 33     _topLeft:cc._pConst(0,0),
 34     _topRight:cc._pConst(0,0),
 35     _top:cc._pConst(0,0),
 36     _bottomLeft:cc._pConst(0,0),
 37     _bottomRight:cc._pConst(0,0),
 38     _bottom:cc._pConst(0,0),
 39     _center:cc._pConst(0,0),
 40     _left:cc._pConst(0,0),
 41     _right:cc._pConst(0,0),
 42     _width:0,
 43     _height:0,
 44     init:function(size){
 45         this._width = size.width;
 46         this._height = size.height;
 47 
 48         var w = this._width;
 49         var h = this._height;
 50 
 51         //top
 52         this._topLeft._y = h;
 53         this._topRight._x = w;
 54         this._topRight._y = h;
 55         this._top._x = w/2;
 56         this._top._y = h;
 57 
 58         //bottom
 59         this._bottomRight._x = w;
 60         this._bottom._x = w/2;
 61 
 62         //center
 63         this._center._x = w/2;
 64         this._center._y = h/2;
 65 
 66         //left
 67         this._left._y = h/2;
 68 
 69         //right
 70         this._right._x = w;
 71         this._right._y = h/2;
 72     },
 73     getWidth:function(){
 74         return this._width;
 75     },
 76     getHeight:function(){
 77         return this._height;
 78     },
 79     topLeft:function(){
 80         return this._topLeft;
 81     },
 82     topRight:function(){
 83         return this._topRight;
 84     },
 85     top:function(){
 86         return this._top;
 87     },
 88     bottomLeft:function(){
 89         return this._bottomLeft;
 90     },
 91     bottomRight:function(){
 92         return this._bottomRight;
 93     },
 94     bottom:function(){
 95         return this._bottom;
 96     },
 97     center:function(){
 98         return this._center;
 99     },
100     left:function(){
101         return this._left;
102     },
103     right:function(){
104         return this._right;
105     }
106 };
