 25 /**
 26  * Base class for ccs.ComRender
 27  * @class
 28  * @extends ccs.Component
 29  */
 30 ccs.ComRender = ccs.Component.extend(/** @lends ccs.ComRender# */{
 31     _render: null,
 32     ctor: function (node, comName) {
 33         cc.Component.prototype.ctor.call(this);
 34         this._render = node;
 35         this._name = comName;
 36     },
 37 
 38     onEnter: function () {
 39         if (this._owner) {
 40             this._owner.addChild(this._render);
 41         }
 42     },
 43 
 44     onExit: function () {
 45         if (this._owner) {
 46             this._owner.removeChild(this._render, true);
 47             this._render = null;
 48         }
 49     },
 50 
 51     /**
 52      * Node getter
 53      * @returns {cc.Node}
 54      */
 55     getNode: function () {
 56         return this._render;
 57     },
 58 
 59     /**
 60      * Node setter
 61      * @param {cc.Node} node
 62      */
 63     setNode: function (node) {
 64         this._render = node;
 65     }
 66 });
 67 /**
 68  * allocates and initializes a ComRender.
 69  * @constructs
 70  * @return {ccs.ComRender}
 71  * @example
 72  * // example
 73  * var com = ccs.ComRender.create();
 74  */
 75 ccs.ComRender.create = function (node, comName) {
 76     var com = new ccs.ComRender(node, comName);
 77     if (com && com.init()) {
 78         return com;
 79     }
 80     return null;
 81 };
